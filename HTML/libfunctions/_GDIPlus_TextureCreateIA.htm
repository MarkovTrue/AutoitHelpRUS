_GDIPlus_TextureCreateIA

Создаёт объект TextureBrush на основе изображения, ограничивающего прямоугольника и набора свойств изображения.

#include <GDIPlus.au3>
_GDIPlus_TextureCreateIA ( $hImage, $nX, $nY, $nWidth, $nHeight [, $pImageAttributes = 0] )

Параметры

$hImage Указатель на объект изображения.
$nX Координата X левого края фрагмента изображения, используемого кистью.
$nY Координата Y верхнего края фрагмента изображения, используемого кистью.
$nWidth Ширина кисти и фрагмента изображения, используемого кистью.
$nHeight Высота кисти и фрагмента изображения, используемого кистью.
$pImageAttributes [необязательный] Указатель на объект ImageAttributes со свойствами изображения

Возвращаемое значение

Успех: Возвращает указатель на новый объект TextureBrush.
Ошибка: Возвращает False и устанавливает @error ≠ 0.
@extended: Может содержать код ошибки GPSTATUS ($GDIP_ERR* см. GPIPlusConstants.au3).

Примечания

После завершения работы с объектом вызовите _GDIPlus_BrushDispose() для освобождения ресурсов.

См. также

_GDIPlus_BrushDispose, _GDIPlus_ImageAttributesCreate, _GDIPlus_ImageAttributesDispose

Дополнительно

Искать GdipCreateTextureIA в библиотеке MSDN

Пример

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>
#include <ScreenCapture.au3>
#include <WinAPIHObj.au3>

Global $g_hHBitmap, $g_hGfxCtxt, $g_hBMP, $g_hGfx
Global $g_iTiles = 24, $g_a_hBmp_Tiles[$g_iTiles][$g_iTiles]

Example()

Func Example()
    AutoItSetOption("GUIOnEventMode", 1)

    Local $iW = 576, $iH = 576, $iTMP, $iSin = 0, $iMulti = 256 / $g_iTiles, $2Multi = 2 * $iMulti, _
            $hCtxt, $hTextureIA, $fValue, $iFrameTimer = TimerInit()

    _GDIPlus_Startup()
    $g_hHBitmap = _ScreenCapture_Capture("", 0, 0, $g_iTiles ^ 2, $g_iTiles ^ 2) ;захватить рабочий стол и сохранить в памяти в формате GDI
    Local $hImage = _GDIPlus_BitmapCreateFromHBITMAP($g_hHBitmap) ;преобразовать GDI-битмап в GDI+-битмап
    ; Local $iImgW = _GDIPlus_ImageGetWidth($hImage) ;get the width of the image
    ; Local $iImgH = _GDIPlus_ImageGetHeight($hImage) ;get the height of the image

    Local $hGUI = GUICreate("GDI+ TextureIA", $iW, $iH) ;создать GUI
    $g_hGfx = _GDIPlus_GraphicsCreateFromHWND($hGUI) ;создать графический объект для отображения GDI+-графики
    $g_hBMP = _GDIPlus_BitmapCreateFromGraphics($iW, $iH, $g_hGfx) ;создать битмап на основе графического объекта
    $g_hGfxCtxt = _GDIPlus_ImageGetGraphicsContext($g_hBMP) ;получить графический контекст image / bitmap

    Local $hBitmap = _GDIPlus_BitmapCreateFromScan0($g_iTiles ^ 2, $g_iTiles ^ 2) ;создать пустой битмап (понадобится при создании текстуры)
    Local $hContext = _GDIPlus_ImageGetGraphicsContext($hBitmap) ;получить графический контекст изображения / битмапа
    _GDIPlus_GraphicsDrawImageRect($hContext, $hImage, 0, 0, $g_iTiles ^ 2, $g_iTiles ^ 2) ;скопировать изображение в пустой битмап
    _GDIPlus_GraphicsDispose($hContext) ;освободить дескриптор графического контекста
    _GDIPlus_ImageDispose($hImage) ;освободить дескриптор изображения

    Local $tCM = _GDIPlus_ColorMatrixCreateScale(0, 0, 0) ;create and initializes a scaling color matrix (5 x 5 homogeneous matrix for a 4-space transformation) -> see http://msdn.microsoft.com/en-us/library/windows/desktop/ms533875(v=vs.85).aspx
    Local $pImgAttr = _GDIPlus_ImageAttributesCreate() ;создать объект атрибутов изображения (для управления цветом)
    Local $fSizeInv = 1 / $g_iTiles

    ;создать $g_iTiles * $g_iTiles малых битмапов (плиток) для заполнения фрагментами захваченного изображения
    For $iY = 0 To $g_iTiles - 1
        For $iX = 0 To $g_iTiles - 1
            $g_a_hBmp_Tiles[$iY][$iX] = _GDIPlus_BitmapCreateFromScan0($g_iTiles, $g_iTiles) ;создать пустой bitmap
            $hCtxt = _GDIPlus_ImageGetGraphicsContext($g_a_hBmp_Tiles[$iY][$iX]) ;получить графический контекст image / bitmap
            $fValue = BitXOR($iX, $iY) * $fSizeInv
            $tCM.m(1) = 1 - $fValue ; красный
            $tCM.m(7) = $fValue ;зелёный
            $tCM.m(19) = 0.4 ;альфа
            _GDIPlus_ImageAttributesSetColorMatrix($pImgAttr, 0, True, $tCM) ;скорректировать цветовую матрицу ImageAttribute для каждого битмапа
            $hTextureIA = _GDIPlus_TextureCreateIA($hBitmap, $iX * $g_iTiles, $iY * $g_iTiles, $g_iTiles, $g_iTiles, $pImgAttr) ;создать текстурную кисть на основе изображения с изменёнными атрибутами
            _GDIPlus_GraphicsFillRect($hCtxt, 0, 0, $g_iTiles, $g_iTiles, $hTextureIA) ;нарисовать прямоугольник, заполненный обработанной плиткой изображения, на битмапе
            _GDIPlus_GraphicsDispose($hCtxt) ;освободить дескриптор графического контекста
            _GDIPlus_BrushDispose($hTextureIA) ;освободить текстуру
        Next
    Next
    _GDIPlus_ImageAttributesDispose($pImgAttr) ;освободить объект атрибутов изображения

    GUISetOnEvent($GUI_EVENT_CLOSE, "_Exit", $hGUI)
    GUISetState(@SW_SHOW, $hGUI)

    While Sleep(10)
        $iSin += 25 / (1000 / TimerDiff($iFrameTimer))
        _GDIPlus_GraphicsClear($g_hGfxCtxt, 0xB0808080) ;очистить графический объект, залив указанным цветом
        For $iX = 0 To $g_iTiles - 1
            $iTMP = 2 * $iX * $iMulti
            For $iY = 0 To $g_iTiles - 1
                ;переместить каждую плитку на новую позицию
                _GDIPlus_GraphicsDrawImageRect($g_hGfxCtxt, $g_a_hBmp_Tiles[$iY][$iX], _ ;дескриптор битмапа
                        25 + 15 * Sin($iY * $iMulti / 64 + $iSin) + $iTMP, _ ;позиция X
                        15 * Cos($iX * $iMulti / 64 + $iSin) + 25 + $iY * $2Multi, _ ;позиция Y
                        $2Multi, _ ;ширина
                        $2Multi) ;высота
            Next
        Next
        _GDIPlus_GraphicsDrawImage($g_hGfx, $g_hBMP, 0, 0) ;вывести готовое изображение в GUI
        $iFrameTimer = TimerInit()
    WEnd
EndFunc   ;==>Example

Func _Exit() ;освободить ресурсы
    _WinAPI_DeleteObject($g_hHBitmap)
    _GDIPlus_GraphicsDispose($g_hGfxCtxt)
    _GDIPlus_BitmapDispose($g_hBMP)
    _GDIPlus_GraphicsDispose($g_hGfx)
    For $iY = 0 To $g_iTiles - 1
        For $iX = 0 To $g_iTiles - 1
            _GDIPlus_BitmapDispose($g_a_hBmp_Tiles[$iY][$iX])
        Next
    Next
    _GDIPlus_Shutdown()
    Exit
EndFunc   ;==>_Exit
