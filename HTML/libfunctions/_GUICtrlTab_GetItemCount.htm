Справочник функций

_GUICtrlTab_GetItemCount

Получает значение.

#include <GuiTab.au3>
_GUICtrlTab_GetItemCount ( $hWnd )

Параметры

$hWnd Идентификатор или дескриптор элемента управления.

Возвращаемое значение

Возвращает количество вкладок.

Пример

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <GuiTab.au3>
#include <MsgBoxConstants.au3>

Example()

Func Example()
    ; Создать GUI
    Local $hGUI = GUICreate("Tab Get ItemCount - v(" & @AutoItVersion & ")", 450, 300, 100, 100)
    Local $idTab = GUICtrlCreateTab(2, 2, 446, 266)
    _MemoMsgBoxStatus() ; Создание статус-бара

    GUISetState(@SW_SHOW)

    ; Добавить вкладки
    _GUICtrlTab_InsertItem($idTab, 0, "Tab 0")
    _GUICtrlTab_InsertItem($idTab, 1, "Tab 1")
    _GUICtrlTab_InsertItem($idTab, 2, "Tab 2")

    ; Показать количество вкладок
    _MemoMsgBox($MB_SYSTEMMODAL, "Information", "Number of tabs: " & _GUICtrlTab_GetItemCount($idTab))

    _MemoMsgBoxStatus("", -1, $hGUI) ; больше действий нет, ждём закрытия GUI

EndFunc   ;==>Example
