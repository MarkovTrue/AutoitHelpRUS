_GDIPlus_GraphicsDrawImage

Рисует объект Image.

#include <GDIPlus.au3>
_GDIPlus_GraphicsDrawImage ( $hGraphics, $hImage, $nX, $nY )

Параметры

$hGraphics Дескриптор объекта Graphics.
$hImage Дескриптор объекта Image.
$nX Координата X верхнего левого угла отображаемого изображения.
$nY Координата Y верхнего левого угла отображаемого изображения.

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False и устанавливает @error ≠ 0.
@extended: Может содержать код ошибки GPSTATUS ($GDIP_ERR* см. GPIPlusConstants.au3).

Дополнительно

Искать GdipDrawImage в библиотеке MSDN

Пример

Пример 1

#include <GDIPlus.au3>
#include <ScreenCapture.au3>
#include <WinAPIHObj.au3>

Example()

Func Example()
    Local $hBitmap1, $hBitmap2, $hImage1, $hImage2, $hGraphic

    ; Инициализировать библиотеку GDI+
    _GDIPlus_Startup()

    ; Захватить весь экран
    $hBitmap1 = _ScreenCapture_Capture("")
    $hImage1 = _GDIPlus_BitmapCreateFromHBITMAP($hBitmap1)

    ; Захватить область экрана
    $hBitmap2 = _ScreenCapture_Capture("", 0, 0, 400, 300)
    $hImage2 = _GDIPlus_BitmapCreateFromHBITMAP($hBitmap2)

    ; Нарисовать одно изображение в другом
    $hGraphic = _GDIPlus_ImageGetGraphicsContext($hImage1)
    _GDIPlus_GraphicsDrawImage($hGraphic, $hImage2, 100, 100)

    ; Нарисовать рамку вокруг вставленного изображения
    _GDIPlus_GraphicsDrawRect($hGraphic, 100, 100, 400, 300)

    ; Сохранить итоговое изображение
    _GDIPlus_ImageSaveToFile($hImage1, @MyDocumentsDir & "\GDIPlus_Image.jpg")

    ; Освободить ресурсы
    _GDIPlus_ImageDispose($hImage1)
    _GDIPlus_ImageDispose($hImage2)
    _WinAPI_DeleteObject($hBitmap1)
    _WinAPI_DeleteObject($hBitmap2)

    ; Завершить работу с библиотекой GDI+
    _GDIPlus_Shutdown()

    ShellExecute(@MyDocumentsDir & "\GDIPlus_Image.jpg")
EndFunc   ;==>Example

Пример 2

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>
#include <ScreenCapture.au3>
#include <WinAPIHObj.au3>

Global $g_hGUI, $g_hGfxCtxt, $g_hBitmap, $g_hBMP, $g_hGraphics

Example()

Func Example()
    AutoItSetOption("GUIOnEventMode", 1)

    _GDIPlus_Startup() ;инициализировать GDI+
    Local Const $iWidth = 600, $iHeight = 600, $iBgColor = 0x303030 ;$iBGColor в формате RRGGBB

    $g_hGUI = GUICreate("GDI+ Example (" & @ScriptName & ")", $iWidth, $iHeight) ;создать тестовый GUI
    GUISetBkColor($iBgColor, $g_hGUI) ;установить цвет фона GUI
    GUISetState(@SW_SHOW)

    ;создать буферизованный набор кадров для плавного перемещения графики
    $g_hGraphics = _GDIPlus_GraphicsCreateFromHWND($g_hGUI) ;создать графический объект из дескриптора окна
    $g_hBitmap = _GDIPlus_BitmapCreateFromGraphics($iWidth, $iHeight, $g_hGraphics)
    $g_hGfxCtxt = _GDIPlus_ImageGetGraphicsContext($g_hBitmap)
    Local $iW = 300, $iH = 300
    Local $hHBmp = _ScreenCapture_Capture("", 0, 0, $iW, $iH) ;создать GDI bitmap захватом области рабочего стола
    $g_hBMP = _GDIPlus_BitmapCreateFromHBITMAP($hHBmp) ;преобразовать GDI bitmap в GDI+
    _WinAPI_DeleteObject($hHBmp) ;освободить ресурс GDI bitmap – больше не нужен
    Local $iVectorX = Random(1.5, 2.5), $iVectorY = Random(1.5, 2.5) ;определить векторы x и y
    Local $iX = 0.0, $iY = 0.0 ;определить начальную координату

    GUISetOnEvent($GUI_EVENT_CLOSE, "_Exit")

    Do
        _GDIPlus_GraphicsClear($g_hGfxCtxt, 0xFF000000 + $iBgColor) ;очистить bitmap для перерисовки
        _GDIPlus_GraphicsDrawImage($g_hGfxCtxt, $g_hBMP, $iX, $iY) ;нарисовать растровое изображение в задний буфер
        _GDIPlus_GraphicsDrawImageRect($g_hGraphics, $g_hBitmap, 0, 0, $iWidth, $iHeight) ;скопировать нарисованный bitmap в дескриптор графики (GUI)
        $iX += $iVectorX ;прибавить x-вектор к текущей позиции x
        $iY += $iVectorY ;прибавить y-вектор к текущей позиции y
        If $iX < 0 Or $iX > ($iWidth - $iW) Then $iVectorX *= -1 ;при достижении границы x развернуть x-вектор
        If $iY < 0 Or $iY > ($iHeight - $iH) Then $iVectorY *= -1 ;при достижении границы y развернуть y-вектор
    Until Not Sleep(10) ;пауза 10 мс для снижения нагрузки на процессор
EndFunc   ;==>Example

Func _Exit()
    ;освободить ресурсы GDI+
    _GDIPlus_GraphicsDispose($g_hGfxCtxt)
    _GDIPlus_GraphicsDispose($g_hGraphics)
    _GDIPlus_BitmapDispose($g_hBitmap)
    _GDIPlus_BitmapDispose($g_hBMP)
    _GDIPlus_Shutdown()
    GUIDelete($g_hGUI)
    Exit
EndFunc   ;==>_Exit
