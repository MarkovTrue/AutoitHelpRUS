Справочник функций

_GUICtrlHeader_DeleteItem

Удаляет элемент из элемента управления Header.

#include <GuiHeader.au3>
_GUICtrlHeader_DeleteItem ( $hWnd, $iIndex )

Параметры

$hWnd Дескриптор элемента управления.
$iIndex Индекс элемента (нумерация с 0).

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False.

См. также

_GUICtrlHeader_AddItem, _GUICtrlHeader_InsertItem

Пример

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <GuiHeader.au3>
#include <MsgBoxConstants.au3>

Example()

Func Example()
    ; Создать GUI
    Local $hGUI = GUICreate("Header Delete Item (v" & @AutoItVersion & ")", 450, 300, 100, 100)
    Local $hHeader = _GUICtrlHeader_Create($hGUI)
    _MemoCreate(2, 52, 444, 220)

;~ _GUICtrlHeader_SetUnicodeFormat($hHeader, True)

    GUISetState(@SW_SHOW)

    ; Добавить столбцы
    _GUICtrlHeader_AddItem($hHeader, "Column 0", 100)
    _GUICtrlHeader_AddItem($hHeader, "Column 1", 100)
    _GUICtrlHeader_AddItem($hHeader, "Column 2", 100)
    _GUICtrlHeader_AddItem($hHeader, "Column 3", 100)

    ; Удалить столбец 2
    _MemoMsgBox($MB_SYSTEMMODAL, "Information", "Deleting Column 2")
    _GUICtrlHeader_DeleteItem($hHeader, 2)

    _MemoMsgBoxStatus("", Default, $hGUI) ; больше действий нет, ждём закрытия GUI

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
EndFunc   ;==>Example
