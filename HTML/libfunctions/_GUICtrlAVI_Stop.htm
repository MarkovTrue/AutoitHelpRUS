Справочник функций

_GUICtrlAVI_Stop

Останавливает воспроизведение AVI-клипа.

#include <GuiAVI.au3>
_GUICtrlAVI_Stop ( $hWnd )

Параметры

$hWnd Идентификатор или дескриптор элемента управления.

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False.

См. также

_GUICtrlAVI_Play

Пример

Пример 1

; используя функцию AutoIt

#include <GuiAVI.au3>
#include <GUIConstantsEx.au3>

_Example_Internal()

Func _Example_Internal()
    Local $idAVI, $idBtn_Start, $idBtn_Stop

    ; Создать GUI
    GUICreate("(Internal) AVI Play/Stop", 300, 200)
    $idAVI = GUICtrlCreateAvi(@SystemDir & "\shell32.dll", 160, 10, 10)
    $idBtn_Start = GUICtrlCreateButton("start", 50, 150, 70, 22)
    $idBtn_Stop = GUICtrlCreateButton("stop", 150, 150, 70, 22)
    GUISetState(@SW_SHOW)

    ; Цикл до выхода пользователя.
    While 1
        Switch GUIGetMsg()
            Case $GUI_EVENT_CLOSE
                ExitLoop
            Case $idBtn_Start
                ; Воспроизвести часть AVI-клипа
                _GUICtrlAVI_Play($idAVI)
            Case $idBtn_Stop
                ; Остановить AVI-клип
                _GUICtrlAVI_Stop($idAVI)
        EndSwitch
    WEnd

    ; Закрыть AVI-клип
    _GUICtrlAVI_Close($idAVI)

    GUIDelete()
EndFunc   ;==>_Example_Internal
