Справочник функций

_GUICtrlComboBox_AutoComplete

Включает или отключает автодополнение.

#include <GuiComboBox.au3>
_GUICtrlComboBox_AutoComplete ( $hWnd )

Параметры

$hWnd Идентификатор или дескриптор элемента управления.

Возвращаемое значение

Отсутствует.

Пример

#include "Extras\WM_NOTIFY.au3"

#include <GuiComboBox.au3>
#include <GUIConstantsEx.au3>
#include <WindowsNotifsConstants.au3>

Global $g_idCombo

Example()

Func Example()
    ; Создать GUI
    GUICreate("ComboBox Auto Complete (v" & @AutoItVersion & ")", 400, 296)
    $g_idCombo = GUICtrlCreateCombo("", 2, 2, 396, 296)
    GUISetState(@SW_SHOW)

    ; Добавить файлы
    _GUICtrlComboBox_BeginUpdate($g_idCombo)
    _GUICtrlComboBox_AddDir($g_idCombo, @WindowsDir & "\*.exe")
    _GUICtrlComboBox_EndUpdate($g_idCombo)

    GUIRegisterMsg($WM_COMMAND, "WM_COMMAND")

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
    GUIDelete()
EndFunc   ;==>Example

Func _Edit_Changed()
    _GUICtrlComboBox_AutoComplete($g_idCombo)
EndFunc   ;==>_Edit_Changed

Func WM_COMMAND($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg
    Local $hWndCombo = $g_idCombo
    If Not IsHWnd($g_idCombo) Then $hWndCombo = GUICtrlGetHandle($g_idCombo)
    Local $hWndFrom = $lParam
    Local $iIDFrom = BitAND($wParam, 0xFFFF) ; Младшее слово
    Local $iCode = BitShift($wParam, 16) ; Старшее слово
    Switch $hWndFrom
        Case $g_idCombo, $hWndCombo
            Switch $iCode
                Case $CBN_CLOSEUP ; Отправляется при закрытии списка combo box
                    _WM_NOTIFY_DebugInfo("$CBN_CLOSEUP", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_DBLCLK ; Отправляется при двойном клике по строке в списке combo box
                    _WM_NOTIFY_DebugInfo("$CBN_DBLCLK", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_DROPDOWN ; Отправляется перед показом списка combo box
                    _WM_NOTIFY_DebugInfo("$CBN_DROPDOWN", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_EDITCHANGE ; Отправляется после действия пользователя, которое могло изменить текст в edit-части combo box
                    _WM_NOTIFY_DebugInfo("$CBN_EDITCHANGE", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    _Edit_Changed()
                    ; нет возвращаемого значения
                Case $CBN_EDITUPDATE ; Отправляется перед показом изменённого текста в edit-части combo box
                    _WM_NOTIFY_DebugInfo("$CBN_EDITUPDATE", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_ERRSPACE ; Отправляется, когда combo box не может выделить достаточно памяти для запроса
                    _WM_NOTIFY_DebugInfo("$CBN_ERRSPACE", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_KILLFOCUS ; Отправляется при потере клавиатурного фокуса combo box
                    _WM_NOTIFY_DebugInfo("$CBN_KILLFOCUS", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_SELCHANGE ; Отправляется при изменении текущего выбора в списке combo box
                    _WM_NOTIFY_DebugInfo("$CBN_SELCHANGE", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_SELENDCANCEL ; Отправляется при выборе элемента, затем переключении на другой контрол или закрытии диалога
                    _WM_NOTIFY_DebugInfo("$CBN_SELENDCANCEL", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_SELENDOK ; Отправляется при выборе элемента списка или при выборе с последующим закрытием списка
                    _WM_NOTIFY_DebugInfo("$CBN_SELENDOK", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_SETFOCUS ; Отправляется при получении клавиатурного фокуса combo box
                    _WM_NOTIFY_DebugInfo("$CBN_SETFOCUS", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
            EndSwitch
    EndSwitch
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_COMMAND
