_WinAPI_CreateDIBColorTable

создаёт таблицу цветов RGB из указанного массива цветов.

#include <WinAPIGdi.au3>
_WinAPI_CreateDIBColorTable ( Const ByRef $aColorTable [, $iStart = 0 [, $iEnd = -1]] )

Параметры

$aColorTable Цвет (RGB).
$iStart Координаты.
$iEnd Координаты.

Возвращаемое значение

Успех: "dword[n]"-структура, представляющая DIB-таблицу цветов.
Ошибка: Устанавливает @error ≠ 0.

Примечания

Таблица цветов, создаваемая этой функцией, обычно используется в _WinAPI_CreateDIB() или _WinAPI_CreateDIBSection() для создания 1-, 4- или 8-битного аппаратно-независимого битмапа (DIB).

См. также

_WinAPI_CreateDIB, _WinAPI_CreateDIBSection

Пример

#include <GUIConstantsEx.au3>
#include <SendMessage.au3>
#include <StaticConstants.au3>
#include <WinAPIGdi.au3>
#include <WinAPIGdiDC.au3>
#include <WinAPIHObj.au3>
#include <WinAPIMem.au3>
#include <WinAPIMisc.au3>

; Создать массив цветов из 256 элементов, нужный для 8-битного bitmap
Local $aColorTable[256]
For $i = 0 To 255
    $aColorTable[$i] = _WinAPI_RGB(0, $i, 255 - $i)
Next

; Создать таблицу цветов из массива цветов
Local $tColorTable = _WinAPI_CreateDIBColorTable($aColorTable)

; Создать 8-битный/пикс device-independent bitmap (DIB) и получить указатель на его биты
Local $hBitmap = _WinAPI_CreateDIB(256, 256, 8, $tColorTable, 256)
Local $pBits = _WinAPI_GetExtended()

; Заполнить индексы цветов bitmap
For $i = 0 To 255
    _WinAPI_FillMemory($pBits + 256 * $i, 256, $i)
Next

; Создать GUI
Local $hForm = GUICreate('Test ' & StringReplace(@ScriptName, '.au3', '()'), 256, 256)
Local $idPic = GUICtrlCreatePic('', 0, 0, 256, 256)
Local $hPic = GUICtrlGetHandle($idPic)

; Создать DDB из DIB для корректного отображения в элементе управления
Local $hDC = _WinAPI_GetDC($hPic)
Local $hDev = _WinAPI_CreateCompatibleBitmap($hDC, 256, 256)
Local $hMemDC = _WinAPI_CreateCompatibleDC($hDC)
Local $hMemSv = _WinAPI_SelectObject($hMemDC, $hDev)
_WinAPI_DrawBitmap($hMemDC, 0, 0, $hBitmap)
_WinAPI_ReleaseDC($hPic, $hDC)
_WinAPI_SelectObject($hMemDC, $hMemSv)
_WinAPI_DeleteDC($hMemDC)

; Установить bitmap для элемента управления
_SendMessage($hPic, $STM_SETIMAGE, 0, $hDev)
Local $hObj = _SendMessage($hPic, $STM_GETIMAGE)
If $hObj <> $hDev Then
    _WinAPI_DeleteObject($hDev)
EndIf

; Показать GUI
GUISetState(@SW_SHOW)

; Сохранить 8-битный bitmap в .bmp файл
Local $sPath = FileSaveDialog('Save Image', @TempDir, 'Bitmap Image Files (*.bmp)', 2 + 16, 'MyImage.bmp', $hForm)
If $sPath Then
    _WinAPI_SaveHBITMAPToFile($sPath, $hBitmap, 2834, 2834)
EndIf

Do
Until GUIGetMsg() = $GUI_EVENT_CLOSE
