_GDIPlus_GraphicsDrawClosedCurve2

Рисует замкнутый кардинальный сплайн.

#include <GDIPlus.au3>
_GDIPlus_GraphicsDrawClosedCurve2 ( $hGraphics, $aPoints, $nTension [, $hPen = 0] )

Параметры

$hGraphics Указатель на объект Graphics.
$aPoints Массив точек замкнутого кардинального сплайна:
[0][0] – количество точек
[1][0] – X позиция точки 1
[1][1] – Y позиция точки 1
[2][0] – X позиция точки 2
[2][1] – Y позиция точки 2
[n][0] – X позиция точки n
[n][1] – Y позиция точки n
$nTension Число, задающее степень изгиба кривой в точках замкнутого кардинального сплайна.
$hPen Дескриптор объекта Pen.

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False и устанавливает @error ≠ 0.
@extended: Может содержать код ошибки GPSTATUS ($GDIP_ERR* см. GPIPlusConstants.au3).

См. также

_GDIPlus_GraphicsDrawClosedCurve

Пример

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>

Global $g_hGUI, $g_hGraphics, $g_hBitmap, $g_hGfxCtxt, $g_hPen

Example()

Func Example()
    AutoItSetOption("GUIOnEventMode", 1)

    _GDIPlus_Startup() ;инициализировать GDI+
    Local Const $iWidth = 600, $iHeight = 600, $iBgColor = 0x303030 ;$iBGColor в формате RRGGBB

    $g_hGUI = GUICreate("GDI+ Example (" & @ScriptName & ")", $iWidth, $iHeight) ;создать тестовый GUI
    GUISetBkColor($iBgColor, $g_hGUI) ;установить цвет фона GUI
    GUISetState(@SW_SHOW)

    ;создать буферизованный набор кадров для плавного перемещения графики
    $g_hGraphics = _GDIPlus_GraphicsCreateFromHWND($g_hGUI) ;создать графический объект из дескриптора окна
    $g_hBitmap = _GDIPlus_BitmapCreateFromGraphics($iWidth, $iHeight, $g_hGraphics)
    $g_hGfxCtxt = _GDIPlus_ImageGetGraphicsContext($g_hBitmap)
    _GDIPlus_GraphicsSetSmoothingMode($g_hGfxCtxt, $GDIP_SMOOTHINGMODE_HIGHQUALITY) ;задаёт качество отрисовки графики (сглаживание)
    $g_hPen = _GDIPlus_PenCreate(0xFFFF8060, 2)
    _GDIPlus_GraphicsSetPixelOffsetMode($g_hGfxCtxt, $GDIP_PIXELOFFSETMODE_HIGHQUALITY)

    Local $aPoints[11][4], $x, $y
    $aPoints[0][0] = 10

    For $y = 0 To 1
        For $x = 1 To 5
            $aPoints[$y * 5 + $x][0] = 100 + 300 * $y + 50 ;координата x точки
            $aPoints[$y * 5 + $x][1] = 150 + $x * 50 ;координата y точки
            $aPoints[$y * 5 + $x][2] = Random(-2, 2) ;вектор x точки
            $aPoints[$y * 5 + $x][3] = Random(-2, 2) ;вектор y точки
        Next
    Next

    GUISetOnEvent($GUI_EVENT_CLOSE, "_Exit")

    Do
        _GDIPlus_GraphicsClear($g_hGfxCtxt, 0xFF000000 + $iBgColor) ;очистить bitmap для перерисовки
        _GDIPlus_GraphicsDrawClosedCurve2($g_hGfxCtxt, $aPoints, 1.25, $g_hPen) ;нарисовать замкнутую кривую
        _GDIPlus_GraphicsDrawImage($g_hGraphics, $g_hBitmap, 0, 0) ;скопировать bitmap в дескриптор графики (GUI)
        For $y = 1 To $aPoints[0][0]
            $aPoints[$y][0] += $aPoints[$y][2] ;вычислить новую позицию x
            If $aPoints[$y][0] < 0 Or $aPoints[$y][0] > $iWidth Then $aPoints[$y][2] *= -1 ;при достижении вертикальной границы инвертировать вектор x
            $aPoints[$y][1] += $aPoints[$y][3] ;вычислить новую позицию y
            If $aPoints[$y][1] < 0 Or $aPoints[$y][1] > $iHeight Then $aPoints[$y][3] *= -1 ;при достижении горизонтальной границы инвертировать вектор y
        Next
    Until Not Sleep(30) ;пауза 30 мс, чтобы избежать высокой загрузки процессора
EndFunc   ;==>Example

Func _Exit()
    ;освободить ресурсы GDI+
    _GDIPlus_PenDispose($g_hPen)
    _GDIPlus_GraphicsDispose($g_hGfxCtxt)
    _GDIPlus_GraphicsDispose($g_hGraphics)
    _GDIPlus_BitmapDispose($g_hBitmap)
    _GDIPlus_Shutdown()
    GUIDelete($g_hGUI)
    Exit
EndFunc   ;==>_Exit
