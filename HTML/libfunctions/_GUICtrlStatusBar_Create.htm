Справочник функций

_GUICtrlStatusBar_Create

Создаёт элемент управления StatusBar.

#include <GuiStatusBar.au3>
_GUICtrlStatusBar_Create ( $hWnd [, $vPartEdge = -1 [, $vPartText = "" [, $iStyles = -1 [, $iExStyles = 0x00000000]]]] )

Параметры

$hWnd Дескриптор родительского окна.
$vPartEdge [необязательный] Ширина элемента управления.
$vPartText [необязательный] Текст части или частей; для нескольких частей передайте массив с отсчётом от 0 в виде:
$vPartText[0] – первая часть
$vPartText[1] – вторая часть
$vPartText[n] – n-я часть
$iStyles [необязательный] Стиль элемента управления.
$iExStyles [необязательный] Расширенный стиль элемента. Соответствует стандартным константам $WS_EX_*. См. таблицу расширенных стилей.

Возвращаемое значение

Успех: Возвращает дескриптор.
Ошибка: Возвращает 0; @error может быть установлен (см. раздел Примечания).

Примечания

Если окно, на которое ссылается $hWnd, принадлежит другому процессу, @error устанавливается в 4.

При использовании GUICtrlCreateMenu() _GUICtrlStatusBar_Create() следует вызывать ПОСЛЕ него, иначе строка состояния не будет видна.

См. также

_GUICtrlStatusBar_Destroy

Пример

Пример 1

; == Пример 1 : с дескриптором окна GUI

#include "Extras\HelpFileInternals.au3"
#include "Extras\WM_NOTIFY.au3"

#include <GUIConstantsEx.au3>
#include <GuiStatusBar.au3>

Global $g_hStatus

Example()

Func Example()
    ; Создать GUI
    Local $hGUI = GUICreate("StatusBar Create (v" & @AutoItVersion & ")", 450, 320, 100, 100)

    ; по умолчанию 1 часть, без текста
    Local $hStatus = _GUICtrlStatusBar_Create($hGUI)
    $g_hStatus = $hStatus

    ; Создать элемент memo
    _MemoCreate(2, 8, 444, 259)

    GUISetState(@SW_SHOW)

    Local $aParts[3] = [75, 150, -1]
    _GUICtrlStatusBar_SetParts($hStatus, $aParts)

    _MemoWrite("StatusBar Created with:" & @CRLF & _
            @TAB & "Handle to GUI window" & @CRLF)

    _WM_NOTIFY_Register($_g_idLst_Memo)

    ; Получить размеры рамок
    _MemoWrite("Horizontal border width .: " & _GUICtrlStatusBar_GetBordersHorz($hStatus))
    _MemoWrite("Vertical border width ...: " & _GUICtrlStatusBar_GetBordersVert($hStatus))
    _MemoWrite("Width between rectangles : " & _GUICtrlStatusBar_GetBordersRect($hStatus))
    _MemoWrite("")

    _MemoMsgBoxStatus("", -1, $hGUI) ; больше действий нет, ждём закрытия GUI

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE

    GUIDelete()
EndFunc   ;==>Example

Func WM_NOTIFY($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $wParam

    Local $tNMHDR = DllStructCreate($tagNMHDR, $lParam)
    Local $hWndFrom = HWnd(DllStructGetData($tNMHDR, "hWndFrom"))
    Local $iCode = DllStructGetData($tNMHDR, "Code")
    Switch $hWndFrom
        Case $g_hStatus
            Switch $iCode
                Case $NM_CLICK ; Пользователь кликнул левой кнопкой мыши в элементе управления
                    _WM_NOTIFY_DebugEvent("$NM_CLICK", $tagNMMOUSE, $lParam, "IDFrom,,ItemSpec,ItemData,X,Y,HitInfo")
                    Return True ; сообщить, что клик мыши обработан, и подавить стандартную обработку системой
                    ; Return FALSE ;чтобы разрешить стандартную обработку клика.
                Case $NM_DBLCLK ; Пользователь дважды кликнул левой кнопкой мыши в элементе управления
                    _WM_NOTIFY_DebugEvent("$NM_DBLCLK", $tagNMMOUSE, $lParam, "IDFrom,,ItemSpec,ItemData,X,Y,HitInfo")
                    Return True ; сообщить, что клик мыши обработан, и подавить стандартную обработку системой
                    ; Return FALSE ;чтобы разрешить стандартную обработку клика.
                Case $NM_RCLICK ; Пользователь кликнул правой кнопкой мыши в элементе управления
                    _WM_NOTIFY_DebugEvent("$NM_RCLICK", $tagNMMOUSE, $lParam, "IDFrom,,ItemSpec,ItemData,X,Y,HitInfo")
                    Return True ; сообщить, что клик мыши обработан, и подавить стандартную обработку системой
                    ; Return FALSE ;чтобы разрешить стандартную обработку клика.
                Case $NM_RDBLCLK ; Пользователь дважды кликнул правой кнопкой мыши в элементе управления
                    _WM_NOTIFY_DebugEvent("$NM_RDBLCLK", $tagNMMOUSE, $lParam, "IDFrom,,ItemSpec,ItemData,X,Y,HitInfo")
                    Return True ; сообщить, что клик мыши обработан, и подавить стандартную обработку системой
                    ; Return FALSE ;чтобы разрешить стандартную обработку клика.
                Case $SBN_SIMPLEMODECHANGE ; Изменения простого режима
                    _WM_NOTIFY_DebugEvent("$SBN_SIMPLEMODECHANGE", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                ; Нет возвращаемого значения
            EndSwitch
    EndSwitch
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_NOTIFY
