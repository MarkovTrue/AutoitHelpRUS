Справочник функций

_GUICtrlHeader_Create

Создаёт элемент управления Header.

#include <GuiHeader.au3>
_GUICtrlHeader_Create ( $hWnd [, $iStyle = 0x00000046] )

Параметры

$hWnd Дескриптор родительского или владеющего окна.
$iStyle [необязательный] Стили элемента управления:
$HDS_BUTTONS – каждый элемент в элементе управления выглядит и работает как кнопка
$HDS_DRAGDROP – разрешает перетаскивание и переупорядочение элементов заголовка
$HDS_FILTERBAR – включает полосу фильтра как часть стандартного элемента управления Header
$HDS_FLAT – рисует элемент управления Header плоским
$HDS_FULLDRAG – заставляет элемент управления Header отображать содержимое столбца
$HDS_HIDDEN – указывает элемент управления Header, который предполагается скрывать
$HDS_HORZ – создаёт элемент управления Header с горизонтальной ориентацией
$HDS_HOTTRACK – включает отслеживание горячих точек

По умолчанию: $HDS_BUTTONS, $HDS_HOTTRACK, $HDS_DRAGDROP
Обязательные: $WS_CHILD, $WS_VISIBLE.

Возвращаемое значение

Успех: Возвращает дескриптор элемента управления Header.
Ошибка: Возвращает 0, переменная @error может быть установлена (см. раздел Примечания).

Примечания

Если окно, на которое ссылается $hWnd, находится в другом процессе, переменная @error будет установлена на 4.

См. также

_GUICtrlHeader_Destroy

EndFunc   ;==>Example

Func WM_NOTIFY($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $wParam

    Local $tNMHDR = DllStructCreate($tagNMHDR, $lParam)
    Local $hWndFrom = HWnd(DllStructGetData($tNMHDR, "hWndFrom"))
    Local $iCode = DllStructGetData($tNMHDR, "Code")
    Switch $hWndFrom
        Case $g_hHeader
            Switch $iCode
                Case $HDN_BEGINDRAG ; Отправляется header-элементом при начале перетаскивания одного из его элементов
                    _WM_NOTIFY_DebugEvent("$HDN_BEGINDRAG", $tagNMHEADER, $lParam, "IDFrom,,Item,Button")
                    Return False ; Чтобы позволить элементу управления Header автоматически управлять операциями перетаскивания
                    ; Возвращает True для указания внешнего (ручного) управления перетаскиванием, позволяющего владельцу
                    ; элемента управления предоставлять пользовательские услуги в рамках процесса перетаскивания
                Case $HDN_BEGINTRACK, $HDN_BEGINTRACKW ; Уведомляет родительское окно header-элемента, что пользователь начал перетаскивать разделитель
                    _WM_NOTIFY_DebugEvent("$HDN_BEGINTRACK", $tagNMHEADER, $lParam, "IDFrom,,Item,Button")
                    Return False ; Чтобы позволить отслеживание разделителя
                    ; Возвращает True для предотвращения отслеживания
                Case $HDN_DIVIDERDBLCLICK, $HDN_DIVIDERDBLCLICKW ; Уведомляет родительское окно header-элемента, что пользователь дважды кликнул по разделителю
                    _WM_NOTIFY_DebugEvent("$HDN_DIVIDERDBLCLICK", $tagNMHEADER, $lParam, "IDFrom,,Item,Button")
                    ; нет возвращаемого значения
                Case $HDN_ENDDRAG ; Отправляется header-элементом при завершении перетаскивания одного из его элементов
                    _WM_NOTIFY_DebugEvent("$HDN_ENDDRAG", $tagNMHEADER, $lParam, "IDFrom,,Item,Button")
                    Return False ; Чтобы позволить элементу управления автоматически размещать и переупорядочивать элемент
                    ; Возвращает True для предотвращения размещения элемента
                Case $HDN_ENDTRACK, $HDN_ENDTRACKW ; Уведомляет родительское окно header-элемента, что пользователь закончил перетаскивать разделитель
                    _WM_NOTIFY_DebugEvent("$HDN_ENDTRACK", $tagNMHEADER, $lParam, "IDFrom,,Item,Button")
                    ; нет возвращаемого значения
                Case $HDN_FILTERBTNCLICK ; Уведомляет родительское окно header-а при клике на кнопку фильтра или в ответ на $HDM_SETITEM
                    _WM_NOTIFY_DebugEvent("$HDN_FILTERBTNCLICK", $tagNMHDFILTERBTNCLICK, $lParam, "IDFrom,,Item,Left,Top,Right,Bottom")
                    ; Return True ; Уведомление $HDN_FILTERCHANGE будет отправлено родительскому окну header-а
                    ; Это уведомление даёт родительскому окну возможность синхронизировать элементы интерфейса
                    Return False ; Если уведомление отправлять не нужно
                Case $HDN_FILTERCHANGE ; Уведомляет родительское окно header-а, что атрибуты фильтра header-а меняются или редактируются
                    _WM_NOTIFY_DebugEvent("$HDN_FILTERCHANGE", $tagNMHEADER, $lParam, "IDFrom,,Item,Button")
                    ; нет возвращаемого значения
                Case $HDN_GETDISPINFO, $HDN_GETDISPINFOW ; Отправляется владельцу header-элемента, когда нужны сведения о callback header item
                    _WM_NOTIFY_DebugEvent("$HDN_GETDISPINFO", $tagNMHDDISPINFO, $lParam, "IDFrom,Item")
                    ; Возвращает LRESULT
                Case $HDN_ITEMCHANGED, $HDN_ITEMCHANGEDW ; Уведомляет родительское окно header-элемента, что атрибуты header-item изменились
                    _WM_NOTIFY_DebugEvent("$HDN_ITEMCHANGED", $tagNMHEADER, $lParam, "IDFrom,,Item,Button")
                    ; нет возвращаемого значения
                Case $HDN_ITEMCHANGING, $HDN_ITEMCHANGINGW ; Уведомляет родительское окно header-элемента, что атрибуты header-item скоро изменятся
                    _WM_NOTIFY_DebugEvent("$HDN_ITEMCHANGING", $tagNMHEADER, $lParam, "IDFrom,,Item,Button")
                    Return False ; Чтобы разрешить изменения
                    ; Возвращает True для их предотвращения
                Case $HDN_ITEMCLICK, $HDN_ITEMCLICKW ; Уведомляет родительское окно header-элемента, что пользователь кликнул по нему
                    _WM_NOTIFY_DebugEvent("$HDN_ITEMCLICK", $tagNMHEADER, $lParam, "IDFrom,,Item,Button")
                    ; нет возвращаемого значения
                Case $HDN_ITEMDBLCLICK, $HDN_ITEMDBLCLICKW ; Уведомляет родительское окно header-элемента, что пользователь дважды кликнул по нему
                    _WM_NOTIFY_DebugEvent("$HDN_ITEMDBLCLICK", $tagNMHEADER, $lParam, "IDFrom,,Item,Button")
                    ; нет возвращаемого значения
                Case $HDN_TRACK, $HDN_TRACKW ; Уведомляет родительское окно header-элемента, что пользователь перетаскивает разделитель
                    _WM_NOTIFY_DebugEvent("$HDN_TRACK", $tagNMHEADER, $lParam, "IDFrom,,Item,Button")
                    Return False ; Чтобы продолжить отслеживание разделителя
                    ; Возвращает True для завершения отслеживания
            EndSwitch
    EndSwitch
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_NOTIFY

; == Пример 2: заголовок создания для внешнего процесса

#include "Extras\HelpFileInternals.au3"

#include <GuiHeader.au3>

Example()

Func Example()
    Local $sFromTo
    Local $hWin = _MemoRunAU3OutProcess($sFromTo, True)

    Local $hHeader = _MemoCreateOutProcess($hWin, "SysHeader32", 0, $sFromTo)

    _GUICtrlHeader_Destroy($hHeader)
    If @error Then _MemoWrite("_GUICtrlHeader_Destroy()" & " @error = " & @error & @CRLF & _
            @TAB & "cannot be retrieved by an external process")

    _GUICtrlHeader_Create($hWin)
    If @error Then _MemoWrite("_GUICtrlHeader_Create()" & " @error = " & @error & @CRLF & _
            @TAB & "cannot be retrieved by an external process")

    _MemoMsgBoxStatus("", Default, $hWin) ; больше действий нет, ждём закрытия GUI, закрываем и внешний GUI

EndFunc   ;==>Example
