Справочник функций

_Crypt_DecryptData

Расшифровывает данные с помощью указанного ключа.

#include <Crypt.au3>
_Crypt_DecryptData ( $vData, $vCryptKey, $iAlgID [, $bFinal = True] )

Параметры

$vData Данные для расшифровки. См. раздел Примечания.
$vCryptKey Пароль или дескриптор ключа при использовании флага $CALG_USERKEY.
$iAlgID Используемый алгоритм. См. _Crypt_DeriveKey().
$bFinal [необязательный] False, если передаётся только фрагмент полных данных.

Возвращаемое значение

Успех: Возвращает двоичную строку с расшифрованными данными.
Ошибка: Устанавливает @error ≠ 0.
@error: 70 – ошибка расшифровки данных
10-30 – не удалось создать ключ
80 – не удалось получить AlgID для $CALG_USERKEY
1000+ – ошибка _Crypt_Startup()

Примечания

Расшифрованные данные всегда возвращаются как двоичная строка, даже если исходные данные были строкой (преобразуйте через BinaryToString()).

Если зашифрованные данные $vData содержат UTF-8 символы в двоичном виде, для получения корректной строки Unicode необходимо вызвать BinaryToString(..., $SB_UTF8). См. пример 3.

См. также

_Crypt_DeriveKey, _Crypt_EncryptData

Дополнительно

Искать CryptDecrypt в библиотеке MSDN

Пример 1

#include <ComboConstants.au3>
#include <Crypt.au3>
#include <EditConstants.au3>
#include <GUIConstantsEx.au3>
#include <MsgBoxConstants.au3>
#include <StringConstants.au3>
#include <WinAPIConv.au3>
#include <WindowsNotifsConstants.au3>

Global $g_hKey = -1, $g_idEdt_Input = -1, $g_idEdt_Output = -1, $g_idEdt_OutputDeCrypted = -1

Example()

Func Example()
    Local $hGUI = GUICreate("Realtime (En/DE)cryption", 400, 470)
    $g_idEdt_Input = GUICtrlCreateEdit("", 0, 0, 400, 150, $ES_WANTRETURN)
    $g_idEdt_Output = GUICtrlCreateEdit("", 0, 150, 400, 150, $ES_READONLY)
    $g_idEdt_OutputDeCrypted = GUICtrlCreateEdit("", 0, 300, 400, 150, $ES_READONLY)
    Local $idCombo = GUICtrlCreateCombo("", 0, 450, 100, 20, $CBS_DROPDOWNLIST)
    GUICtrlSetData($idCombo, "3DES (168bit)|AES (128bit)|AES (192bit)|AES (256bit)|DES (56bit)|RC2 (128bit)|RC4 (128bit)", "RC4 (128bit)")
    GUIRegisterMsg($WM_COMMAND, "WM_COMMAND")
    GUISetState(@SW_SHOW, $hGUI)

    _Crypt_Startup() ; Для оптимизации производительности запустить библиотеку Crypt.

    Local $iAlgorithm = $CALG_RC4
    $g_hKey = _Crypt_DeriveKey(StringToBinary("CryptPassword"), $iAlgorithm) ; Объявить парольную строку и алгоритм для создания криптографического ключа.

    While 1
        Switch GUIGetMsg()
            Case $GUI_EVENT_CLOSE
                Exit

            Case $idCombo ; Проверить выбор в combobox и получить соответствующий алгоритм.
                Switch GUICtrlRead($idCombo) ; Прочитать выбор в combobox.
                    Case "3DES (168bit)"
                        $iAlgorithm = $CALG_3DES

                    Case "AES (128bit)"
                        $iAlgorithm = $CALG_AES_128

                    Case "AES (192bit)"
                        $iAlgorithm = $CALG_AES_192

                    Case "AES (256bit)"
                        $iAlgorithm = $CALG_AES_256

                    Case "DES (56bit)"
                        $iAlgorithm = $CALG_DES

                    Case "RC2 (128bit)"
                        $iAlgorithm = $CALG_RC2

                    Case "RC4 (128bit)"
                        $iAlgorithm = $CALG_RC4

                EndSwitch

                _Crypt_DestroyKey($g_hKey) ; Уничтожить криптографический ключ.
                $g_hKey = _Crypt_DeriveKey(StringToBinary("CryptPassword"), $iAlgorithm) ; Повторно объявить парольную строку и алгоритм для создания нового криптографического ключа.

                Local $sRead = GUICtrlRead($g_idEdt_Input)
                If StringStripWS($sRead, $STR_STRIPALL) <> "" Then ; Проверить, есть ли текст для шифрования.
                    Local $dEncrypted = _Crypt_EncryptData($sRead, $g_hKey, $CALG_USERKEY) ; Зашифровать текст новым криптографическим ключом.
                    GUICtrlSetData($g_idEdt_Output, $dEncrypted) ; Установить поле вывода с зашифрованным текстом.
                    Local $dDecrypted = _Crypt_DecryptData($dEncrypted, $g_hKey, $CALG_USERKEY) ; Расшифровать текст новым криптографическим ключом.
                    GUICtrlSetData($g_idEdt_OutputDeCrypted, BinaryToString($dDecrypted)) ; Установить поле вывода с зашифрованным текстом.
                EndIf
        EndSwitch
    WEnd

    GUIDelete($hGUI) ; Удалить предыдущий GUI и все элементы управления.
    _Crypt_DestroyKey($g_hKey) ; Уничтожить криптографический ключ.
    _Crypt_Shutdown() ; Завершить работу библиотеки Crypt.
EndFunc   ;==>Example

Func WM_COMMAND($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $lParam

    Switch _WinAPI_LoWord($wParam)
        Case $g_idEdt_Input
            Switch _WinAPI_HiWord($wParam)
                Case $EN_CHANGE
                    Local $dEncrypted = _Crypt_EncryptData(GUICtrlRead($g_idEdt_Input), $g_hKey, $CALG_USERKEY) ; Зашифровать текст криптографическим ключом.
                    GUICtrlSetData($g_idEdt_Output, $dEncrypted) ; Установить поле вывода с зашифрованным текстом.
                    Local $dDecrypted = _Crypt_DecryptData($dEncrypted, $g_hKey, $CALG_USERKEY) ; Расшифровать текст новым криптографическим ключом.
                    GUICtrlSetData($g_idEdt_OutputDeCrypted, BinaryToString($dDecrypted)) ; Установить поле вывода с зашифрованным текстом.
            EndSwitch
    EndSwitch
EndFunc   ;==>WM_COMMAND

Пример 2

#include <Crypt.au3>
#include <MsgBoxConstants.au3>

Example()

Func Example()
    ; Зашифровать текст общим паролем.
    Local $dEncrypted = StringEncrypt(True, 'Encrypt this data.', 'securepassword')

    ; Показать зашифрованный текст.
    MsgBox($MB_SYSTEMMODAL, 'Encrypted', $dEncrypted)

    ; Расшифровать зашифрованный текст общим паролем.
    Local $sDecrypted = StringEncrypt(False, $dEncrypted, 'securepassword')

    ; Показать расшифрованный текст.
    MsgBox($MB_SYSTEMMODAL, 'Decrypted', $sDecrypted)
EndFunc   ;==>Example

Func StringEncrypt($bEncrypt, $sData, $sPassword)
    _Crypt_Startup() ; Запустить библиотеку Crypt.
    Local $vReturn = ''
    If $bEncrypt Then ; Если флаг True – шифруем, иначе – расшифровываем.
        $vReturn = _Crypt_EncryptData($sData, $sPassword, $CALG_RC4)
    Else
        $vReturn = BinaryToString(_Crypt_DecryptData($sData, $sPassword, $CALG_RC4))
    EndIf
    _Crypt_Shutdown() ; Завершить работу библиотеки Crypt.
    Return $vReturn
EndFunc   ;==>StringEncrypt

Пример 3

; == Пример 3 обработка строки не-ANSI

#include <Crypt.au3>
#include <MsgBoxConstants.au3>
#include <StringConstants.au3>

Local $sPlaintext = "Hello! ជំរាបសួរ! Allô! Привет! 您好！مرحبا! હેલો! שלום! こんにちは！"

Local $dPlaintextUTF8 = StringToBinary($sPlaintext, $SB_UTF8) ; Преобразовать в бинарную строку с конвертацией Unicode-символов в UTF8
;~ $dPlaintextUTF8 = $sPlaintext ; If Uncommented willshow why UTF8 conversion is needed

Local $iAlgorithm = $CALG_3DES
Local $hKey = _Crypt_DeriveKey("CryptPassword", $iAlgorithm)

Local $dEncrypted = _Crypt_EncryptData($dPlaintextUTF8, $hKey, $CALG_USERKEY) ; Зашифровать текст новым криптографическим ключом.

Local $dDecrypted = _Crypt_DecryptData($dEncrypted, $hKey, $CALG_USERKEY) ; Расшифровать данные общей парольной строкой. Возвращает бинарную строку.

Local $sDecrypted = BinaryToString($dDecrypted, $SB_UTF8) ; Преобразовать бинарную строку через BinaryToString и показать исходные зашифрованные данные.

If $sPlaintext = $sDecrypted Then
    MsgBox($MB_SYSTEMMODAL, "Decrypted data", $sDecrypted)
Else
    MsgBox($MB_SYSTEMMODAL, "BAD Decrypted data", $sPlaintext & @CRLF & "-->" & @CRLF & $sDecrypted)
EndIf
