Справочник функций

_GUIToolTip_GetCurrentTool

Получает информацию о текущем инструменте.

#include <GuiToolTip.au3>
_GUIToolTip_GetCurrentTool ( $hTool )

Параметры

$hTool Дескриптор элемента управления ToolTip (возвращается функцией _GUIToolTip_Create).

Возвращаемое значение

Успех: Возвращает массив со следующим форматом:
$aArray[0] – флаги контролирующие отображение подсказки. Может быть комбинацией следующих значений:
$TTF_IDISHWND = $iID содержит дескриптор окна инструмента вместо ID
$TTF_CENTERTIP = Подсказка центрируется под инструментом указанным в $iID
$TTF_RTLREADING = Текст отображается в противоположном направлении
$TTF_SUBCLASS = Элемент управления выполняет подклассирование окна инструмента для перехвата сообщений
$TTF_TRACK = Позиционирует элемент управления рядом с инструментом к которому он относится
$TTF_ABSOLUTE = Позиционирует окно с координатами переданными в TTM_TRACKPOSITION
$TTF_TRANSPARENT = Заставляет элемент управления передавать сообщения мыши окну-родителю
$TTF_PARSELINKS = Указывает что ссылки в тексте подсказки должны быть обработаны
$aArray[1] – дескриптор окна содержащего инструмент ($hWnd)
$aArray[2] – идентификатор инструмента ($iID)
$aArray[3] – координата X верхнего левого угла прямоугольника
$aArray[4] – координата Y верхнего левого угла прямоугольника
$aArray[5] – координата X нижнего правого угла прямоугольника
$aArray[6] – координата Y нижнего правого угла прямоугольника
$aArray[7] – дескриптор экземпляра содержащего ресурс строки для инструмента
$aArray[8] – текст для инструмента
$aArray[9] – определяемое приложением значение связанное с инструментом
Ошибка: Устанавливает @error (см. раздел Примечания).

Примечания

Если элемент управления находится в другом процессе и оба процесса работают в разных режимах AutoIt (@AutoItVersion), @error будет установлена на 6.

Подсказка должна отображаться когда эта функция вызывается, иначе возвращённые результаты будут нули.
Флаги можно конвертировать в читаемую строку используя функцию _GUIToolTip_BitsToTTF().

См. также

_GUIToolTip_EnumTools, _GUIToolTip_GetToolInfo, _GUIToolTip_BitsToTTF

Пример

Пример 1

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <GUIToolTip.au3>
#include <MsgBoxConstants.au3>

Global $g_hToolTip

; Наведите мышь на кнопку в GUI и нажмите клавишу "g" чтобы отобразить информацию
HotKeySet("g", "Get_Tool")

Example()

Func Example()
    Local $hGUI = GUICreate("ToolTip Get Current Tool v(" & @AutoItVersion & ")", 450, 300, 100, 100)

    ; создать элемент tooltip с настройками по умолчанию
    $g_hToolTip = _GUIToolTip_Create(0)
    _MemoSetHandleInProcess($g_hToolTip)

    Local $idButton = GUICtrlCreateButton("Button ToolTip", 30, 32, 130, 28)
    Local $hButton = GUICtrlGetHandle($idButton)

;~ $hGUI = 0 ; is OK
    ; добавить tool в элемент tooltip
    _GUIToolTip_AddTool($g_hToolTip, $hGUI, "This is a ToolTip", $hButton)

    GUISetState(@SW_SHOW)

    ; Показать tooltip, связанный с кнопкой
    Opt("MouseCoordMode", 2)
    MouseMove(50, 42, 0)
    Sleep(250)

    Get_Tool()

    While 1
        If GUIGetMsg() = $GUI_EVENT_CLOSE Then ExitLoop
    WEnd

    ; Уничтожить элемент tooltip
    _MemoResetHandleInProcess($g_hToolTip)
    _GUIToolTip_Destroy($g_hToolTip)
    GUIDelete($hGUI)
EndFunc   ;==>Example

Func Get_Tool()
    Local $aTool = _GUIToolTip_GetCurrentTool($g_hToolTip)
    If @error Then
        _MemoMsgBox($MB_ICONERROR, "Tooltip info @error = " & @error, "No Tooltip displayed")
    Else
        _MemoMsgBox($MB_SYSTEMMODAL, "Tooltip info", "Flags: " & @TAB & _GUIToolTip_BitsToTTF($aTool[0]) & @CRLF & _
                "HWnd: " & @TAB & $aTool[1] & @CRLF & _
                "ID: " & @TAB & "0x" & Hex($aTool[2]) & @CRLF & _
                "Left X:" & @TAB & $aTool[3] & @CRLF & _
                "Left Y:" & @TAB & $aTool[4] & @CRLF & _
                "Right X:" & @TAB & $aTool[5] & @CRLF & _
                "Right Y:" & @TAB & $aTool[6] & @CRLF & _
                "Instance:" & @TAB & $aTool[7] & @CRLF & _
                "Text:" & @TAB & $aTool[8] & @CRLF & _
                "lParam:" & @TAB & $aTool[9])
    EndIf

EndFunc   ;==>Get_Tool

Пример 2

#include <GUIConstantsEx.au3>
#include <GUIToolTip.au3>
#include <MsgBoxConstants.au3>

Global $g_iPID

; Наведите мышь на один из символов в приложении Charmap чтобы отобразить подсказку, затем нажмите 'g' чтобы
; получить его информацию.
HotKeySet('g', "_Read_Tip")

Example()

Func Example()
    ; Запустить программу таблицы символов
    $g_iPID = Run("charmap.exe")
    ; Дождитесь пока оно станет активным окном
;~ WinWaitActive("Character Map", "", 10)
    WinWaitActive("Table des caract", "", 10)

    ; Показать tooltip, связанный с кнопкой
    Opt("MouseCoordMode", 2)
    MouseMove(50, 72, 0)
    Sleep(250)

     _Read_Tip()

    While ProcessExists($g_iPID)
        Sleep(100)
    WEnd
EndFunc   ;==>Example

Func _Read_Tip()
    ; Получить список подсказок
    Local $aTipList = WinList("[CLASS:tooltips_class32]")
    WinWaitActive("Table des caract", "", 10)

    ; Посмотрите какие принадлежат вашему приложению
    For $i = 1 To $aTipList[0][0]
        If WinGetProcess($aTipList[$i][1]) = $g_iPID Then ; Посмотрите какая активна для вашего приложения
            ; Если какая-то активна то отобразите её
            Local $aTool = _GUIToolTip_GetCurrentTool($aTipList[$i][1])
            MsgBox($MB_SYSTEMMODAL, "Tooltip info $i = " & $i, "Flags: " & @TAB & _GUIToolTip_BitsToTTF($aTool[0]) & @CRLF & _
                    "HWnd: " & @TAB & $aTool[1] & @CRLF & _
                    "ID: " & @TAB & $aTool[2] & @CRLF & _
                    "Left X:" & @TAB & $aTool[3] & @CRLF & _
                    "Left Y:" & @TAB & $aTool[4] & @CRLF & _
                    "Right X:" & @TAB & $aTool[5] & @CRLF & _
                    "Right Y:" & @TAB & $aTool[6] & @CRLF & _
                    "Instance:" & @TAB & $aTool[7] & @CRLF & _
                    "Text:" & @TAB & $aTool[8] & @TAB & @TAB & @CRLF & _
                    "lParam:" & @TAB & $aTool[9])
        EndIf
    Next
EndFunc   ;==>_Read_Tip
