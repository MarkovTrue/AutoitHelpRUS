Справочник функций

_GUICtrlTab_SetItemImage

Устанавливает изображение.

#include <GuiTab.au3>
_GUICtrlTab_SetItemImage ( $hWnd, $iIndex, $iImage )

Параметры

$hWnd Идентификатор или дескриптор элемента управления.
$iIndex Индекс элемента (нумерация с 0).
$iImage Индекс изображения элемента (нумерация с 0).

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False.

См. также

_GUICtrlTab_GetItemImage

Пример

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <GuiImageList.au3>
#include <GuiTab.au3>
#include <MsgBoxConstants.au3>
#include <WinAPIGdi.au3>

Example()

Func Example()
    ; Создать GUI
    Local $hGUI = GUICreate("Tab Get/Set Item Image (v" & @AutoItVersion & ")", 450, 300, 100, 100)
    Local $idTab = GUICtrlCreateTab(2, 2, 446, 266)
    _MemoMsgBoxStatus() ; Создание статус-бара

    GUISetState(@SW_SHOW)

    ; Создать изображения
    Local $hImage = _GUIImageList_Create()
    _GUIImageList_Add($hImage, _WinAPI_CreateSolidBitmap($hGUI, 0xFF0000, 16, 16))
    _GUIImageList_Add($hImage, _WinAPI_CreateSolidBitmap($hGUI, 0x00FF00, 16, 16))
    _GUIImageList_Add($hImage, _WinAPI_CreateSolidBitmap($hGUI, 0x0000FF, 16, 16))
    _GUICtrlTab_SetImageList($idTab, $hImage)

    ; Добавить вкладки
    _GUICtrlTab_InsertItem($idTab, 0, "Tab 0")
    _GUICtrlTab_InsertItem($idTab, 1, "Tab 1")
    _GUICtrlTab_InsertItem($idTab, 2, "Tab 2")

    ; Получить/установить изображение вкладки 1
    _GUICtrlTab_SetItemImage($idTab, 1, 1)
    _MemoMsgBox($MB_SYSTEMMODAL, "Information", "Tab 1 image index: " & _GUICtrlTab_GetItemImage($idTab, 1))

    _MemoMsgBoxStatus("", -1, $hGUI) ; больше действий нет, ждём закрытия GUI

EndFunc   ;==>Example
