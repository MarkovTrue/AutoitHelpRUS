_GUIToolTip_TrackActivate

Активирует или деактивирует отслеживающую подсказку.

#include <GuiToolTip.au3>
_GUIToolTip_TrackActivate ( $hTool [, $bActivate = True [, $hWnd = 0 [, $iID = 0]]] )

Параметры

$hTool Дескриптор элемента управления ToolTip (возвращается функцией _GUIToolTip_Create).
$bActivate [необязательный] True для активации, False для деактивации
$hWnd Дескриптор окна.
$iID [необязательный] дескриптор элемента управления, которому назначен инструмент, или определённый приложением идентификатор инструмента

Возвращаемое значение

Отсутствует.

Примечания

Вы должны использовать эту функцию для активации отслеживаемой подсказки перед попыткой обновления её положения с помощью _GUIToolTip_TrackPosition()
Если вы используете _GUIToolTip_TrackPosition() без использования _GUIToolTip_TrackActivate(), инструмент будет отображаться только если мышь находится над назначенным окном или элементом управления, если применимо.

См. также

_GUIToolTip_TrackPosition

Пример

Пример 1

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <GUIToolTip.au3>

Example()

Func Example()
    Local $hGUI = GUICreate("ToolTip Track Activate v(" & @AutoItVersion & ")", 450, 300, 100, 100)

    Local $hToolTip = _GUIToolTip_Create(0, $TTS_BALLOON)
    _MemoSetHandleInProcess($hToolTip)

    Local $idButton = GUICtrlCreateButton("Button", 30, 32, 130, 28)
    Local $hButton = GUICtrlGetHandle($idButton)

;~ $hGUI = 0 ; Is OK
;~ $hButton = 0 ; Is OK

    ; Добавить инструмент к элементу управления ToolTip, не используя элемент управления или GUI для связи
    _GUIToolTip_AddTool($hToolTip, $hGUI, " ", $hButton, 0, 0, 0, 0, $TTF_SUBCLASS)

    _GUIToolTip_SetTitle($hToolTip, 'Mouse position', $TTI_INFO)

    GUISetState(@SW_SHOW)

    ; Показать tooltip, связанный с кнопкой
    Opt("MouseCoordMode", 2)
    MouseMove(50, 42, 0)
    Sleep(250)

    _GUIToolTip_TrackActivate($hToolTip, True, $hGUI, $hButton)

    Local $aPos, $iOldaPos0, $iOldaPos1
    While 1
        Sleep(10)
        $aPos = MouseGetPos()
        If $aPos[0] <> $iOldaPos0 Or $aPos[1] <> $iOldaPos1 Then
            _GUIToolTip_TrackPosition($hToolTip, $aPos[0] + 10, $aPos[1] + 20)
            _GUIToolTip_UpdateTipText($hToolTip, $hGUI, $hButton, "X: " & $aPos[0] & " Y: " & $aPos[1])
            $iOldaPos0 = $aPos[0]
            $iOldaPos1 = $aPos[1]
        EndIf
        If GUIGetMsg() = $GUI_EVENT_CLOSE Then ExitLoop
    WEnd

    ; Уничтожить элемент tooltip
    _MemoResetHandleInProcess($hTooltip)
    _GUIToolTip_Destroy($hToolTip)
    GUIDelete($hGUI)
EndFunc   ;==>Example

; == Пример 2: ToolTip – активация отслеживания подсказки на внешнем процессе

#include "Extras\HelpFileInternals.au3"

#include <GUIToolTip.au3>

Example()

Func Example()
    Local $sFromTo
    Local $hWin = _MemoRunAU3OutProcess($sFromTo) ; OK также в одинаковом режиме
    Local $hButton = _MemoCreateOutProcess($hWin, "Button", 0, $sFromTo)

    Local $hToolTip = _MemoGetHandleInProcess()

    _GUIToolTip_TrackActivate($hToolTip, False, $hWin, $hButton)
    If @error Then
        _MemoMsgBox($MB_ICONERROR, "Info" & $sFromTo, "_GUIToolTip_TrackActivate()" & " @error = " & @error & @CRLF & _
                @TAB & "cannot be accessed from an external process" & @CRLF & _
                @TAB & "running in different mode")
    Else
        _MemoMsgBox($MB_SYSTEMMODAL, "ToolTip TrackActivate" & $sFromTo, "Deactivate")
    EndIf

    _MemoMsgBoxStatus("", Default, $hWin) ; больше действий нет, ждём закрытия GUI, закрываем и внешний GUI

EndFunc   ;==>Example
