Справочник функций

_WinAPI_DefSubclassProc

Вызывает следующий обработчик в цепочке подклассов окна.

#include <WinAPIShellEx.au3>
_WinAPI_DefSubclassProc ( $hWnd, $iMsg, $wParam, $lParam )

Параметры

$hWnd Дескриптор окна.
$iMsg Отправляемое сообщение.
$wParam Первое слово предложения сообщение-specific information.
$lParam Первое слово предложения сообщение-specific information.

Возвращаемое значение

Возвращает значение, специфичное для отправленного сообщения.

Примечания

Последний обработчик в цепочке подклассов – исходная оконная процедура указанного окна.
Вызывать оконную процедуру по умолчанию вручную не нужно: эта функция делает это автоматически.

См. также

_WinAPI_SetWindowSubclass

Дополнительно

Искать DefSubclassProc в библиотеке MSDN

Пример

#include "Extras\WMDebug.au3"

#include <GUIConstantsEx.au3>
#include <WinAPIShellEx.au3>

OnAutoItExitRegister('OnAutoItExit')

; Создать GUI
Global $g_hForm = GUICreate('Test ' & StringReplace(@ScriptName, '.au3', '()'))

; Зарегистрировать DLL callback для использования как процедуру подкласса окна
Global $g_hDll = DllCallbackRegister('_SubclassProc', 'lresult', 'hwnd;uint;wparam;lparam;uint_ptr;dword_ptr')
Global $g_pDll = DllCallbackGetPtr($g_hDll)

; Установить callback подкласса окна
_WinAPI_SetWindowSubclass($g_hForm, $g_pDll, 1000, 0)

GUISetState(@SW_SHOW)

Do
Until GUIGetMsg() = $GUI_EVENT_CLOSE

Func _SubclassProc($hWnd, $iMsg, $wParam, $lParam, $iID, $pData)
    #forceref $iID, $pData
    ; Объявлено в WMDebug.au3
    _WM_Debug($hWnd, $iMsg, $wParam, $lParam)
    Return _WinAPI_DefSubclassProc($hWnd, $iMsg, $wParam, $lParam)
EndFunc   ;==>_SubclassProc

Func OnAutoItExit()
    _WinAPI_RemoveWindowSubclass($g_hForm, $g_pDll, 1000)
    DllCallbackFree($g_hDll)
EndFunc   ;==>OnAutoItExit
