Справочник функций

_GUICtrlComboBox_Create

Создаёт элемент управления ComboBox.

#include <GuiComboBox.au3>
_GUICtrlComboBox_Create ( $hWnd, $sText, $iX, $iY [, $iWidth = 100 [, $iHeight = 120 [, $iStyle = 0x00200042 [, $iExStyle = 0x00000000]]]] )

Параметры

$hWnd Дескриптор родительского или владеющего окна.
$sText Строка с разделителями для добавления в выпадающий список.
$iX Горизонтальная позиция элемента управления.
$iY Вертикальная позиция элемента управления.
$iWidth [необязательный] Ширина элемента управления
$iHeight [необязательный] Высота элемента управления
$iStyle [необязательный] Стиль элемента управления:
$CBS_AUTOHSCROLL – автоматически прокручивает текст в поле ввода вправо, когда пользователь вводит символ в конце строки.
$CBS_DISABLENOSCROLL – показывает отключённую вертикальную полосу прокрутки
$CBS_DROPDOWN – похож на $CBS_SIMPLE, за исключением того, что список не отображается, пока пользователь не выберет значок рядом с полем ввода
$CBS_DROPDOWNLIST – похож на $CBS_DROPDOWN, за исключением того, что поле ввода заменяется на статический элемент текста, отображающий текущий выбор из списка
$CBS_LOWERCASE – преобразует весь текст в оба поля (выбор и список) в нижний регистр
$CBS_NOINTEGRALHEIGHT – задаёт, что размер выпадающего списка точно равен размеру, указанному приложением при его создании
$CBS_OEMCONVERT – преобразует текст, введённый в поле ввода выпадающего списка из набора символов Windows в набор символов OEM и обратно в набор символов Windows
$CBS_OWNERDRAWFIXED – задаёт, что владелец списка отвечает за рисование его содержимого, а все элементы списка имеют одинаковую высоту
$CBS_OWNERDRAWVARIABLE – задаёт, что владелец списка отвечает за рисование его содержимого, а элементы списка имеют переменную высоту
$CBS_SIMPLE – отображает список всё время
$CBS_SORT – автоматически сортирует добавленные строки
$CBS_UPPERCASE – преобразует весь текст в оба поля (выбор и список) в верхний регистр

По умолчанию: $CBS_DROPDOWN, $CBS_AUTOHSCROLL, $WS_VSCROLL
Принудительно: $WS_CHILD, $WS_TABSTOP, $WS_VISIBLE.
$iExStyle [необязательный] Расширенный стиль элемента. Соответствует стандартным константам $WS_EX_*. См. таблицу расширенных стилей.

Возвращаемое значение

Успех: Возвращает дескриптор элемента управления ComboBox.
Ошибка: Возвращает 0.

Примечания

Данная функция предназначена для опытных пользователей и для изучения принципов работы элемента управления.

См. также

_GUICtrlComboBox_Destroy

Пример

Example : Created with UDF

; == Пример: создано с помощью UDF

#include "Extras\WM_NOTIFY.au3"

#include <DirConstants.au3>
#include <GuiComboBox.au3>
#include <GUIConstantsEx.au3>
#include <WindowsNotifsConstants.au3>

Global $g_hCombo

Example()

Func Example()
    ; Создать GUI
    Local $hGUI = GUICreate("ComboBox Create (v" & @AutoItVersion & ")", 400, 296)
    $g_hCombo = _GUICtrlComboBox_Create($hGUI, "", 2, 2, 396, 296)
    GUISetState(@SW_SHOW)

    ; Добавить файлы
    _GUICtrlComboBox_BeginUpdate($g_hCombo)
    _GUICtrlComboBox_AddDir($g_hCombo, "", $DDL_DRIVES, False)
    _GUICtrlComboBox_EndUpdate($g_hCombo)

    GUIRegisterMsg($WM_COMMAND, "WM_COMMAND")

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
    GUIDelete()
EndFunc   ;==>Example

Func WM_COMMAND($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg
    Local $hWndFrom = $lParam
    Local $iIDFrom = BitAND($wParam, 0xFFFF) ; Младшее слово
    Local $iCode = BitShift($wParam, 16) ; Старшее слово
    Switch $hWndFrom
        Case $g_hCombo
            Switch $iCode
                Case $CBN_CLOSEUP ; Отправляется при закрытии списка combo box
                    _WM_NOTIFY_DebugInfo("$CBN_CLOSEUP", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_DBLCLK ; Отправляется при двойном клике по строке в списке combo box
                    _WM_NOTIFY_DebugInfo("$CBN_DBLCLK", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_DROPDOWN ; Отправляется перед показом списка combo box
                    _WM_NOTIFY_DebugInfo("$CBN_DROPDOWN", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_EDITCHANGE ; Отправляется после действия пользователя, которое могло изменить текст в edit-части combo box
                    _WM_NOTIFY_DebugInfo("$CBN_EDITCHANGE", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_EDITUPDATE ; Отправляется перед показом изменённого текста в edit-части combo box
                    _WM_NOTIFY_DebugInfo("$CBN_EDITUPDATE", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_ERRSPACE ; Отправляется, когда combo box не может выделить достаточно памяти для запроса
                    _WM_NOTIFY_DebugInfo("$CBN_ERRSPACE", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_KILLFOCUS ; Отправляется при потере клавиатурного фокуса combo box
                    _WM_NOTIFY_DebugInfo("$CBN_KILLFOCUS", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_SELCHANGE ; Отправляется при изменении текущего выбора в списке combo box
                    _WM_NOTIFY_DebugInfo("$CBN_SELCHANGE", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_SELENDCANCEL ; Отправляется при выборе элемента, затем переключении на другой контрол или закрытии диалога
                    _WM_NOTIFY_DebugInfo("$CBN_SELENDCANCEL", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_SELENDOK ; Отправляется при выборе элемента списка или при выборе с последующим закрытием списка
                    _WM_NOTIFY_DebugInfo("$CBN_SELENDOK", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $CBN_SETFOCUS ; Отправляется при получении клавиатурного фокуса combo box
                    _WM_NOTIFY_DebugInfo("$CBN_SETFOCUS", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
            EndSwitch
    EndSwitch
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_COMMAND
