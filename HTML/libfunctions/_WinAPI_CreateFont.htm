_WinAPI_CreateFont

создаёт логический шрифт с указанными характеристиками.

#include <WinAPIGdi.au3>
_WinAPI_CreateFont ( $iHeight, $iWidth [, $iEscape = 0 [, $iOrientn = 0 [, $iWeight = $FW_NORMAL [, $bItalic = False [, $bUnderline = False [, $bStrikeout = False [, $iCharset = $DEFAULT_CHARSET [, $iOutputPrec = $OUT_DEFAULT_PRECIS [, $iClipPrec = $CLIP_DEFAULT_PRECIS [, $iQuality = $DEFAULT_QUALITY [, $iPitch = $FF_DONTCARE [, $sFace = 'Arial']]]]]]]]]]]] )

Параметры

$iHeight высота шрифта
$iWidth средняя ширина символа
$iEscape Необязательный параметр.
$iOrientn Необязательный параметр.
$iWeight Координаты.
$bItalic Необязательный параметр.
$bUnderline Необязательный параметр.
$bStrikeout Необязательный параметр.
$iCharset Имя.
$iOutputPrec Имя.
$iClipPrec Количество / Размер.
$iQuality Координаты.
$iPitch Координаты.
$sFace Имя.

Возвращаемое значение

Успех: Возвращает дескриптор на созданный шрифт.
Ошибка: Возвращает 0.

Примечания

Когда шрифт больше не нужен, удалите его вызовом _WinAPI_DeleteObject().

Перечисленные константы требуют #include <FontConstants.au3>

Дополнительно

Искать CreateFont в библиотеке MSDN

Пример

#include <FontConstants.au3>
#include <StructureConstants.au3>
#include <WinAPIGdi.au3>
#include <WinAPIGdiDC.au3>
#include <WinAPIHObj.au3>
#include <WinAPISysWin.au3>

Global $g_tRECT, $g_hFont, $g_hOldFont, $g_hDC

HotKeySet("{ESC}", "_Exit")

$g_tRECT = DllStructCreate($tagRect)
DllStructSetData($g_tRECT, "Left", 5)
DllStructSetData($g_tRECT, "Top", 5)
DllStructSetData($g_tRECT, "Right", 250)
DllStructSetData($g_tRECT, "Bottom", 50)

$g_hDC = _WinAPI_GetDC(0)
$g_hFont = _WinAPI_CreateFont(50, 0, 0, 0, 400, False, False, False, $DEFAULT_CHARSET, _
        $OUT_DEFAULT_PRECIS, $CLIP_DEFAULT_PRECIS, $DEFAULT_QUALITY, 0, 'Arial')
$g_hOldFont = _WinAPI_SelectObject($g_hDC, $g_hFont)

_WinAPI_SetTextColor($g_hDC, 0x0000FF)
_WinAPI_SetBkColor($g_hDC, 0x000000)
; закомментируйте следующую строку для чёрного фона вместо прозрачного
_WinAPI_SetBkMode($g_hDC, $TRANSPARENT)

While 1
    _WinAPI_DrawText($g_hDC, "Hello world!", $g_tRECT, $DT_CENTER)
    Sleep(100)
WEnd

Func _Exit()
    _WinAPI_SelectObject($g_hDC, $g_hOldFont)
    _WinAPI_DeleteObject($g_hFont)
    _WinAPI_ReleaseDC(0, $g_hDC)
    _WinAPI_InvalidateRect(0, 0)
    $g_tRECT = 0
    Exit
EndFunc   ;==>_Exit
