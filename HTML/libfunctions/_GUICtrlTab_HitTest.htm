Справочник функций

_GUICtrlTab_HitTest

Получает диапазон.

#include <GuiTab.au3>
_GUICtrlTab_HitTest ( $hWnd, $iX, $iY )

Параметры

$hWnd Идентификатор или дескриптор элемента управления.
$iX Проверяемая координата X.
$iY Проверяемая координата Y.

Возвращаемое значение

Возвращает индекс.

Пример

Пример 1

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <GuiTab.au3>
#include <MsgBoxConstants.au3>

Example()

Func Example()
    ; Создать GUI
    Local $hGUI = GUICreate("Tab Control HitTest (v" & @AutoItVersion & ")", 450, 300, 100, 100)
    Local $idTab = GUICtrlCreateTab(2, 2, 446, 266)
    _MemoMsgBoxStatus() ; Создание статус-бара

    GUISetState(@SW_SHOW)

    ; Добавить вкладки
    _GUICtrlTab_InsertItem($idTab, 0, "Tab 0")
    _GUICtrlTab_InsertItem($idTab, 1, "Tab 1")
    _GUICtrlTab_InsertItem($idTab, 2, "Tab 2")

    ; Выполнить тест попадания
    Local $aHit = _GUICtrlTab_HitTest($idTab, 80, 10)
    _MemoMsgBox($MB_SYSTEMMODAL, "Information", "Point [80,10] is over tab " & $aHit[0])

    _MemoMsgBoxStatus("", -1, $hGUI) ; больше действий нет, ждём закрытия GUI

EndFunc   ;==>Example

Example (OutProcess) : Tab HitTest to an External process

; == Example (OutProcess) : Tab HitTest to an External process

#include "Extras\HelpFileInternals.au3"

#include <GuiTab.au3>
#include <MsgBoxConstants.au3>

Example()

Func Example()
    Local $sFromTo
    Local $hWin = _MemoRunAU3OutProcess($sFromTo, False) ; OK также при запуске в другом режиме
    Local $hTabCtrl = _MemoCreateOutProcess($hWin, "SysTabControl32", 1, $sFromTo)

    Local $aHit = _GUICtrlTab_HitTest($hTabCtrl, 100, 10)
    _MemoMsgBox($MB_SYSTEMMODAL, "Info" & $sFromTo, "Point [100,10] is over tab " & $aHit[0])

    _MemoMsgBoxStatus("", -1, $hWin) ; больше действий нет, ждём закрытия GUI

EndFunc   ;==>Example
