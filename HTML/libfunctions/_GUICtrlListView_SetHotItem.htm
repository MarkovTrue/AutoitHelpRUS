Справочник функций

_GUICtrlListView_SetHotItem

Устанавливает активный элемент.

#include <GuiListView.au3>
_GUICtrlListView_SetHotItem ( $hWnd, $iIndex )

Параметры

$hWnd Идентификатор или дескриптор элемента управления.
$iIndex Индекс элемента (нумерация с 0).

Возвращаемое значение

Успех: Возвращает индекс.
Ошибка: Возвращает -1.

См. также

_GUICtrlListView_GetHotItem

Пример

#include "Extras\WM_NOTIFY.au3"

#include <GUIConstantsEx.au3>
#include <GuiListView.au3>
#include <GuiStatusBar.au3>
#include <MsgBoxConstants.au3>
#include <StructureConstants.au3>
#include <WindowsNotifsConstants.au3>

Global $g_idListView, $g_hStatus

Example()

Func Example()
    Local $hGUI = GUICreate("ListView Get/Set Hot Item (v" & @AutoItVersion & ")", 392, 322)

    $g_idListView = GUICtrlCreateListView("", 2, 2, 394, 268)
    $g_idListView = GUICtrlGetHandle($g_idListView)
    _GUICtrlListView_SetExtendedListViewStyle($g_idListView, BitOR($LVS_EX_FULLROWSELECT, $LVS_EX_GRIDLINES, $LVS_EX_TRACKSELECT))
    $g_hStatus = _GUICtrlStatusBar_Create($hGUI)
    _GUICtrlStatusBar_SetSimple($g_hStatus, True)
    GUISetState(@SW_SHOW)

    ; Добавить столбцы
    _GUICtrlListView_AddColumn($g_idListView, "Column 0", 100)
    _GUICtrlListView_AddColumn($g_idListView, "Column 1", 100)
    _GUICtrlListView_AddColumn($g_idListView, "Column 2", 100)

    _GUICtrlListView_InsertItem($g_idListView, "Row 0: Col 0", -1, 0)
    _GUICtrlListView_AddSubItem($g_idListView, 0, "Row 0: Col 1", 1, 1)
    _GUICtrlListView_AddSubItem($g_idListView, 0, "Row 0: Col 2", 2, 2)
    _GUICtrlListView_InsertItem($g_idListView, "Row 1: Col 0", -1, 1)
    _GUICtrlListView_AddSubItem($g_idListView, 1, "Row 1: Col 1", 1, 2)
    _GUICtrlListView_InsertItem($g_idListView, "Row 2: Col 0", -1, 2)

    ; Установить активный элемент
    _GUICtrlListView_SetHotItem($g_idListView, 1)
    MsgBox($MB_SYSTEMMODAL, "Information", "Hot Item: " & _GUICtrlListView_GetHotItem($g_idListView))

    ;Зарегистрировать события WM_NOTIFY
    _WM_NOTIFY_Register()

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
    GUIDelete()
EndFunc   ;==>Example

Func ListView_HOTTRACK($iSubItem)
    Local $iHotItem = _GUICtrlListView_GetHotItem($g_idListView)
    If $iHotItem <> -1 Then _GUICtrlStatusBar_SetText($g_hStatus, "Hot Item: " & $iHotItem & " SubItem: " & $iSubItem)
EndFunc   ;==>ListView_HOTTRACK

Func WM_NOTIFY($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $wParam
    Local $hWndListView = $g_idListView
    If Not IsHWnd($g_idListView) Then $hWndListView = GUICtrlGetHandle($g_idListView)

    Local $tNMHDR = DllStructCreate($tagNMHDR, $lParam)
    Local $hWndFrom = HWnd(DllStructGetData($tNMHDR, "hWndFrom"))
    Local $iCode = DllStructGetData($tNMHDR, "Code")
    Switch $hWndFrom
        Case $hWndListView
            Switch $iCode
                Case $LVN_COLUMNCLICK ; Был кликнут столбец
                    _WM_NOTIFY_DebugEvent("$LVN_COLUMNCLICK", $tagNMLISTVIEW, $lParam, "IDFrom,,Item,SubItem,NewState,OldState,Changed,ActionX,ActionY,Param")
                    ; Нет возвращаемого значения
                Case $LVN_DELETEITEM ; Элемент будет удалён
                    _WM_NOTIFY_DebugEvent("$LVN_DELETEITEM", $tagNMLISTVIEW, $lParam, "IDFrom,,Item,SubItem,NewState,OldState,Changed,ActionX,ActionY,Param")
                    ; Нет возвращаемого значения
                Case $LVN_HOTTRACK ; Отправляется list-view элементом при наведении мыши на элемент
                    Local $tInfo = DllStructCreate($tagNMLISTVIEW, $lParam)
                    ListView_HOTTRACK(DllStructGetData($tInfo, "SubItem"))
                    Return 0 ; разрешить list-view выполнить стандартную обработку track select.
                    ; Return 1 – элемент не будет выбран.
                Case $LVN_KEYDOWN ; Нажата клавиша
                    _WM_NOTIFY_DebugEvent("$LVN_KEYDOWN", $tagNMLVKEYDOWN, $lParam, "IDFrom,,VKey,Flags")
                    ; Нет возвращаемого значения
                Case $NM_CLICK ; Отправляется элементом list-view, когда пользователь кликает по элементу левой кнопкой мыши
                    _WM_NOTIFY_DebugEvent("$NM_CLICK", $tagNMITEMACTIVATE, $lParam, "IDFrom,,Index,SubItem,NewState,OldState,Changed,X,Y,lParam,KeyFlags")
                    ; Нет возвращаемого значения
                Case $NM_DBLCLK ; Отправляется элементом list-view при двойном клике левой кнопкой мыши
                    _WM_NOTIFY_DebugEvent("$NM_DBLCLK", $tagNMITEMACTIVATE, $lParam, "IDFrom,,Index,SubItem,NewState,OldState,Changed,X,Y,lParam,KeyFlags")
                    ; Нет возвращаемого значения
                Case $NM_KILLFOCUS ; Элемент управления потерял фокус ввода
                    _WM_NOTIFY_DebugEvent("$NM_KILLFOCUS", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Нет возвращаемого значения
                Case $NM_RCLICK ; Отправляется элементом list-view при клике правой кнопкой мыши
                    _WM_NOTIFY_DebugEvent("$NM_RCLICK", $tagNMITEMACTIVATE, $lParam, "IDFrom,,Index,SubItem,NewState,OldState,Changed,X,Y,lParam,KeyFlags")
                    ;Return 1 ; чтобы запретить стандартную обработку
                    Return 0 ; разрешить стандартную обработку
                Case $NM_RDBLCLK ; Отправляется элементом list-view при двойном клике правой кнопкой мыши
                    _WM_NOTIFY_DebugEvent("$NM_RDBLCLK", $tagNMITEMACTIVATE, $lParam, "IDFrom,,Index,SubItem,NewState,OldState,Changed,X,Y,lParam,KeyFlags")
                    ; Нет возвращаемого значения
                Case $NM_RETURN ; Элемент управления в фокусе и пользователь нажал ENTER
                    _WM_NOTIFY_DebugEvent("$NM_RETURN", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Нет возвращаемого значения
                Case $NM_SETFOCUS ; Элемент управления получил фокус ввода
                    _WM_NOTIFY_DebugEvent("$NM_SETFOCUS", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Нет возвращаемого значения
            EndSwitch
    EndSwitch
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_NOTIFY
