Справочник функций

_GUIToolTip_HitTest

Получает информацию о инструменте в заданной точке.

#include <GuiToolTip.au3>
_GUIToolTip_HitTest ( $hTool, $hWnd, $iX, $iY )

Параметры

$hTool Дескриптор элемента управления ToolTip (возвращается функцией _GUIToolTip_Create).
$hWnd Дескриптор окна.
$iX Проверяемая координата X.
$iY Проверяемая координата Y.

Возвращаемое значение

Успех: Возвращает массив со следующим форматом:
$aArray[0] – флаги управления отображением подсказки. Этот элемент может быть комбинацией следующих значений:
$TTF_IDISHWND – $iID содержит дескриптор окна инструмента вместо ID
$TTF_CENTERTIP – подсказка центрируется под инструментом
$TTF_RTLREADING – текст отображается в противоположном направлении
$TTF_SUBCLASS – элемент управления выполняет подклассирование окна инструмента
$TTF_TRACK – позиционирует элемент управления рядом с инструментом
$TTF_ABSOLUTE – позиционирует окно с координатами из TTM_TRACKPOSITION
$TTF_TRANSPARENT – передаёт сообщения мыши окну-родителю
$TTF_PARSELINKS – обрабатывает ссылки в тексте подсказки
$aArray[1] – дескриптор окна, содержащего инструмент
$aArray[2] – идентификатор инструмента
$aArray[3] – координата X верхнего левого угла прямоугольника
$aArray[4] – координата Y верхнего левого угла прямоугольника
$aArray[5] – координата X нижнего правого угла прямоугольника
$aArray[6] – координата Y нижнего правого угла прямоугольника
$aArray[7] – дескриптор экземпляра, содержащего ресурс строки для инструмента
$aArray[8] – текст инструмента
$aArray[9] – определённое приложением значение, связанное с инструментом
Ошибка: Устанавливает @error (см. раздел Примечания).

Примечания

Если элемент управления находится в другом процессе и оба процесса работают в разных режимах AutoIt (@AutoItVersion), @error будет установлена на 6.

Эту функцию можно использовать только если инструмент имеет установленный флаг $TTF_TRACK. _GUIToolTip_HitTest() завершится неудачей, если $TTF_TRACK не установлена, независимо от того, находится ли точка попадания в прямоугольник инструмента или нет.
Флаги можно преобразовать в понятную строку с помощью _GUIToolTip_BitsToTTF().

См. также

_GUIToolTip_BitsToTTF

Пример

Пример 1

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <GUIToolTip.au3>
#include <MsgBoxConstants.au3>
#include <WinAPIIcons.au3>

Global $g_hToolTip, $g_aPos, $g_hButton

; Нажмите G для отображения текущей информации подсказки.
HotKeySet("g", "GetInfo")

Example()

Func Example()
    Local $hGUI = GUICreate("ToolTip Hit Test v(" & @AutoItVersion & ")", 450, 300, 100, 100)

    $g_hToolTip = _GUIToolTip_Create(0, $TTS_CLOSE + $TTS_BALLOON)
    _MemoSetHandleInProcess($g_hToolTip)

    Local $idButton = GUICtrlCreateButton("Button ToolTip", 30, 32, 130, 28)
    $g_hButton = GUICtrlGetHandle($idButton)

;~ $hGUI = 0 ; is OK
    _GUIToolTip_AddTool($g_hToolTip, $hGUI, "X: " & @TAB & " Y: " & @TAB, $g_hButton)

    Local $hIcon = _WinAPI_LoadShell32Icon(15)

    _GUIToolTip_SetTitle($g_hToolTip, 'Title', $hIcon)

    GUISetState(@SW_SHOW)

    ; Показать tooltip, связанный с кнопкой
    Opt("MouseCoordMode", 2)
    MouseMove(50, 42, 0)
    Sleep(250)

    $g_aPos = MouseGetPos()
    _GUIToolTip_TrackPosition($g_hToolTip, $g_aPos[0] + 10, $g_aPos[1] + 20)
    _GUIToolTip_TrackActivate($g_hToolTip, True, $hGUI, $g_hButton)
    _GUIToolTip_UpdateTipText($g_hToolTip, $hGUI, $g_hButton, "X: " & $g_aPos[0] & " Y: " & $g_aPos[1])

    GetInfo()

    While 1
        Sleep(100)
        $g_aPos = MouseGetPos()
        _GUIToolTip_TrackPosition($g_hToolTip, $g_aPos[0] + 10, $g_aPos[1] + 20)
        _GUIToolTip_UpdateTipText($g_hToolTip, $hGUI, $g_hButton, "X: " & $g_aPos[0] & " Y: " & $g_aPos[1])
        If GUIGetMsg() = $GUI_EVENT_CLOSE Then ExitLoop
    WEnd

    _MemoResetHandleInProcess($g_hToolTip)
    _GUIToolTip_Destroy($g_hToolTip)
    GUIDelete($hGUI)
EndFunc   ;==>Example

Func GetInfo()
    Local $aTool = _GUIToolTip_HitTest($g_hToolTip, $g_hButton, $g_aPos[0], $g_aPos[1])
    _MemoMsgBox($MB_SYSTEMMODAL, "Tooltip info", "Flags: " & @TAB & _GUIToolTip_BitsToTTF($aTool[0]) & @CRLF & _
            "HWnd: " & @TAB & $aTool[1] & @CRLF & _
            "ID: " & @TAB & "0x" & $aTool[2] & @CRLF & _
            "Left X:" & @TAB & $aTool[3] & @CRLF & _
            "Left Y:" & @TAB & $aTool[4] & @CRLF & _
            "Right X:" & @TAB & $aTool[5] & @CRLF & _
            "Right Y:" & @TAB & $aTool[6] & @CRLF & _
            "Instance:" & @TAB & $aTool[7] & @CRLF & _
            "Text:" & @TAB & $aTool[8] & @CRLF & _
            "lParam:" & @TAB & $aTool[9])
EndFunc   ;==>GetInfo

; == Пример 2: ToolTip – проверка попадания указателя на внешний процесс

#include "Extras\HelpFileInternals.au3"

#include <GUIToolTip.au3>

Example()

Func Example()
    Local $sFromTo
    Local $hWin = _MemoRunAU3OutProcess($sFromTo, True) ; OK если одинаковый режим, KO если разный
    Local $hButton = _MemoCreateOutProcess($hWin, "Button", 1, $sFromTo)

    Local $hToolTip = _MemoGetHandleInProcess()

    Local $aPos = MouseGetPos()

    Local $aTool = _GUIToolTip_HitTest($hToolTip, $hButton, $aPos[0], $aPos[1])
    If @error Then
        _MemoMsgBox($MB_ICONERROR, "Info" & $sFromTo, "_GUIToolTip_HitTest()" & " @error = " & @error & @CRLF & _
                @TAB & "cannot be accessed from an external process" & @CRLF & _
                @TAB & "running in different mode")
    Else
        ; работает только если оба процесса запущены в одном режиме
        _MemoWrite("<<< Tooltip info" & $sFromTo & @CRLF & _
                "Flags: " & @TAB & _GUIToolTip_BitsToTTF($aTool[0]) & @CRLF & _
                "HWnd: " & @TAB & "0x" & Hex($aTool[1]) & @CRLF & _
                "ID: " & @TAB & "0x" & Hex($aTool[2]) & @CRLF & _
                "Left X:" & @TAB & @TAB & $aTool[3] & @CRLF & _
                "Left Y:" & @TAB & @TAB & $aTool[4] & @CRLF & _
                "Right X:" & @TAB & $aTool[5] & @CRLF & _
                "Right Y:" & @TAB & $aTool[6] & @CRLF & _
                "Instance:" & @TAB & $aTool[7] & @CRLF & _
                "Text:" & @TAB & $aTool[8] & @CRLF & _
                "lParam:" & @TAB & $aTool[9])
    EndIf

    _MemoMsgBoxStatus("", Default, $hWin) ; больше действий нет, ждём закрытия GUI, закрываем и внешний GUI

EndFunc   ;==>Example
