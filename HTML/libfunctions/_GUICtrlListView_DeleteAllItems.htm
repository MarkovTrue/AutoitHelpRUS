Справочник функций

_GUICtrlListView_DeleteAllItems

Удаляет все элементы из элемента управления ListView.

#include <GuiListView.au3>
_GUICtrlListView_DeleteAllItems ( $hWnd )

Параметры

$hWnd Идентификатор или дескриптор элемента управления.

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False.

Примечания

Удаление всех элементов из нативного ListView может занять некоторое время – рекомендуется заключить эту команду в пару _GUICtrlListView_Begin/EndUpdate в этом случае.

См. также

_GUICtrlListView_DeleteItem, _GUICtrlListView_DeleteItemsSelected

Пример

Пример 1

#include <GUIConstantsEx.au3>
#include <GuiListView.au3>
#include <MsgBoxConstants.au3>

Example()

Func Example()
    GUICreate("ListView Item Deletion", 400, 500)
    Local $idListview = GUICtrlCreateListView("Col 1 |Col 2 |Col 3 ", 10, 10, 380, 480, $LVS_SHOWSELALWAYS)

    ; Отобразить GUI.
    GUISetState(@SW_SHOW)

    For $i = 0 To 9
        GUICtrlCreateListViewItem("Native Item " & $i & "|Item " & $i & "-1|Item " & $i & "-2", $idListview)
    Next
    For $i = 10 To 20
        _GUICtrlListView_AddItem($idListview, "UDF Item " & $i, -1, 1000 + $i)
        _GUICtrlListView_AddSubItem($idListview, $i, "Item " & $i & "-1", 1)
        _GUICtrlListView_AddSubItem($idListview, $i, "Item " & $i & "-2", 2)
    Next

    MsgBox($MB_SYSTEMMODAL, "Delete all", "Deleting both native and UDF Items")

    ; Передайте controlID нативного ListView
    _GUICtrlListView_DeleteAllItems($idListview)

    ; Цикл до выхода пользователя
    While 1
        Switch GUIGetMsg()
            Case $GUI_EVENT_CLOSE
                ExitLoop

        EndSwitch
    WEnd

    ; Удалить предыдущий GUI и все элементы управления.
    GUIDelete()
EndFunc   ;==>Example

Пример 2

#include <GUIConstantsEx.au3>
#include <GuiListView.au3>
#include <MsgBoxConstants.au3>

Example() ; Элемент ListView, созданный через ListView UDF.

Func Example()
    Local $aItems[10][3]

    Local $hGUI = GUICreate("(UDF Created) ListView Delete All Items (v" & @AutoItVersion & ")", 400, 300)
    Local $hListView = _GUICtrlListView_Create($hGUI, "col1|col2|col3", 10, 10, 380, 280)

    ; Отобразить GUI.
    GUISetState(@SW_SHOW)

    ; Установить формат ANSI
;~ _GUICtrlListView_SetUnicodeFormat($hListView, False)

    ; загрузка 3 столбцов
    For $i = 0 To UBound($aItems) - 1
        $aItems[$i][0] = "Item " & $i
        $aItems[$i][1] = "Item " & $i & "-1"
        $aItems[$i][2] = "Item " & $i & "-2"
    Next

    _GUICtrlListView_AddArray($hListView, $aItems)

    MsgBox($MB_SYSTEMMODAL, "Delete all", "Delete all items")

    ; Передать дескриптор UDF-ListView
    _GUICtrlListView_DeleteAllItems($hListView)

    ; Цикл до выхода пользователя
    While 1
        Switch GUIGetMsg()
            Case $GUI_EVENT_CLOSE
                ExitLoop

        EndSwitch
    WEnd

    ; Удалить предыдущий GUI и все элементы управления.
    GUIDelete()
EndFunc   ;==>Example
