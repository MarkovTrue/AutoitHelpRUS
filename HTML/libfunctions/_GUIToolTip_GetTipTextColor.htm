_GUIToolTip_GetTipTextColor

Получает цвет текста элемента управления ToolTip.

#include <GuiToolTip.au3>
_GUIToolTip_GetTipTextColor ( $hTool )

Параметры

$hTool Дескриптор элемента управления ToolTip (возвращается функцией _GUIToolTip_Create).

Возвращаемое значение

Возвращает значение COLORREF (BGR) цвета текста элемента управления ToolTip.

См. также

_GUIToolTip_GetTipBkColor, _GUIToolTip_SetTipBkColor, _GUIToolTip_SetTipTextColor

Пример

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <GUIToolTip.au3>
#include <MsgBoxConstants.au3>
#include <WinAPITheme.au3>

Example()

Func Example()
    Local $hGUI = GUICreate("ToolTip Get/Set Tip TextColor (v" & @AutoItVersion & ")", 450, 300, 100, 100)

    Local $idBtn_Add = GUICtrlCreateButton("Button", 30, 32, 130, 28)
    Local $hAdd = GUICtrlGetHandle($idBtn_Add)

    ; Создать элемент tooltip
    Local $hToolTip = _GUIToolTip_Create(0, $TTS_BALLOON)
    _WinAPI_SetWindowTheme($hToolTip, "", "")

    ; Изменить настройки цвета подсказки
    ; Значение цвета, используемое в функции _GUIToolTip_SetTipBkColor – это значение COLORREF (BGR)
    _GUIToolTip_SetTipBkColor($hToolTip, 0x395A00)
    ; Значение цвета, используемое в функции _GUIToolTip_SetTipTextColor – это значение COLORREF (BGR)
    _GUIToolTip_SetTipTextColor($hToolTip, 0x1EBFFF)

;~ $hGUI = 0 ; is OK
    ; Добавить tool в элемент tooltip
    _GUIToolTip_AddTool($hToolTip, $hGUI, "This is the ToolTip text", $hAdd)

    GUISetState(@SW_SHOW)

    ; Показать tooltip, связанный с кнопкой
    Opt("MouseCoordMode", 2)
    MouseMove(50, 42, 0)
    Sleep(250)

    ; Получить цвет текста подсказки.
    _MemoMsgBox($MB_SYSTEMMODAL, 'Message', 'Text color : 0x' & Hex(_GUIToolTip_GetTipTextColor($hToolTip), 6))

    While 1
        If GUIGetMsg() = $GUI_EVENT_CLOSE Then ExitLoop
    WEnd

    ; Уничтожить элемент tooltip
    _GUIToolTip_Destroy($hToolTip)
    GUIDelete($hGUI)
EndFunc   ;==>Example
