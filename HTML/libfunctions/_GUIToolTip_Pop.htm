Справочник функций

_GUIToolTip_Pop

Удаляет отображаемую подсказку из поля зрения.

#include <GuiToolTip.au3>
_GUIToolTip_Pop ( $hTool )

Параметры

$hTool Дескриптор элемента управления ToolTip (возвращается функцией _GUIToolTip_Create).

Возвращаемое значение

Отсутствует.

См. также

_GUIToolTip_PopUp

Пример

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <GUIToolTip.au3>

Example()

Func Example()
    Local $hGUI = GUICreate("ToolTip Pop - v(" & @AutoItVersion & ")", 450, 300, 100, 100)

    Local $idButton = GUICtrlCreateButton(" Button ", 30, 32, 130, 28)
    ; Local $hButton = GUICtrlGetHandle($idButton)

    Local $hToolTip = _GUIToolTip_Create(0, $TTS_BALLOON)
    Local $aPos = MouseGetPos()

;~ $hGUI = 0 ; is OK
    _GUIToolTip_AddTool($hToolTip, $hGUI, " ", 0, 0, 0, 0, 0, 0)

    _GUIToolTip_SetTitle($hToolTip, 'Mouse Coordinates', $TTI_INFO)

    GUISetState(@SW_SHOW)

    ; Показать tooltip, связанный с кнопкой
    Opt("MouseCoordMode", 2)
    MouseMove(50, 42, 0)
    Sleep(250)

    ; Получить позицию мыши как стартовую точку для отслеживания
    _GUIToolTip_TrackPosition($hToolTip, $aPos[0], $aPos[1])
    _GUIToolTip_TrackActivate($hToolTip, True, $hGUI, 0)

    Local $hTimer = TimerInit()
    While 1
        Sleep(50)
        ; Прочитать позицию мыши
        $aPos = MouseGetPos()
        ; Использовать позицию мыши для установки положения подсказки
        _GUIToolTip_TrackPosition($hToolTip, $aPos[0], $aPos[1])
        _GUIToolTip_UpdateTipText($hToolTip, $hGUI, 0, "X: " & $aPos[0] & " Y: " & $aPos[1])

        Switch GUIGetMsg()
            Case $GUI_EVENT_CLOSE
                ExitLoop
            Case $idButton
                ; ...
        EndSwitch
        ; Через 3 секунды после запуска это удалит инструмент из отображения
        If TimerDiff($hTimer) >= 3000 Then
            _GUIToolTip_Pop($hToolTip)
        EndIf
    WEnd

    ; Уничтожить элемент tooltip
    _GUIToolTip_Destroy($hToolTip)
    GUIDelete($hGUI)
EndFunc   ;==>Example
