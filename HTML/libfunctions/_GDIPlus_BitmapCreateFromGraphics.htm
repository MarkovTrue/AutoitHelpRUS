_GDIPlus_BitmapCreateFromGraphics

Создаёт объект Bitmap на основе объекта Graphics, ширины и высоты.

#include <GDIPlus.au3>
_GDIPlus_BitmapCreateFromGraphics ( $iWidth, $iHeight, $hGraphics )

Параметры

$iWidth Ширина изображения в пикселях.
$iHeight Высота изображения в пикселях.
$hGraphics Дескриптор объекта Graphics.

Возвращаемое значение

Успех: Возвращает дескриптор объекта Bitmap.
Ошибка: Возвращает 0 и устанавливает @error ≠ 0.
@extended: Может содержать код ошибки GPSTATUS ($GDIP_ERR* см. GPIPlusConstants.au3).

Примечания

После завершения работы с объектом Bitmap вызовите _GDIPlus_BitmapDispose() для освобождения ресурсов.

См. также

_GDIPlus_BitmapDispose

Дополнительно

Искать GdipCreateBitmapFromGraphics в библиотеке MSDN

Пример

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>

Global $g_hGUI, $g_hBrush, $g_hGfx, $g_hGfxCtxt, $g_hBitmap

Example()

Func Example()
    AutoItSetOption("GUIOnEventMode", 1)

    _GDIPlus_Startup() ;инициализировать GDI+
    Local Const $iWidth = 600, $iHeight = 600, $iBgColor = 0x303030 ;$iBGColor в формате RRGGBB

    $g_hGUI = GUICreate("GDI+ Example (" & @ScriptName & ")", $iWidth, $iHeight) ;создать тестовый GUI
    GUISetBkColor($iBgColor, $g_hGUI) ;установить цвет фона GUI
    GUISetState(@SW_SHOW)

    ;создать буферизованный набор кадров для плавного перемещения графики
    $g_hGfx = _GDIPlus_GraphicsCreateFromHWND($g_hGUI) ;создать графический объект из дескриптора окна
    $g_hBitmap = _GDIPlus_BitmapCreateFromGraphics($iWidth, $iHeight, $g_hGfx) ; Создать объект Bitmap на основе объекта Graphics.
    $g_hGfxCtxt = _GDIPlus_ImageGetGraphicsContext($g_hBitmap) ; Получить графический контекст изображения/Bitmap для рисования.
    _GDIPlus_GraphicsSetSmoothingMode($g_hGfxCtxt, $GDIP_SMOOTHINGMODE_HIGHQUALITY) ;задаёт качество отрисовки графики (сглаживание)

    $g_hBrush = _GDIPlus_BrushCreateSolid(0xFF8080FF) ; Создать объект сплошной кисти.

    GUISetOnEvent($GUI_EVENT_CLOSE, "_Exit")

    Local Const $iDeg = ACos(-1) / 180 ;ACos(-1) приблизительно равен Pi
    Local $iSize = 50, $iX_Center = ($iWidth - $iSize) / 2, $iY_Center = ($iHeight - $iSize) / 2, $iXPos, $iYPos, $iAngle = 0
    Local Const $iDots = 16, $iAngelDist = 360 / $iDots, $iRadius = 200

    Do
        _GDIPlus_GraphicsClear($g_hGfxCtxt, 0xFF000000 + $iBgColor) ; Очистить изображение заданным цветом (формат AARRGGBB).
        For $i = 1 To $iDots
            $iXPos = $iX_Center + Cos($iAngle * $iDeg) * $iRadius
            $iYPos = $iY_Center + Sin($iAngle * $iDeg) * $iRadius
            _GDIPlus_GraphicsFillEllipse($g_hGfxCtxt, $iXPos, $iYPos, $iSize, $iSize, $g_hBrush) ; Нарисовать точки по окружности.
            $iAngle += $iAngelDist ; Увеличить угол до следующей точки.
        Next
        $iAngle += 1 ; Увеличить общий угол.
        _GDIPlus_GraphicsDrawImageRect($g_hGfx, $g_hBitmap, 0, 0, $iWidth, $iHeight) ; Скопировать нарисованное изображение в GUI.
    Until Not Sleep(20) ;Sleep() always returns 1 and Not 1 is 0
EndFunc   ;==>Example

Func _Exit() ;освободить ресурсы GDI+
    _GDIPlus_BrushDispose($g_hBrush)
    _GDIPlus_GraphicsDispose($g_hGfxCtxt)
    _GDIPlus_GraphicsDispose($g_hGfx)
    _GDIPlus_BitmapDispose($g_hBitmap)
    _GDIPlus_Shutdown()
    GUIDelete($g_hGUI)
    Exit
EndFunc   ;==>_Exit
