Справочник функций

_GUICtrlMonthCal_GetMonthDelta

Возвращает шаг прокрутки месячного календаря.

#include <GuiMonthCal.au3>
_GUICtrlMonthCal_GetMonthDelta ( $hWnd )

Параметры

$hWnd Идентификатор или дескриптор элемента управления.

Возвращаемое значение

Возвращает целое значение:
    текущий шаг прокрутки календаря, если он был установлен через _GUICtrlMonthCal_SetMonthDelta();
    текущее количество видимых месяцев, если шаг ранее не задавался или был сброшен в значение по умолчанию.

См. также

_GUICtrlMonthCal_SetMonthDelta

Пример

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <GuiMonthCal.au3>
#include <WindowsStylesConstants.au3>

Example()

Func Example()
    ; Создать GUI
    GUICreate("Month Calendar Get/Set Month Delta (v" & @AutoItVersion & ")", 420, 300)
    Local $idMonthCal = GUICtrlCreateMonthCal("", 4, 4, -1, -1, BitOR($WS_BORDER, $MCS_MULTISELECT), 0x00000000)

    ; Создать элемент memo
    _MemoCreate(4, 168, 412, 128, 0)
    GUISetState(@SW_SHOW)

    ; Получить/установить шаг месяцев
    _GUICtrlMonthCal_SetMonthDelta($idMonthCal, 2)
    _MemoWrite("Month delta: " & _GUICtrlMonthCal_GetMonthDelta($idMonthCal))

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
    GUIDelete()
EndFunc   ;==>Example
