_GDIPlus_PathAddCurve3

Добавляет кардинальный сплайн к текущей фигуре пути.

#include <GDIPlus.au3>
_GDIPlus_PathAddCurve3 ( $hPath, $aPoints, $iOffset, $iNumOfSegments [, $nTension = 0.5] )

Параметры

$hPath Указатель на объект GraphicsPath.
$aPoints Массив точек, задающих кардинальный сплайн:
[0][0] – количество точек
[1][0] – координата X точки 1
[1][1] – координата Y точки 1
[2][0] – координата X точки 2
[2][1] – координата Y точки 2
[n][0] – координата X точки n
[n][1] – координата Y точки n
$iOffset Индекс элемента массива, используемого как первая точка кардинального сплайна; равен индексу конкретной точки в массиве минус 1.
$iNumOfSegments Количество сегментов кардинального сплайна. Сегменты – это кривые, соединяющие последовательные точки массива.
$nTension [необязательный] Неотрицательное вещественное число, определяющее длину кривой и характер её изгиба.
С увеличением значения кривая становится более плавной.

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False и устанавливает @error ≠ 0.
@extended: Может содержать код ошибки GPSTATUS ($GDIP_ERR* см. GPIPlusConstants.au3).

Примечания

Кардинальный сплайн проходит через подмножество точек массива, заданное параметрами $iOffset и $iNumOfSegments.

См. также

_GDIPlus_PathAddCurve, _GDIPlus_PathAddCurve2

Дополнительно

Искать GdipAddPathCurve3 в библиотеке MSDN

Пример

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>

Global $g_hGUI, $g_hGraphics, $g_hBmp_Buffer, $g_hGfx_Buffer, $g_hPen, $g_hPen2, $g_hPath, $g_hPath2

Example()

Func Example()
    AutoItSetOption("GUIOnEventMode", 1)

    Local Const $iW = 400, $iH = 400

    $g_hGUI = GUICreate("GDI+", $iW, $iH)
    GUISetOnEvent($GUI_EVENT_CLOSE, "_Exit")
    GUISetState(@SW_SHOW)

    _GDIPlus_Startup()
    $g_hGraphics = _GDIPlus_GraphicsCreateFromHWND($g_hGUI) ;Создать графический объект из дескриптора окна
    $g_hBmp_Buffer = _GDIPlus_BitmapCreateFromGraphics($iW, $iH, $g_hGraphics)
    $g_hGfx_Buffer = _GDIPlus_ImageGetGraphicsContext($g_hBmp_Buffer)
    _GDIPlus_GraphicsSetSmoothingMode($g_hGfx_Buffer, $GDIP_SMOOTHINGMODE_HIGHQUALITY) ;Задаёт качество отрисовки графики (сглаживание)

    $g_hPen = _GDIPlus_PenCreate(0xFF8800AA, 2)
    $g_hPen2 = _GDIPlus_PenCreate(0xFFFFDD00, 3)

    $g_hPath = _GDIPlus_PathCreate() ;Создать новый объект path

    Local $aPoints[201][2] = [[200]], $iRadius = 0
    For $i = 1 To $aPoints[0][0]
        $aPoints[$i][0] = Cos($i / 3.14) * $iRadius + $iW / 2
        $aPoints[$i][1] = Sin($i / 3.14) * $iRadius + $iH / 2
        $iRadius += 1
    Next

    _GDIPlus_PathAddCurve($g_hPath, $aPoints)

    $g_hPath2 = _GDIPlus_PathCreate() ;Создать новый объект path
    Local $iOff = 1, $iNum = 4
    While Sleep(20)
        _GDIPlus_PathReset($g_hPath2)
        _GDIPlus_PathAddCurve3($g_hPath2, $aPoints, $iOff, $iNum)

        $iOff += 1
        If $iOff > $aPoints[0][0] - $iNum Then $iOff = 1

        _GDIPlus_GraphicsClear($g_hGfx_Buffer, 0xFF000000)
        _GDIPlus_GraphicsDrawPath($g_hGfx_Buffer, $g_hPath, $g_hPen) ;Нарисовать path в дескрипторе графики (GUI)
        _GDIPlus_GraphicsDrawPath($g_hGfx_Buffer, $g_hPath2, $g_hPen2) ;Нарисовать path в дескрипторе графики (GUI)
        _GDIPlus_GraphicsDrawImage($g_hGraphics, $g_hBmp_Buffer, 0, 0)
    WEnd
EndFunc   ;==>Example

Func _Exit()
    ; Освободить ресурсы
    _GDIPlus_PathDispose($g_hPath)
    _GDIPlus_PathDispose($g_hPath2)
    _GDIPlus_PenDispose($g_hPen)
    _GDIPlus_PenDispose($g_hPen2)
    _GDIPlus_GraphicsDispose($g_hGfx_Buffer)
    _GDIPlus_BitmapDispose($g_hBmp_Buffer)
    _GDIPlus_GraphicsDispose($g_hGraphics)
    _GDIPlus_Shutdown()
    GUIDelete($g_hGUI)
    Exit
EndFunc   ;==>_Exit
