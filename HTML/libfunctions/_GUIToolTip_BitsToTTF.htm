Справочник функций

_GUIToolTip_BitsToTTF

Декодирует битовые флаги в строки TTF_*.

#include <GuiToolTip.au3>
_GUIToolTip_BitsToTTF ( $iFlags )

Параметры

$iFlags Значение представляющее константы ToolTip которые нужно декодировать.

Возвращаемое значение

Возвращает строку с разделением запятыми, показывающую какие константы ToolTip составляют значение $iFlags.

Пример

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <GUIToolTip.au3>
#include <MsgBoxConstants.au3>

Example()

Func Example()
    Local $hGUI = GUICreate("ToolTip BitsToTTF - v(" & @AutoItVersion & ")", 450, 300, 100, 100)

    Local $idButton = GUICtrlCreateButton("This is a button", 30, 32, 130, 28)
    Local $hButton = GUICtrlGetHandle($idButton)
    ; создать элемент tooltip с настройками по умолчанию
    Local $hToolTip = _GUIToolTip_Create(0)

;~ $hGUI = 0 ; no text
    ; добавить tool в элемент tooltip
    _GUIToolTip_AddTool($hToolTip, $hGUI, "This is a ToolTip", $hButton)

    GUISetState(@SW_SHOW)

    ; Получить информацию для инструмента, назначенного кнопке
    Local $aTool = _GUIToolTip_GetToolInfo($hToolTip, $hGUI, $hButton)
    ; Конвертировать массив в текст используя функцию _GUIToolTip_BitsToTTF для конвертирования флага
    ; информации в строку текста.
    _MemoMsgBox($MB_SYSTEMMODAL, "Tooltip info", "Flags: " & @TAB & _GUIToolTip_BitsToTTF($aTool[0]) & @CRLF & _
            "HWnd: " & @TAB & $aTool[1] & @CRLF & _
            "ID: " & @TAB & $aTool[2] & @CRLF & _
            "Left X:" & @TAB & $aTool[3] & @CRLF & _
            "Left Y:" & @TAB & $aTool[4] & @CRLF & _
            "Right X:" & @TAB & $aTool[5] & @CRLF & _
            "Right Y:" & @TAB & $aTool[6] & @CRLF & _
            "Instance:" & @TAB & $aTool[7] & @CRLF & _
            "Text:" & @TAB & $aTool[8] & @CRLF & _
            "lParam:" & @TAB & $aTool[9])

    While 1
        If GUIGetMsg() = $GUI_EVENT_CLOSE Then ExitLoop
    WEnd

    ; Уничтожить элемент tooltip
    _GUIToolTip_Destroy($hToolTip)
    GUIDelete($hGUI)
EndFunc   ;==>Example
