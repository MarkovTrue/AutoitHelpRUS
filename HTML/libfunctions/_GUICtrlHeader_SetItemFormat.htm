Справочник функций

_GUICtrlHeader_SetItemFormat

Устанавливает формат элемента.

#include <GuiHeader.au3>
_GUICtrlHeader_SetItemFormat ( $hWnd, $iIndex, $iFormat )

Параметры

$hWnd Дескриптор элемента управления.
$iIndex Индекс элемента (нумерация с 0).
$iFormat Комбинация следующих идентификаторов формата:
$HDF_CENTER – содержимое элемента выравнено по центру
$HDF_LEFT – содержимое элемента выравнено по левому краю
$HDF_RIGHT – содержимое элемента выравнено по правому краю
$HDF_BITMAP – элемент отображает растровое изображение
$HDF_BITMAP_ON_RIGHT – растровое изображение отображается справа от текста
$HDF_OWNERDRAW – владелец элемента управления Header рисует элемент
$HDF_STRING – элемент отображает строку
$HDF_IMAGE – отображает изображение из списка изображений
$HDF_RTLREADING – текст будет читаться в противоположном направлении от текста в родительском окне
$HDF_SORTDOWN – рисует стрелку вниз на этом элементе
$HDF_SORTUP – рисует стрелку вверх на этом элементе.

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False.

См. также

_GUICtrlHeader_GetItemFormat

Пример

Пример 1

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <GuiHeader.au3>
#include <GuiImageList.au3>
#include <WinAPIGdi.au3>

Example()

Func Example()
    ; Создать GUI
    Local $hGUI = GUICreate("Header Get/Set ItemFormat (v" & @AutoItVersion & ")", 450, 300, 100, 100)
    Local $hHeader = _GUICtrlHeader_Create($hGUI)
    _MemoCreate(2, 52, 444, 220)

;~ _GUICtrlHeader_SetUnicodeFormat($hHeader, True)

    GUISetState(@SW_SHOW)

    ; Создать image list с изображениями
    Local $hImage = _GUIImageList_Create(11, 11)
    _GUIImageList_Add($hImage, _WinAPI_CreateSolidBitmap($hGUI, 0xFF0000, 11, 11))
    _GUIImageList_Add($hImage, _WinAPI_CreateSolidBitmap($hGUI, 0x00FF00, 11, 11))
    _GUIImageList_Add($hImage, _WinAPI_CreateSolidBitmap($hGUI, 0x0000FF, 11, 11))
    _GUICtrlHeader_SetImageList($hHeader, $hImage)

    ; Добавить столбцы
    _GUICtrlHeader_AddItem($hHeader, "Column 0", 100, 0, 0)
    _GUICtrlHeader_AddItem($hHeader, "Column 1", 100, 0, 1)
    _GUICtrlHeader_AddItem($hHeader, "Column 2", 100, 0, 2)
    _GUICtrlHeader_AddItem($hHeader, "Column 3", 100)

    ; Задать формат столбца 0
    _GUICtrlHeader_SetItemFormat($hHeader, 0, BitOR($HDF_CENTER, $HDF_STRING))

    ; Показать формат столбца 0
    _MemoWrite("Column 0 format: " & "0x" & Hex(_GUICtrlHeader_GetItemFormat($hHeader, 0)))

    Sleep(1000)

    ; Задать формат столбца 0
    _GUICtrlHeader_SetItemFormat($hHeader, 0, BitOR($HDF_LEFT, $HDF_STRING, $HDF_IMAGE))

    ; Показать формат столбца 0
    _MemoWrite("Column 0 format: " & "0x" & Hex(_GUICtrlHeader_GetItemFormat($hHeader, 0)))

    _MemoMsgBoxStatus("", -1, $hGUI) ; больше действий нет, ждём закрытия GUI

EndFunc   ;==>Example

; == Пример 2 : Header Set ItemFormat во внешнем процессе

#include "Extras\HelpFileInternals.au3"

#include <GuiHeader.au3>

Example()

Func Example()
    Local $sFromTo
    Local $hWin = _MemoRunAU3OutProcess($sFromTo, False) ; OK при запуске в другом режиме

    Local $hHeader = _MemoCreateOutProcess($hWin, "SysHeader32", 0, $sFromTo)

    ; Задать формат столбца 0
    _GUICtrlHeader_SetItemFormat($hHeader, 0, BitOR($HDF_CENTER, $HDF_STRING))

    ; Показать формат столбца 0
    _MemoWrite("Column 0 format: " & "0x" & Hex(_GUICtrlHeader_GetItemFormat($hHeader, 0)))

    _MemoMsgBoxStatus("", -1, $hWin) ; больше действий нет, ждём закрытия GUI

EndFunc   ;==>Example
