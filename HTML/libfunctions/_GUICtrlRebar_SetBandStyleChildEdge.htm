Справочник функций

_GUICtrlRebar_SetBandStyleChildEdge

Устанавливает, имеет ли полоса грани сверху и снизу дочернего окна.

#include <GuiReBar.au3>
_GUICtrlRebar_SetBandStyleChildEdge ( $hWnd, $iIndex [, $bEnabled = True] )

Параметры

$hWnd Дескриптор элемента управления Rebar.
$iIndex Индекс полосы (нумерация с 0).
$bEnabled [необязательный] Если True – состояние элемента устанавливается, иначе – нет

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False.

См. также

_GUICtrlRebar_GetBandStyleChildEdge, _GUICtrlRebar_SetBandStyle

Пример

#include "Extras\HelpFileInternals.au3"

#include <AutoItConstants.au3>
#include <GUIConstantsEx.au3>
#include <GuiReBar.au3>
#include <GuiToolbar.au3>
#include <MsgBoxConstants.au3>
#include <WinAPIConstants.au3>
#include <WindowsStylesConstants.au3>

Example()

Func Example()
    Local $hGui, $idBtn_Exit, $hReBar, $hToolbar, $idInput
    Local Enum $e_idNew = 1000, $e_idOpen, $e_idSave, $e_idHelp

    $hGui = GUICreate("Rebar", 400, 396, -1, -1, BitOR($WS_MINIMIZEBOX, $WS_CAPTION, $WS_POPUP, $WS_SYSMENU, $WS_MAXIMIZEBOX))

    ; создать элемент rebar
    $hReBar = _GUICtrlRebar_Create($hGui, BitOR($CCS_TOP, $WS_BORDER, $RBS_VARHEIGHT, $RBS_AUTOSIZE, $RBS_BANDBORDERS))

    _MemoCreate(2, 100, 396, 250, $WS_VSCROLL)

    ; создать toolbar для размещения в rebar
    $hToolbar = _GUICtrlToolbar_Create($hGui, BitOR($TBSTYLE_FLAT, $CCS_NORESIZE, $CCS_NOPARENTALIGN))

    ; Добавить стандартные системные изображения
    Switch _GUICtrlToolbar_GetBitmapFlags($hToolbar)
        Case 0
            _GUICtrlToolbar_AddBitmap($hToolbar, 1, -1, $IDB_STD_SMALL_COLOR)
        Case 2
            _GUICtrlToolbar_AddBitmap($hToolbar, 1, -1, $IDB_STD_LARGE_COLOR)
    EndSwitch

    ; Добавить кнопки
    _GUICtrlToolbar_AddButton($hToolbar, $e_idNew, $STD_FILENEW)
    _GUICtrlToolbar_AddButton($hToolbar, $e_idOpen, $STD_FILEOPEN)
    _GUICtrlToolbar_AddButton($hToolbar, $e_idSave, $STD_FILESAVE)
    _GUICtrlToolbar_AddButtonSep($hToolbar)
    _GUICtrlToolbar_AddButton($hToolbar, $e_idHelp, $STD_HELP)

    ; создать поле ввода для размещения в rebar
    $idInput = GUICtrlCreateInput("Input control", 0, 0, 120, 20)

    ;добавить полосу с элементом управления
    _GUICtrlRebar_AddBand($hReBar, GUICtrlGetHandle($idInput), 120, 200, "Name:")

    ; добавить полосу с элементом управления в начало rebar
    _GUICtrlRebar_AddToolBarBand($hReBar, $hToolbar, "", 0)

    $idBtn_Exit = GUICtrlCreateButton("Exit", 150, 360, 100, 25)
    GUICtrlSetState($idBtn_Exit, $GUI_DEFBUTTON)
    GUICtrlSetState($idBtn_Exit, $GUI_FOCUS)

    GUISetState(@SW_SHOW)

    _GUICtrlRebar_SetBandStyleBreak($hReBar, 1)

    MsgBox($MB_SYSTEMMODAL, "Information", "Setting Child Edge Style")

    For $x = 0 To _GUICtrlRebar_GetBandCount($hReBar) - 1
        _GUICtrlRebar_SetBandStyleChildEdge($hReBar, $x, False)
        _MemoWrite("Band Index " & $x & @TAB & "$RBBS_CHILDEDGE.....: " & _GUICtrlRebar_GetBandStyleChildEdge($hReBar, $x))
        _MemoWrite("============================================")
    Next

    MsgBox($MB_SYSTEMMODAL, "Information", "Setting Child Edge Style")

    For $x = 0 To _GUICtrlRebar_GetBandCount($hReBar) - 1
        _GUICtrlRebar_SetBandStyleChildEdge($hReBar, $x)
        _MemoWrite("Band Index " & $x & @TAB & "$RBBS_CHILDEDGE.....: " & _GUICtrlRebar_GetBandStyleChildEdge($hReBar, $x))
        _MemoWrite("============================================")
    Next

    While 1
        Switch GUIGetMsg()
            Case $GUI_EVENT_CLOSE, $idBtn_Exit
                Exit
        EndSwitch
    WEnd
EndFunc   ;==>Example
