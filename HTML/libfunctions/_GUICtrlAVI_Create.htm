Справочник функций

_GUICtrlAVI_Create

Создаёт элемент управления AVI.

#include <GuiAVI.au3>
_GUICtrlAVI_Create ( $hWnd [, $sFilePath = "" [, $iSubFileID = -1 [, $iX = 0 [, $iY = 0 [, $iWidth = 0 [, $iHeight = 0 [, $iStyle = 0x00000006 [, $iExStyle = 0x00000000]]]]]]]] )

Параметры

$hWnd Дескриптор родительского или владеющего окна.
$sFilePath [необязательный] Имя файла видео. Поддерживаются только файлы .avi
$iSubFileID [необязательный] Идентификатор подфайла для использования
$iX Горизонтальная позиция элемента управления.
$iY Вертикальная позиция элемента управления.
$iWidth [необязательный] Ширина элемента управления
$iHeight [необязательный] Высота элемента управления
$iStyle [необязательный] Стили элемента управления:
$ACS_CENTER – центрирует анимацию в окне элемента управления
$ACS_TRANSPARENT – создаёт элемент управления с прозрачным фоном
$ACS_AUTOPLAY – начинает воспроизводить анимацию сразу при открытии AVI-клипа
$ACS_TIMER – элемент управления воспроизводит клип без создания потока
По умолчанию: $ACS_TRANSPARENT, $ACS_AUTOPLAY
Обязательные: $WS_CHILD, $WS_VISIBLE.
$iExStyle [необязательный] Расширенный стиль элемента. Соответствует стандартным константам $WS_EX_*. См. таблицу расширенных стилей.

Возвращаемое значение

Успех: Возвращает дескриптор элемента управления анимацией.
Ошибка: Возвращает 0.

Примечания

Данная функция предназначена для опытных пользователей и для изучения принципов работы элемента управления.

См. также

_GUICtrlAVI_Destroy

Пример

Пример 1

; == Пример 1: создано с помощью UDF

#include "Extras\WM_NOTIFY.au3"

#include <GuiAVI.au3>
#include <GUIConstantsEx.au3>
#include <WindowsNotifsConstants.au3>

Global $g_hAVI

Example()

Func Example()
    Local $sWow64 = ""
    If @AutoItX64 Then $sWow64 = "\Wow6432Node"
    Local $hGUI, $sFile = RegRead("HKEY_LOCAL_MACHINE\SOFTWARE" & $sWow64 & "\AutoIt v3\AutoIt", "InstallDir") & "\Examples\GUI\SampleAVI.avi"

    ; Создать GUI
    $hGUI = GUICreate("AVI Create (v" & @AutoItVersion & ")", 300, 100)
    $g_hAVI = _GUICtrlAVI_Create($hGUI, $sFile, -1, 10, 10)
    GUISetState(@SW_SHOW)

    GUIRegisterMsg($WM_COMMAND, "WM_COMMAND")

    ; Воспроизвести пример AVI AutoIt
    _GUICtrlAVI_Play($g_hAVI)

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE

    ; Закрыть AVI-клип
    _GUICtrlAVI_Close($g_hAVI)

    GUIDelete()
EndFunc   ;==>Example

Func WM_COMMAND($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg
    Local $hWndFrom = $lParam
    Local $iIDFrom = BitAND($wParam, 0xFFFF) ; Младшее слово
    Local $iCode = BitShift($wParam, 16) ; Старшее слово
    Switch $hWndFrom
        Case $g_hAVI
            Switch $iCode
                Case $ACN_START ; Уведомляет родительское окно анимационного элемента о начале воспроизведения связанного AVI-ролика
                    _WM_NOTIFY_DebugInfo("$ACN_START", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $ACN_STOP ; Уведомляет родительское окно анимационного элемента об остановке воспроизведения связанного AVI-ролика
                    _WM_NOTIFY_DebugInfo("$ACN_STOP", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
            EndSwitch
    EndSwitch
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_COMMAND
