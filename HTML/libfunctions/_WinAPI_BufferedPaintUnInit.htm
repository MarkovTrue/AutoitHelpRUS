Справочник функций

_WinAPI_BufferedPaintUnInit

Закрывает буферизированную раскраску для текущего потока.

#include <WinAPITheme.au3>
_WinAPI_BufferedPaintUnInit ( )

Возвращаемое значение

Успех: Возвращает 1.
Ошибка: Возвращает 0 и устанавливает @error ≠ 0.
@extended: Может содержать код ошибки HRESULT.

Примечания

Для этой функции требуется Windows Vista или более поздняя версия.

См. также

_WinAPI_BufferedPaintInit

Дополнительно

Искать BufferedPaintUnInit в библиотеке MSDN

Пример

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <MsgBoxConstants.au3>
#include <SendMessage.au3>
#include <StaticConstants.au3>
#include <WinAPIGdi.au3>
#include <WinAPIGdiDC.au3>
#include <WinAPIHObj.au3>
#include <WinAPIIcons.au3>
#include <WinAPIMisc.au3>
#include <WinAPIRes.au3>
#include <WinAPIShellEx.au3>
#include <WinAPISys.au3>
#include <WinAPITheme.au3>
#include <WindowsStylesConstants.au3>

If Number(_WinAPI_GetVersion()) < 6.0 Then
    MsgBox(($MB_ICONERROR + $MB_SYSTEMMODAL), 'Error', 'Require Windows Vista or later.')
    Exit
EndIf

; Загрузить bitmap и иконку
Local $sBmp = _Extras_PathFull('Tech.bmp')
Local $hTech = _WinAPI_LoadImage(0, $sBmp, $IMAGE_BITMAP, 0, 0, $LR_LOADFROMFILE)
Local $sIco = _Extras_PathFull('NVIDIA.ico')
Local $hIcon = _WinAPI_ShellExtractIcon($sIco, 0, 256, 256)

; Создать GUI
GUICreate('Test ' & StringReplace(@ScriptName, '.au3', '()'), 302, 302, -1, -1, $WS_POPUP, $WS_EX_TOPMOST)
GUICtrlCreatePic('', 0, 0, 302, 302)
Local $hPic = GUICtrlGetHandle(-1)

; Инициализировать буферизованную отрисовку
_WinAPI_BufferedPaintInit()

; Выбор растрового изображения в контекст устройства
Local $hDC = _WinAPI_GetDC($hPic)
Local $hSrcDC = _WinAPI_CreateCompatibleDC($hDC)
Local $hSrcSv = _WinAPI_SelectObject($hSrcDC, $hTech)

; Начать операцию буферизованной отрисовки
Local $hNewDC
Local $hBP = _WinAPI_BeginBufferedPaint($hSrcDC, _WinAPI_CreateRectEx(22, 22, 256, 256), $hNewDC, $BPBF_COMPATIBLEBITMAP, 0, 0, 160)

; Нарисовать иконку с альфа-каналом
_WinAPI_DrawIconEx($hNewDC, 22, 22, $hIcon)

; Завершить буферизованную отрисовку и обновить опорный DC
_WinAPI_EndBufferedPaint($hBP)

; Слить bitmap и освободить ресурсы
Local $hDestDC = _WinAPI_CreateCompatibleDC($hDC)
Local $hBitmap = _WinAPI_CreateCompatibleBitmapEx($hDC, 302, 302, 0x606060)
Local $hDestSv = _WinAPI_SelectObject($hDestDC, $hBitmap)
_WinAPI_BitBlt($hDestDC, 1, 1, 300, 300, $hSrcDC, 0, 0, $MERGECOPY)
_WinAPI_SelectObject($hDestDC, $hDestSv)
_WinAPI_DeleteDC($hDestDC)
_WinAPI_SelectObject($hSrcDC, $hSrcSv)
_WinAPI_DeleteDC($hSrcDC)
_WinAPI_ReleaseDC($hPic, $hDC)
_WinAPI_DeleteObject($hTech)
_WinAPI_DestroyIcon($hIcon)

; Установить bitmap для элемента управления
_SendMessage($hPic, $STM_SETIMAGE, 0, $hBitmap)
Local $hObj = _SendMessage($hPic, $STM_GETIMAGE)
If $hObj <> $hBitmap Then
    _WinAPI_DeleteObject($hBitmap)
EndIf

; Завершить буферизованную отрисовку
_WinAPI_BufferedPaintUnInit()

GUISetState(@SW_SHOW)

Do
Until GUIGetMsg() = $GUI_EVENT_CLOSE
