Справочник функций

_GDIPlus_GraphicsDrawClosedCurve

Рисует замкнутый кардинальный сплайн.

#include <GDIPlus.au3>
_GDIPlus_GraphicsDrawClosedCurve ( $hGraphics, $aPoints [, $hPen = 0] )

Параметры

$hGraphics Дескриптор объекта Graphics.
$aPoints Массив точек кривой:
$aArray[0][0] – количество точек
$aArray[1][0] – координата X точки 1
$aArray[1][1] – координата Y точки 1
$aArray[2][0] – координата X точки 2
$aArray[2][1] – координата Y точки 2
$aArray[n][0] – координата X точки n
$aArray[n][1] – координата Y точки n
$hPen [необязательный] Дескриптор объекта Pen для рисования сплайна. Если 0, используется сплошная чёрная линия шириной 1.

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False и устанавливает @error ≠ 0.
@extended: Может содержать код ошибки GPSTATUS ($GDIP_ERR* см. GPIPlusConstants.au3).

Примечания

В замкнутом кардинальном сплайне кривая продолжается через последнюю точку массива и соединяется с первой точкой.
Массив должен содержать не менее трёх элементов.

Дополнительно

Искать GdipDrawClosedCurve в библиотеке MSDN

Пример

Пример 1

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>

Example()

Func Example()
    Local $hGUI, $hGraphic, $aPoints[8][2]

    ; Создать GUI
    $hGUI = GUICreate("GDI+", 400, 300)
    GUISetState(@SW_SHOW)

    ; Нарисовать кардинальный сплайн
    _GDIPlus_Startup()
    $hGraphic = _GDIPlus_GraphicsCreateFromHWND($hGUI)

    $aPoints[0][0] = 7
    $aPoints[1][0] = 50
    $aPoints[1][1] = 50
    $aPoints[2][0] = 100
    $aPoints[2][1] = 25
    $aPoints[3][0] = 200
    $aPoints[3][1] = 5
    $aPoints[4][0] = 250
    $aPoints[4][1] = 50
    $aPoints[5][0] = 300
    $aPoints[5][1] = 100
    $aPoints[6][0] = 350
    $aPoints[6][1] = 200
    $aPoints[7][0] = 250
    $aPoints[7][1] = 250

    _GDIPlus_GraphicsDrawClosedCurve($hGraphic, $aPoints)

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE

    ; Освободить ресурсы
    _GDIPlus_GraphicsDispose($hGraphic)
    _GDIPlus_Shutdown()
EndFunc   ;==>Example

Пример 2

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>

Example()

Func Example()
    _GDIPlus_Startup() ; инициализировать GDI+
    Local Const $iWidth = 600, $iHeight = 600, $iBgColor = 0x303030 ; $iBGColor в формате RRGGBB

    Local $hGUI = GUICreate("GDI+ Example (" & @ScriptName & ")", $iWidth, $iHeight) ; создать тестовый GUI
    GUISetBkColor($iBgColor, $hGUI) ; установить цвет фона GUI
    GUISetState(@SW_SHOW)

    Local $hGraphics = _GDIPlus_GraphicsCreateFromHWND($hGUI) ; создать графический объект из дескриптора окна
    _GDIPlus_GraphicsSetSmoothingMode($hGraphics, $GDIP_SMOOTHINGMODE_HIGHQUALITY) ; задаёт качество отрисовки графики (сглаживание)
    Local $hPen = _GDIPlus_PenCreate(0xFFFFFFFF, 4) ; цвет в формате AARRGGBB (hex)

    Local $aPoints[8][2]
    $aPoints[0][0] = 7
    $aPoints[1][0] = 50.5
    $aPoints[1][1] = 50.5
    $aPoints[2][0] = 100.75
    $aPoints[2][1] = 25.33
    $aPoints[3][0] = 200.66
    $aPoints[3][1] = 5.5
    $aPoints[4][0] = 250.25
    $aPoints[4][1] = 50.75
    $aPoints[5][0] = 300.125
    $aPoints[5][1] = 100.375
    $aPoints[6][0] = 590.8
    $aPoints[6][1] = 200.222
    $aPoints[7][0] = 250.55
    $aPoints[7][1] = 590.45

    _GDIPlus_GraphicsDrawClosedCurve($hGraphics, $aPoints, $hPen)

    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE

    ; освободить ресурсы GDI+
    _GDIPlus_PenDispose($hPen)
    _GDIPlus_GraphicsDispose($hGraphics)
    _GDIPlus_Shutdown()
    GUIDelete($hGUI)
EndFunc   ;==>Example
