Справочник функций

_GUICtrlListView_HitTest

Определяет.

#include <GuiListView.au3>
_GUICtrlListView_HitTest ( $hWnd [, $iX = -1 [, $iY = -1]] )

Параметры

$hWnd Идентификатор или дескриптор элемента управления.
$iX [необязательный] Координата X в клиентских координатах для проверки, или -1 для использования текущей позиции мыши
$iY [необязательный] Координата Y в клиентских координатах для проверки, или -1 для использования текущей позиции мыши

Возвращаемое значение

Успех: Возвращает массив в следующем формате:
$aArray[0] – индекс элемента в позиции (с нулевым смещением), или -1
$aArray[1] – True, если позиция в клиентской области, но не на элементе
$aArray[2] – True, если позиция над иконкой элемента
$aArray[3] – True, если позиция над текстом элемента
$aArray[4] – True, если позиция над изображением состояния элемента
$aArray[5] – True, если позиция где-то на элементе
$aArray[6] – True, если позиция выше клиентской области
$aArray[7] – True, если позиция ниже клиентской области
$aArray[8] – True, если позиция левее клиентской области
$aArray[9] – True, если позиция правее клиентской области
Ошибка: Устанавливает @error ≠ 0.

Пример

#include "Extras\WM_NOTIFY.au3"

#include <GUIConstantsEx.au3>
#include <GuiImageList.au3>
#include <GuiListView.au3>
#include <GuiStatusBar.au3>
#include <StructureConstants.au3>
#include <WindowsNotifsConstants.au3>

Global $g_idListView, $g_hStatusBar, $g_iIndex = -1

Example()

Func Example()
    ; Создать GUI
    Local $hGUI = GUICreate("ListView Hit Test (v" & @AutoItVersion & ")", 400, 300)
    $g_idListView = GUICtrlCreateListView("", 2, 2, 394, 268)
    $g_idListView = GUICtrlGetHandle($g_idListView) ; получить дескриптор для использования в событиях notify
    $g_hStatusBar = _GUICtrlStatusBar_Create($hGUI, -1, "")

    ; Включить расширенные стили элемента управления
    _GUICtrlListView_SetExtendedListViewStyle($g_idListView, BitOR($LVS_EX_FULLROWSELECT, $LVS_EX_SUBITEMIMAGES))
    GUISetState(@SW_SHOW)

    ; Установить формат ANSI
;~ _GUICtrlListView_SetUnicodeFormat($g_idListView, False)

    _WM_NOTIFY_Register()

    ; Загрузить изображения
    Local $hImage = _GUIImageList_Create()
    _GUIImageList_Add($hImage, _GUICtrlListView_CreateSolidBitMap($g_idListView, 0xFF0000, 16, 16))
    _GUIImageList_Add($hImage, _GUICtrlListView_CreateSolidBitMap($g_idListView, 0x00FF00, 16, 16))
    _GUIImageList_Add($hImage, _GUICtrlListView_CreateSolidBitMap($g_idListView, 0x0000FF, 16, 16))
    _GUICtrlListView_SetImageList($g_idListView, $hImage, 1)

    ; Добавить столбцы
    _GUICtrlListView_AddColumn($g_idListView, "Column 1", 100)
    _GUICtrlListView_AddColumn($g_idListView, "Column 2", 100)
    _GUICtrlListView_AddColumn($g_idListView, "Column 3", 100)

    ; Добавить элементы
    _GUICtrlListView_AddItem($g_idListView, "Row 1: Col 1", 0)
    _GUICtrlListView_AddSubItem($g_idListView, 0, "Row 1: Col 2", 1, 1)
    _GUICtrlListView_AddSubItem($g_idListView, 0, "Row 1: Col 3", 2, 2)
    _GUICtrlListView_AddItem($g_idListView, "Row 2: Col 1", 1)
    _GUICtrlListView_AddSubItem($g_idListView, 1, "Row 2: Col 2", 1, 2)
    _GUICtrlListView_AddItem($g_idListView, "Row 3: Col 1", 2)

    ; Цикл до выхода пользователя.
    While 1
        Switch GUIGetMsg()
            Case $GUI_EVENT_CLOSE
                ExitLoop
        EndSwitch
    WEnd
    GUIDelete()
EndFunc   ;==>Example

Func _ListView_Click()
    Local $aHit

    $aHit = _GUICtrlListView_HitTest($g_idListView)
    If ($aHit[0] <> -1) And ($aHit[0] <> $g_iIndex) Then
        _GUICtrlStatusBar_SetText($g_hStatusBar, @TAB & StringFormat("HitTest Item: %d", $aHit[0]))
        $g_iIndex = $aHit[0]
    EndIf
EndFunc   ;==>_ListView_Click

Func WM_NOTIFY($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $wParam
    Local $hWndListView = $g_idListView
    If Not IsHWnd($g_idListView) Then $hWndListView = GUICtrlGetHandle($g_idListView)

    Local $tNMHDR = DllStructCreate($tagNMHDR, $lParam)
    Local $hWndFrom = HWnd(DllStructGetData($tNMHDR, "hWndFrom"))
    Local $iCode = DllStructGetData($tNMHDR, "Code")
    Switch $hWndFrom
        Case $hWndListView
            Switch $iCode
                Case $LVN_COLUMNCLICK ; Был кликнут столбец
                    _WM_NOTIFY_DebugEvent("$LVN_COLUMNCLICK", $tagNMLISTVIEW, $lParam, "IDFrom,,Item,SubItem,NewState,OldState,Changed,ActionX,ActionY,Param")
                    ; Нет возвращаемого значения
                Case $LVN_KEYDOWN ; Нажата клавиша
                    _WM_NOTIFY_DebugEvent("$LVN_KEYDOWN", $tagNMLVKEYDOWN, $lParam, "IDFrom,,VKey,Flags")
                    ; Нет возвращаемого значения
                Case $NM_CLICK ; Отправляется элементом list-view, когда пользователь кликает по элементу левой кнопкой мыши
                    _WM_NOTIFY_DebugEvent("$NM_CLICK", $tagNMITEMACTIVATE, $lParam, "IDFrom,,Index,SubItem,NewState,OldState,Changed,X,Y,lParam,KeyFlags")
                    _ListView_Click()
                    ; Нет возвращаемого значения
                Case $NM_DBLCLK ; Отправляется элементом list-view при двойном клике левой кнопкой мыши
                    _WM_NOTIFY_DebugEvent("$NM_DBLCLK", $tagNMITEMACTIVATE, $lParam, "IDFrom,,Index,SubItem,NewState,OldState,Changed,X,Y,lParam,KeyFlags")
                    ; Нет возвращаемого значения
                Case $NM_KILLFOCUS ; Элемент управления потерял фокус ввода
                    _WM_NOTIFY_DebugEvent("$NM_KILLFOCUS", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Нет возвращаемого значения
                Case $NM_RCLICK ; Отправляется элементом list-view при клике правой кнопкой мыши
                    _WM_NOTIFY_DebugEvent("$NM_RCLICK", $tagNMITEMACTIVATE, $lParam, "IDFrom,,Index,SubItem,NewState,OldState,Changed,X,Y,lParam,KeyFlags")
                    ; Return 1 ; чтобы запретить стандартную обработку
                    Return 0 ; разрешить стандартную обработку
                Case $NM_RDBLCLK ; Отправляется элементом list-view при двойном клике правой кнопкой мыши
                    _WM_NOTIFY_DebugEvent("$NM_RDBLCLK", $tagNMITEMACTIVATE, $lParam, "IDFrom,,Index,SubItem,NewState,OldState,Changed,X,Y,lParam,KeyFlags")
                    ; Нет возвращаемого значения
                Case $NM_RETURN ; Элемент управления в фокусе и пользователь нажал ENTER
                    _WM_NOTIFY_DebugEvent("$NM_RETURN", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Нет возвращаемого значения
                Case $NM_SETFOCUS ; Элемент управления получил фокус ввода
                    _WM_NOTIFY_DebugEvent("$NM_SETFOCUS", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Нет возвращаемого значения
            EndSwitch
    EndSwitch
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_NOTIFY
