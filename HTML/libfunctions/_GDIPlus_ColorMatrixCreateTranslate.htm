_GDIPlus_ColorMatrixCreateTranslate

Создаёт и инициализирует матрицу цветов для сдвига компонентов.

#include <GDIPlus.au3>
_GDIPlus_ColorMatrixCreateTranslate ( $fRed, $fGreen, $fBlue [, $fAlpha = 0] )

Параметры

$fRed Коэффициент сдвига красного компонента.
$fGreen Коэффициент сдвига зелёного компонента.
$fBlue Коэффициент сдвига синего компонента.
$fAlpha [необязательный] Коэффициент сдвига альфа-компонента

Возвращаемое значение

Возвращает структуру $tagGDIPCOLORMATRIX, содержащую матрицу сдвига цветов.

Примечания

Матрица сдвига цветов прибавляет (или вычитает) заданные величины к компонентам цвета.

См. также

$tagGDIPCOLORMATRIX

Дополнительно

Искать ColorMatrix в библиотеке MSDN

Пример

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>
#include <ScreenCapture.au3>
#include <WinAPIHObj.au3>

Example()

Func Example()
    _GDIPlus_Startup() ;инициализировать GDI+
    Local Const $iWidth = 600, $iHeight = 600

    Local $hGUI = GUICreate("GDI+ Example (" & @ScriptName & ")", $iWidth, $iHeight) ;создать тестовый GUI
    GUISetState(@SW_SHOW)

    Local $hGraphics = _GDIPlus_GraphicsCreateFromHWND($hGUI) ;создать графический объект из дескриптора окна
    Local $hIA = _GDIPlus_ImageAttributesCreate() ;создать объект ImageAttribute

    Local $tColorMatrix = _GDIPlus_ColorMatrixCreateTranslate(0.75, 0.75, 0.75) ;создать матрицу яркости (RGB)

    _GDIPlus_ImageAttributesSetColorMatrix($hIA, 0, True, $tColorMatrix) ;увеличить яркость изображения на 75%

    Local $hHBmp = _ScreenCapture_Capture("", 0, 0, $iWidth, $iHeight) ;создать GDI bitmap захватом области рабочего стола
    Local $hBitmap = _GDIPlus_BitmapCreateFromHBITMAP($hHBmp) ;преобразовать GDI bitmap в GDI+
    _WinAPI_DeleteObject($hHBmp) ;освободить ресурс GDI bitmap – больше не нужен

    _GDIPlus_GraphicsDrawImageRectRect($hGraphics, $hBitmap, 0, 0, $iWidth, $iHeight, 0, 0, $iWidth, $iHeight, $hIA) ;нарисовать bitmap с применением цветокоррекции

    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE

    ;освободить ресурсы GDI+
    _GDIPlus_ImageAttributesDispose($hIA)
    _GDIPlus_GraphicsDispose($hGraphics)
    _GDIPlus_BitmapDispose($hBitmap)
    _GDIPlus_Shutdown()
    GUIDelete($hGUI)
EndFunc   ;==>Example
