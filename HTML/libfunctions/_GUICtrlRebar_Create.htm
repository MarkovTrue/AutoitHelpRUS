Справочник функций

_GUICtrlRebar_Create

Создаёт элемент управления Rebar.

#include <GuiReBar.au3>
_GUICtrlRebar_Create ( $hWnd [, $iStyles = 0x513] )

Параметры

$hWnd Дескриптор родительского или владеющего окна.
$iStyles [необязательный] Элементы управления Rebar поддерживают различные стили в дополнение к стандартным стилям окна:
$RBS_AUTOSIZE – версия 4.71. Rebar автоматически изменяет расположение полос при изменении размера или позиции элемента управления.
Отправляется уведомление $RBN_AUTOSIZE
$RBS_BANDBORDERS – версия 4.71. Rebar отображает узкие линии для разделения соседних полос
$RBS_DBLCLKTOGGLE – версия 4.71. Полоса переключает состояние развёртывания/свёртывания при двойном щелчке.
Без этого стиля состояние переключается при одиночном щелчке
$RBS_FIXEDORDER – версия 4.70. Rebar всегда отображает полосы в одинаковом порядке. Полосы можно перемещать между строками, но порядок фиксирован
$RBS_REGISTERDROP – версия 4.71. Rebar генерирует уведомления $RBN_GETOBJECT при перетаскивании объекта над полосой
$RBS_TOOLTIPS – версия 4.71. Ещё не поддерживается
$RBS_VARHEIGHT – версия 4.71. Rebar отображает полосы с минимальной требуемой высотой, если возможно.
Без этого стиля все полосы отображаются с одинаковой высотой
$RBS_VERTICALGRIPPER – версия 4.71. Рычаг изменения размера отображается вертикально в вертикальном Rebar.
Этот стиль игнорируется для Rebar без стиля $CCS_VERT
$CCS_LEFT – версия 4.70. Отображает элемент управления вертикально в левой части родительского окна
$CCS_NODIVIDER – предотвращает отрисовку двухпиксельной линии в верхней части элемента управления
$CCS_RIGHT – версия 4.70. Отображает элемент управления вертикально в правой части родительского окна
$CCS_VERT – версия 4.70. Отображает элемент управления вертикально

По умолчанию: $CCS_TOP, $RBS_VARHEIGHT
Обязательные: $WS_CHILD, $WS_VISIBLE, $WS_CLIPCHILDREN, $WS_CLIPSIBLINGS.

Возвращаемое значение

Успех: Возвращает дескриптор элемента управления Rebar.
Ошибка: Устанавливает @error ≠ 0.

См. также

_GUICtrlRebar_Destroy

#include "Extras\WM_NOTIFY.au3"

#include <WinAPIConstants.au3>
#include <WindowsStylesConstants.au3>

Global $g_hReBar

Example()

Func Example()
    Local $hGui = GUICreate("Rebar Create (v" & @AutoItVersion & ")", 400, 396, -1, -1, BitOR($WS_MINIMIZEBOX, $WS_CAPTION, $WS_POPUP, $WS_SYSMENU, $WS_MAXIMIZEBOX))

    _WM_NOTIFY_Register()

    ; создать элемент rebar
    $g_hReBar = _GUICtrlRebar_Create($hGui, BitOR($CCS_TOP, $WS_BORDER, $RBS_VARHEIGHT, $RBS_AUTOSIZE, $RBS_BANDBORDERS))

    ; создать toolbar для размещения в rebar
    Local $hToolbar = _GUICtrlToolbar_Create($hGui, BitOR($TBSTYLE_FLAT, $CCS_NORESIZE, $CCS_NOPARENTALIGN))

    ; Добавить стандартные системные изображения
    Switch _GUICtrlToolbar_GetBitmapFlags($hToolbar)
        Case 0
            _GUICtrlToolbar_AddBitmap($hToolbar, 1, -1, $IDB_STD_SMALL_COLOR)
        Case 2
            _GUICtrlToolbar_AddBitmap($hToolbar, 1, -1, $IDB_STD_LARGE_COLOR)
    EndSwitch

    ; Добавить кнопки
    Local Enum $e_idNew = 1000, $e_idOpen, $e_idSave, $e_idHelp
    _GUICtrlToolbar_AddButton($hToolbar, $e_idNew, $STD_FILENEW)
    _GUICtrlToolbar_AddButton($hToolbar, $e_idOpen, $STD_FILEOPEN)
    _GUICtrlToolbar_AddButton($hToolbar, $e_idSave, $STD_FILESAVE)
    _GUICtrlToolbar_AddButtonSep($hToolbar)
    _GUICtrlToolbar_AddButton($hToolbar, $e_idHelp, $STD_HELP)

    ; создать combobox для размещения в rebar
    Local $hCombo = _GUICtrlComboBox_Create($hGui, "", 0, 0, 120)

    _GUICtrlComboBox_BeginUpdate($hCombo)
    _GUICtrlComboBox_AddDir($hCombo, @WindowsDir & "\*.exe")
    _GUICtrlComboBox_EndUpdate($hCombo)

    ; создание средства выбора даты/времени для размещения в Rebar
    Local $hDTP = _GUICtrlDTP_Create($hGui, 0, 0, 190)

    ; создать поле ввода для размещения в rebar
    ; $hInput = GUICtrlCreateInput("Input control", 0, 0, 120, 20)
    Local $hInput = _GUICtrlEdit_Create($hGui, "Input control", 0, 0, 120, 20)

    ; по умолчанию добавление – в конец

    ; добавить полосу с элементом управления
    _GUICtrlRebar_AddBand($g_hReBar, $hCombo, 120, 200, "Dir *.exe")

    ; добавить полосу со средством выбора даты/времени
    _GUICtrlRebar_AddBand($g_hReBar, $hDTP, 120)

    ; добавить полосу с панелью инструментов в начало Rebar
    _GUICtrlRebar_AddToolBarBand($g_hReBar, $hToolbar, "", 0)

    ; добавить другой элемент управления
    ; _GUICtrlRebar_AddBand($g_hReBar, GUICtrlGetHandle($hInput), 120, 200, "Name:")
    _GUICtrlRebar_AddBand($g_hReBar, $hInput, 120, 200, "Name:")

    Local $idBtn_Exit = GUICtrlCreateButton("Exit", 150, 360, 100, 25)
    GUISetState(@SW_SHOW)

    While 1
        Switch GUIGetMsg()
            Case $GUI_EVENT_CLOSE, $idBtn_Exit
                Exit
        EndSwitch
    WEnd
EndFunc   ;==>Example

Func WM_NOTIFY($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $wParam

    Local $tNMHDR = DllStructCreate($tagNMHDR, $lParam)
    Local $hWndFrom = HWnd(DllStructGetData($tNMHDR, "hWndFrom"))
    Local $iCode = DllStructGetData($tNMHDR, "Code")
    Switch $hWndFrom
        Case $g_hReBar
            Switch $iCode
                Case $RBN_AUTOBREAK
                    ; Уведомляет родителя Rebar о появлении разрыва. Родитель определяет, создавать ли разрыв
                    _WM_NOTIFY_DebugEvent("$RBN_AUTOBREAK", $tagNMREBARAUTOBREAK, $lParam, "IDFrom,,uBand,wID,lParam,uMsg,fStyleCurrent,fAutoBreak")
                    ; Возвращаемое значение не используется
                Case $RBN_AUTOSIZE
                    ; Отправляется Rebar со стилем $RBS_AUTOSIZE при автоматическом изменении размера
                    _WM_NOTIFY_DebugEvent("$RBN_AUTOSIZE", $tagNMRBAUTOSIZE, $lParam, "IDFrom,,fChanged,TargetLeft,TargetTop,TargetRight,TargetBottom,,ActualLeft,ActualTop,ActualRight,ActualBottom")
                    ; Возвращаемое значение не используется
                Case $RBN_BEGINDRAG
                    ; Отправляется при начале перетаскивания полосы пользователем
                    _WM_NOTIFY_DebugEvent("$RBN_BEGINDRAG", $tagNMREBAR, $lParam, "IDFrom,,dwMask,uBand,fStyle,wID,lParam")
                    Return 0 ; для продолжения операции перетаскивания
                    ; Возвращает 1 – ненулевое значение для отмены операции перетаскивания
                Case $RBN_CHEVRONPUSHED
                    ; Отправляется при нажатии шеврона
                    ; При получении этого уведомления приложение отвечает за отображение всплывающего меню с элементами для каждого скрытого инструмента.
                    ; Используйте член rc структуры NMREBARCHEVRON для определения правильной позиции всплывающего меню
                    _WM_NOTIFY_DebugEvent("$RBN_CHEVRONPUSHED", $tagNMREBARCHEVRON, $lParam, "IDFrom,,uBand,wID,lParam,Left,Top,Right,lParamNM")
                    ; Возвращаемое значение не используется
                Case $RBN_CHILDSIZE
                    ; Отправляется при изменении размера дочернего окна полосы
                    _WM_NOTIFY_DebugEvent("$RBN_CHILDSIZE", $tagNMREBARCHILDSIZE, $lParam, "IDFrom,,uBand,wID,CLeft,CTop,CRight,CBottom,BandLeft,,BTop,BRight,BBottom")
                    ; Возвращаемое значение не используется
                Case $RBN_DELETEDBAND
                    ; Отправляется после удаления полосы
                    _WM_NOTIFY_DebugEvent("$RBN_DELETEDBAND", $tagNMREBAR, $lParam, "IDFrom,,dwMask,uBand,fStyle,wID,lParam")
                    ; Возвращаемое значение не используется
                Case $RBN_DELETINGBAND
                    ; Отправляется перед удалением полосы
                    _WM_NOTIFY_DebugEvent("$RBN_DELETINGBAND", $tagNMREBAR, $lParam, "IDFrom,,dwMask,uBand,fStyle,wID,lParam")
                    ; Возвращаемое значение не используется
                Case $RBN_ENDDRAG
                    ; Отправляется при завершении перетаскивания полосы пользователем
                    _WM_NOTIFY_DebugEvent("$RBN_ENDDRAG", $tagNMREBAR, $lParam, "IDFrom,,dwMask,uBand,fStyle,wID,lParam")
                    ; Возвращаемое значение не используется
                Case $RBN_GETOBJECT
                    ; Отправляется Rebar со стилем $RBS_REGISTERDROP при перетаскивании объекта над полосой
                    _WM_NOTIFY_DebugEvent("$RBN_GETOBJECT", $tagNMOBJECTNOTIFY, $lParam, "IDFrom,,Item,piid,pObject,Result")
                    ; Возвращаемое значение не используется
                Case $RBN_HEIGHTCHANGE
                    ; Отправляется при изменении высоты Rebar
                    ; Rebar со стилем $CCS_VERT отправляет это уведомление при изменении ширины
                    _WM_NOTIFY_DebugEvent("$RBN_HEIGHTCHANGE", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Возвращаемое значение не используется
                Case $RBN_LAYOUTCHANGED
                    ; Отправляется при изменении расположения полос пользователем
                    _WM_NOTIFY_DebugEvent("$RBN_LAYOUTCHANGED", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Возвращаемое значение не используется
                Case $RBN_MINMAX
                    ; Отправляется перед развёртыванием или свёртыванием полосы
                    _WM_NOTIFY_DebugEvent("$RBN_MINMAX", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Возвращает 1 – ненулевое значение для отмены операции
                    Return 0 ; ноль для продолжения
            EndSwitch
    EndSwitch
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_NOTIFY
