Справочник функций

_WinAPI_CreateANDBitmap

создаёт маску AND DIB из указанного растрового изображения.

#include <WinAPIGdi.au3>
_WinAPI_CreateANDBitmap ( $hBitmap )

Параметры

$hBitmap Дескриптор растрового изображения.

Возвращаемое значение

Успех: Возвращает дескриптор на созданное растровое изображение.
Ошибка: Возвращает 0 и устанавливает @error ≠ 0.

Примечания

Функция _WinAPI_CreateANDBitmap() создаёт 1-битный AND-маскированный DIB из 32-битного DIB или DDB.
Если исходный битмап не 32-битный или сжат, функция завершается с ошибкой.

Возвращённый битмап можно использовать для создания значка через _WinAPI_CreateIconIndirect().

Когда битмап больше не нужен, удалите его функцией _WinAPI_DeleteObject().

См. также

_WinAPI_CreateIconIndirect, _WinAPI_DeleteObject

Пример

#include "Extras\HelpFileInternals.au3"

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>
#include <StaticConstants.au3>
#include <WinAPIGdi.au3>
#include <WinAPIIcons.au3>
#include <WinAPIShPath.au3>

Local Const $sPng = _Extras_PathFull('Silverlight.png')

Local $a_hBitmap[2]
; Создание растрового изображения 32 бит/пикс
Global $g_hDll = _GDIPlus_Startup(Default, True)
Local $hPng = _GDIPlus_ImageLoadFromFile($sPng)
$a_hBitmap[0] = _GDIPlus_BitmapCreateDIBFromBitmap($hPng)
Local $iWidth = _GDIPlus_ImageGetWidth($hPng)
Local $iHeight = _GDIPlus_ImageGetHeight($hPng)
_GDIPlus_ImageDispose($hPng)
_GDIPlus_Shutdown()

; Создание растрового изображения 1 бит/пикс
$a_hBitmap[1] = _WinAPI_CreateANDBitmap($a_hBitmap[0])

; Создать GUI
Local $hForm = GUICreate('Test ' & StringReplace(@ScriptName, '.au3', '()'), $iWidth * 2, $iHeight)
Local $aidPic[2]
$aidPic[0] = GUICtrlCreatePic('', 0, 0, $iWidth, $iHeight)
$aidPic[1] = GUICtrlCreatePic('', $iWidth, 0, $iWidth, $iHeight)

; Задать оба bitmap для элементов управления
For $i = 0 To 1
    GUICtrlSendMsg($aidPic[$i], $STM_SETIMAGE, 0, $a_hBitmap[$i])
Next

; Показать GUI
GUISetState(@SW_SHOW)

; Создание и сохранение значка в файл .ico
Local $hIcon
Local $sPath = FileSaveDialog('Save Icon', @ScriptDir, 'Icon Files (*.ico)', 2 + 16, _WinAPI_PathStripPath(_WinAPI_PathRenameExtension($sPng, '.ico')), $hForm)
If $sPath Then
    $hIcon = _WinAPI_CreateIconIndirect($a_hBitmap[0], $a_hBitmap[1])
    If $hIcon Then
        _WinAPI_SaveHICONToFile($sPath, $hIcon)
        _WinAPI_DestroyIcon($hIcon)
    EndIf
EndIf

Do
Until GUIGetMsg() = $GUI_EVENT_CLOSE
