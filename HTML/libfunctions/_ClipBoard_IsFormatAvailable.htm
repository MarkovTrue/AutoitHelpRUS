_ClipBoard_IsFormatAvailable

Проверяет, содержит ли буфер обмена данные в указанном формате.

#include <Clipboard.au3>
_ClipBoard_IsFormatAvailable ( $iFormat )

Параметры

$iFormat Стандартный или зарегистрированный формат буфера обмена.

Возвращаемое значение

Успех: Возвращает True – буфер обмена содержит данные в указанном формате.
Ошибка: Возвращает False – буфер обмена не содержит данные в указанном формате.

Дополнительно

Искать IsClipboardFormatAvailable в библиотеке MSDN

Пример

#include "Extras\HelpFileInternals.au3"

#include <Clipboard.au3>
#include <GUIConstantsEx.au3>
#include <WindowsStylesConstants.au3>

Example()

Func Example()
    Local $aFormats[3] = [2, $CF_TEXT, $CF_OEMTEXT]

    ; Создать GUI
    GUICreate("Clipboard", 600, 400)
    _MemoCreate(2, 2, 596, 396, $WS_VSCROLL)
    GUISetState(@SW_SHOW)

    ; Проверить, есть ли текст в буфере обмена
    _MemoWrite("Priority formats .:. " & _ClipBoard_GetPriorityFormat($aFormats))
    _MemoWrite("Unicode available .: " & _ClipBoard_IsFormatAvailable($CF_UNICODETEXT))

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
EndFunc   ;==>Example
