Справочник функций

_GUICtrlButton_Create

Создаёт элемент управления Button.

#include <GuiButton.au3>
_GUICtrlButton_Create ( $hWnd, $sText, $iX, $iY, $iWidth, $iHeight [, $iStyle = -1 [, $iExStyle = -1]] )

Параметры

$hWnd Дескриптор родительского или владеющего окна.
$sText Текст для добавления к кнопке.
$iX Горизонтальная позиция элемента управления.
$iY Вертикальная позиция элемента управления.
$iWidth Ширина элемента управления.
$iHeight Высота элемента управления.
$iStyle [необязательный] Стиль элемента управления:
$BS_AUTO3STATE – создаёт флажок с тремя состояниями: установлен, недоступен и снят. Состояние переключается циклически при каждом выборе флажка пользователем.
$BS_AUTOCHECKBOX – создаёт флажок, состояние которого переключается между установленным и снятым при каждом выборе пользователем.
$BS_AUTORADIOBUTTON – то же, что и переключатель, но при выборе пользователем кнопка автоматически выделяется и снимает выделение с других переключателей того же стиля в той же группе.
$BS_FLAT – задаёт двухмерное оформление кнопки без объёмного эффекта.
$BS_GROUPBOX – создаёт прямоугольник для группировки других кнопок. Текст, связанный с этим стилем, отображается в верхнем левом углу прямоугольника.
$BS_PUSHLIKE – заставляет кнопку (флажок, трёхпозиционный флажок или переключатель) выглядеть и работать как обычная кнопка. Она выглядит выпуклой в обычном состоянии и вдавленной при нажатии или установке.

$BS_DEFPUSHBUTTON – создаёт кнопку с жирной чёрной рамкой. Если кнопка находится в диалоговом окне, пользователь может активировать её нажатием Enter, даже если она не в фокусе ввода. Удобно для быстрого выбора наиболее вероятного варианта или значения по умолчанию.

$BS_BOTTOM – размещает текст в нижней части прямоугольника кнопки.
$BS_CENTER – центрирует текст по горизонтали.
$BS_LEFT – выравнивает текст по левому краю.
$BS_MULTILINE – переносит текст кнопки на несколько строк, если он не помещается в одну строку.
$BS_RIGHT – выравнивает текст по правому краю.
$BS_RIGHTBUTTON – размещает квадрат флажка справа от прямоугольника кнопки.
$BS_TOP – размещает текст в верхней части прямоугольника кнопки.
$BS_VCENTER – центрирует текст по вертикали.

$BS_ICON – указывает, что на кнопке отображается значок.
$BS_BITMAP – указывает, что на кнопке отображается растровое изображение.

$BS_NOTIFY – позволяет кнопке отправлять родительскому окну уведомления BN_KILLFOCUS и BN_SETFOCUS. Сообщение BN_CLICKED кнопка отправляет независимо от наличия этого стиля. Для получения BN_DBLCLK кнопка должна иметь стиль BS_RADIOBUTTON или BS_OWNERDRAW.

Стили Vista:
$BS_SPLITBUTTON – создаёт составную кнопку с раскрывающейся стрелкой.
$BS_DEFSPLITBUTTON – создаёт составную кнопку, которая ведёт себя как $BS_PUSHBUTTON, но с особым внешним видом.
$BS_COMMANDLINK – создаёт кнопку-команду.
$BS_DEFCOMMANDLINK – создаёт кнопку-команду, ведущую себя как $BS_PUSHBUTTON.

По умолчанию: (-1) – нет.
Принудительно: $WS_CHILD, $WS_TABSTOP, $WS_VISIBLE, $BS_NOTIFY.
$iExStyle [необязательный] Расширенный стиль элемента. Соответствует стандартным константам $WS_EX_*. См. таблицу расширенных стилей.

Возвращаемое значение

Успех: Возвращает дескриптор элемента управления Button.
Ошибка: Возвращает 0.

Примечания

Указанные выше константы требуют #include <ButtonConstants.au3>.

Эта функция предназначена для опытных пользователей и для изучения работы элемента управления.

См. также

_GUICtrlButton_Destroy

Пример

#include "Extras\HelpFileInternals.au3"
#include "Extras\WM_NOTIFY.au3"

#include <GuiButton.au3>
#include <GUIConstantsEx.au3>
#include <StructureConstants.au3>
#include <WindowsNotifsConstants.au3>
#include <WindowsStylesConstants.au3>

Global $g_hBtn, $g_hRdo, $g_hChk

; Примечание: дескриптор этих кнопок НЕЛЬЗЯ прочитать через GUICtrlRead

Example()

Func Example()
    Local $hGUI = GUICreate("Button Create (v" & @AutoItVersion & ")", 400, 400)
    _MemoCreate(119, 10, 276, 374, $WS_VSCROLL)

    $g_hBtn = _GUICtrlButton_Create($hGUI, "Button1", 10, 10, 90, 50)

    $g_hRdo = _GUICtrlButton_Create($hGUI, "Radio1", 10, 60, 90, 50, $BS_AUTORADIOBUTTON)

    $g_hChk = _GUICtrlButton_Create($hGUI, "Check1", 10, 120, 90, 50, $BS_AUTO3STATE)

    GUIRegisterMsg($WM_COMMAND, "WM_COMMAND")
    _WM_NOTIFY_Register($_g_idLst_Memo)

    GUISetState(@SW_SHOW)

    _MemoWrite("$g_hBtn handle: " & $g_hBtn)
    _MemoWrite("$g_hRdo handle: " & $g_hRdo)
    _MemoWrite("$g_hChk handle: " & $g_hChk & @CRLF)

    While 1
        Switch GUIGetMsg()
            Case $GUI_EVENT_CLOSE
                ExitLoop
        EndSwitch
    WEnd

    Exit
EndFunc   ;==>Example

Func WM_NOTIFY($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $wParam
    Local Const $BCN_HOTITEMCHANGE = -1249
    Local $tagNMBHOTITEM = $tagNMHDR & ";dword dwFlags"
    Local $tNMBHOTITEM = DllStructCreate($tagNMBHOTITEM, $lParam)
    Local $iCode = DllStructGetData($tNMBHOTITEM, "Code")
    Local $hWndFrom = DllStructGetData($tNMBHOTITEM, "hWndFrom")
    Local $iFlags = DllStructGetData($tNMBHOTITEM, "dwFlags")
    Local $sText = ""

    Switch $iCode
        Case $BCN_HOTITEMCHANGE ; Win XP и выше
            $sText = "Text=" & _GUICtrlButton_GetText($hWndFrom)
            If BitAND($iFlags, 0x10) = 0x10 Then
                _WM_NOTIFY_DebugEvent("$BCN_HOTITEMCHANGE - Entering", $tagNMBHOTITEM, $lParam, "IDFrom", $sText)
            ElseIf BitAND($iFlags, 0x20) = 0x20 Then
                _WM_NOTIFY_DebugEvent("$BCN_HOTITEMCHANGE - Leaving", $tagNMBHOTITEM, $lParam, "IDFrom", $sText)
            EndIf
    EndSwitch
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_NOTIFY

; Реакция на клик по кнопке
Func WM_COMMAND($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg
    Local $iCode = BitShift($wParam, 16)
    Local $hCtrl = $lParam
    Local $sCode, $sText

    Switch $hCtrl
        Case $g_hBtn, $g_hRdo, $g_hChk
            Switch $iCode
                Case $BN_CLICKED
                    $sCode = "$BN_CLICKED"
                Case $BN_PAINT
                    $sCode = "$BN_PAINT"
                Case $BN_PUSHED
                    $sCode = "$BN_PUSHED"
                Case $BN_HILITE
                    $sCode = "$BN_HILITE"
                Case $BN_UNPUSHED
                    $sCode = "$BN_UNPUSHED"
                Case $BN_UNHILITE
                    $sCode = "$BN_UNHILITE"
                Case $BN_DISABLE
                    $sCode = "$BN_DISABLE"
                Case $BN_DBLCLK
                    $sCode = "$BN_DBLCLK"
                Case $BN_DOUBLECLICKED
                    $sCode = "$BN_DOUBLECLICKED"
                Case $BN_SETFOCUS
                    $sCode = "$BN_SETFOCUS"
                Case $BN_KILLFOCUS
                    $sCode = "$BN_KILLFOCUS"
            EndSwitch
            $sText = "Text=" & _GUICtrlButton_GetText($hCtrl)
            _WM_NOTIFY_DebugEvent($sCode, $tagNMHDR, $lParam, "IDFrom", $sText)
            Return 0 ; Работает только при клике по кнопке
    EndSwitch
    ; Продолжить стандартные внутренние команды сообщений AutoIt3.
    ; Эту строку также можно вообще пропустить.
    ; !!! Но только 'Return' (без значения) не продолжит
    ; стандартную обработку сообщений AutoIt3 в дальнейшем !!!
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_COMMAND
