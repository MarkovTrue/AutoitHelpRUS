Справочник функций

_WinAPI_BitBlt

Выполняет поблочную передачу растра цветовых данных.

#include <WinAPIGdi.au3>
_WinAPI_BitBlt ( $hDestDC, $iXDest, $iYDest, $iWidth, $iHeight, $hSrcDC, $iXSrc, $iYSrc, $iROP )

Параметры

$hDestDC Дескриптор контекста устройства назначения.
$iXDest Значение X верхнего левого угла целевого прямоугольника.
$iYDest Значение Y верхнего левого угла целевого прямоугольника.
$iWidth Ширина исходного и целевого прямоугольников.
$iHeight Высота исходного и целевого прямоугольников.
$hSrcDC Дескриптор исходного контекста устройства.
$iXSrc Значение X верхнего левого угла исходного прямоугольника.
$iYSrc Значение Y верхнего левого угла исходного прямоугольника.
$iROP Задаёт код растровой операции. Эти коды определяют способ объединения цветовых данных исходного прямоугольника с цветовыми данными целевого прямоугольника для получения итогового цвета:
$BLACKNESS - заполняет целевой прямоугольник цветом, связанным с индексом палитры 0
$CAPTUREBLT - включает окна, которые находятся поверх вашего окна в результирующем изображении
$DSTINVERT - инвертирует целевой прямоугольник
$MERGECOPY - объединяет цвет исходного прямоугольника с кистью, выбранной в hDest, используя оператор AND
$MERGEPAINT - объединяет цвет инвертированного исходного прямоугольника с цветами целевого прямоугольника, используя оператор OR
$NOMIRRORBITMAP - предотвращает зеркальное отражение растрового изображения
$NOTSRCCOPY - копирует инвертированный исходный прямоугольник в целевой
$NOTSRCERASE - объединяет цвета исходного и целевого прямоугольников, используя оператор OR, и затем инвертирует полученный цвет
$PATCOPY - копирует кисть, выбранную в hdcDest, в целевое растровое изображение
$PATINVERT - объединяет цвета кисти, выбранной в hDest, с цветами целевого прямоугольника, используя оператор XOR
$PATPAINT - объединяет цвета кисти, выбранной в hDest, с цветами инвертированного исходного прямоугольника, используя оператор OR. Результат этой операции объединяется с цветом целевого прямоугольника, используя оператор OR
$SRCAND - объединяет цвета исходного и целевого прямоугольников, используя оператор AND
$SRCCOPY - копирует исходный прямоугольник напрямую в целевой прямоугольник
$SRCERASE - объединяет инвертированный цвет целевого прямоугольника с цветами исходного прямоугольника, используя оператор AND
$SRCINVERT - объединяет цвета исходного и целевого прямоугольников, используя оператор XOR
$SRCPAINT - объединяет цвета исходного и целевого прямоугольников, используя оператор OR
$WHITENESS - заполняет целевой прямоугольник цветом, связанным с индексом 1 в физической палитре.

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False.

Пример

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>
#include <WinAPIGdi.au3>
#include <WinAPIGdiDC.au3>
#include <WinAPIHObj.au3>

Global $g_hGUI, $g_hHBITMAP, $g_hDC, $g_hDC_Backbuffer, $g_oDC_Obj, $g_hGfxCtxt, $g_hPen

Example()

Func Example()
    AutoItSetOption("GUIOnEventMode", 1)

    _GDIPlus_Startup() ; инициализировать GDI+
    Local Const $iWidth = 600, $iHeight = 600, $iBgColor = 0x202020 ; $iBGColor в формате RRGGBB

    $g_hGUI = GUICreate("GDI+ Example (" & @ScriptName & ")", $iWidth, $iHeight) ; создать тестовый GUI
    GUISetBkColor($iBgColor, $g_hGUI) ; установить цвет фона GUI
    GUISetState(@SW_SHOW)

    ; создание более быстрого набора буферизированного графического кадра для плавного движения объектов
    Local $hBitmap = _GDIPlus_BitmapCreateFromScan0($iWidth, $iHeight) ; создать пустой bitmap
    $g_hHBITMAP = _GDIPlus_BitmapCreateHBITMAPFromBitmap($hBitmap) ; преобразование растрового изображения GDI+ в растровое изображение GDI
    _GDIPlus_BitmapDispose($hBitmap) ; удаление растрового изображения GDI+ так как оно больше не нужно

    $g_hDC = _WinAPI_GetDC($g_hGUI) ; получение контекста устройства из GUI
    $g_hDC_Backbuffer = _WinAPI_CreateCompatibleDC($g_hDC) ; создание контекста устройства памяти, совместимого с указанным устройством
    $g_oDC_Obj = _WinAPI_SelectObject($g_hDC_Backbuffer, $g_hHBITMAP) ; выбор объекта в указанный контекст устройства
    $g_hGfxCtxt = _GDIPlus_GraphicsCreateFromHDC($g_hDC_Backbuffer) ; создание графического объекта из контекста устройства (DC)
    _GDIPlus_GraphicsSetSmoothingMode($g_hGfxCtxt, $GDIP_SMOOTHINGMODE_HIGHQUALITY) ; установить режим сглаживания (box-фильтр 8 X 4)
    _GDIPlus_GraphicsSetPixelOffsetMode($g_hGfxCtxt, $GDIP_PIXELOFFSETMODE_HIGHQUALITY)

    $g_hPen = _GDIPlus_PenCreate() ; создание объекта пера

    GUISetOnEvent($GUI_EVENT_CLOSE, "_Exit")

    Local Const $fDeg = ACos(-1) / 180 ; ACos(-1) приблизительно равен Pi
    Local $iX_Center = $iWidth / 2, $iY_Center = $iHeight / 2, $iXPos, $iYPos, $fAngle = 0, $iRound = 0
    Local Const $iDots = 7, $fAngelDist = 360 / $iDots, $iRadius = 250
    Local $aCoordinates[$iDots][2] ; создание массива для сохранения координат x/y
    Do
        _GDIPlus_GraphicsClear($g_hGfxCtxt, 0xFF000000 + $iBgColor) ; Очистить изображение заданным цветом (формат AARRGGBB).
        For $i = 0 To $iDots - 1
            $iXPos = $iX_Center + Cos($fAngle * $fDeg) * $iRadius ; расчёт позиции x
            $iYPos = $iY_Center + Sin($fAngle * $fDeg) * $iRadius ; расчёт позиции y
            $aCoordinates[$i][0] = $iXPos
            $aCoordinates[$i][1] = $iYPos
            _GDIPlus_PenSetColor($g_hPen, 0xFFFFFF00) ; установка цвета пера для внутренних линий
            _GDIPlus_PenSetWidth($g_hPen, 2) ; установка размера пера для внешних линий
            _GDIPlus_GraphicsDrawLine($g_hGfxCtxt, $aCoordinates[$i][0], $aCoordinates[$i][1], _ ; рисование внутренних линий
                    $aCoordinates[Mod(($i + $iDots / 2), $iDots)][0], $aCoordinates[Mod(($i + $iDots / 2), $iDots)][1], $g_hPen) ; рисование на противоположной стороне
            _GDIPlus_PenSetColor($g_hPen, 0xFFFF8000) ; установка цвета пера
            _GDIPlus_PenSetWidth($g_hPen, 3) ; установка размера пера
            ; массив координат должен быть заполнен перед первой отрисовкой на экран
            If $i < $iDots - 1 Then _GDIPlus_GraphicsDrawLine($g_hGfxCtxt, $aCoordinates[$i][0], $aCoordinates[$i][1], $aCoordinates[$i + 1][0], $aCoordinates[$i + 1][1], $g_hPen) ;; рисование внешних линий
            $fAngle += $fAngelDist ; Увеличить угол до следующей точки.
        Next
        ; рисование последней линии к первой линии
        _GDIPlus_GraphicsDrawLine($g_hGfxCtxt, $aCoordinates[$i - 1][0], $aCoordinates[$i - 1][1], $aCoordinates[0][0], $aCoordinates[0][1], $g_hPen)

        If $iRound Then _WinAPI_BitBlt($g_hDC, 0, 0, $iWidth, $iHeight, $g_hDC_Backbuffer, 0, 0, $SRCCOPY) ; копирование заднего буфера на экран (GUI)
        $fAngle -= 0.5 ; уменьшение общего угла
        $iRound += 1
    Until Not Sleep(30) ; Sleep() always returns 1 and Not 1 is 0 correspond to False

    _Exit()
EndFunc   ;==>Example

Func _Exit() ; освободить ресурсы GDI+
    _GDIPlus_PenDispose($g_hPen)
    _WinAPI_SelectObject($g_hDC_Backbuffer, $g_oDC_Obj)
    _GDIPlus_GraphicsDispose($g_hGfxCtxt)
    _WinAPI_DeleteObject($g_hHBITMAP)
    _WinAPI_ReleaseDC($g_hGUI, $g_hDC)
    GUIDelete($g_hGUI)
    Exit
EndFunc   ;==>_Exit
