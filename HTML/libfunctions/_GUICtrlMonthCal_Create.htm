_GUICtrlMonthCal_Create

Создаёт элемент управления месячный календарь.

#include <GuiMonthCal.au3>
_GUICtrlMonthCal_Create ( $hWnd, $iX, $iY [, $iStyle = 0x00000000 [, $iExStyle = 0x00000000]] )

Параметры

$hWnd Дескриптор родительского или владеющего окна.
$iX Горизонтальная позиция элемента управления.
$iY Вертикальная позиция элемента управления.
$iStyle [необязательный] Стили элемента управления:
$MCS_DAYSTATE – месячный календарь отправляет уведомления $MCN_GETDAYSTATE для запроса информации о том, какие дни должны отображаться жирным шрифтом.
$MCS_MULTISELECT – месячный календарь позволяет пользователю выбирать диапазон дат в элементе управления
$MCS_WEEKNUMBERS – элемент управления месячный календарь отображает номера недель слева от каждой строки дней
$MCS_NOTODAYCIRCLE – элемент управления месячный календарь не выделяет кружком дату сегодня
$MCS_NOTODAY – элемент управления месячный календарь не отображает дату сегодня внизу

Обязательные: $WS_CHILD, $WS_VISIBLE.
$iExStyle [необязательный] Расширенный стиль элемента. Соответствует стандартным константам $WS_EX_*. См. таблицу расширенных стилей.

Возвращаемое значение

Успех: Возвращает дескриптор окна месячного календаря.
Ошибка: Возвращает 0.

Примечания

Данная функция предназначена для опытных пользователей и для изучения принципов работы элемента управления.

См. также

_GUICtrlMonthCal_Destroy, _GUICtrlMonthCal_GetColorArray

Пример

#include "Extras\WM_NOTIFY.au3"

#include <GUIConstantsEx.au3>
#include <GuiMonthCal.au3>
#include <StructureConstants.au3>
#include <WindowsStylesConstants.au3>

Global $g_hMonthCal, $g_idMemo

Example()

Func Example()
    ; Создать GUI
    Local $hGUI = GUICreate("Month Calendar Create (v" & @AutoItVersion & ")", 400, 300)
    $g_hMonthCal = _GUICtrlMonthCal_Create($hGUI, 4, 4, $WS_BORDER)

    ; Создать элемент memo
    $g_idMemo = GUICtrlCreateEdit("", 4, 168, 392, 128, 0)
    GUICtrlSetFont($g_idMemo, 9, 400, 0, "Courier New")
    GUISetState(@SW_SHOW)

    _WM_NOTIFY_Register($g_idMemo)

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
    GUIDelete()
EndFunc   ;==>Example

Func WM_NOTIFY($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $wParam

    Local $tNMHDR = DllStructCreate($tagNMHDR, $lParam)
    Local $hWndFrom = HWnd(DllStructGetData($tNMHDR, "hWndFrom"))
    Local $iCode = DllStructGetData($tNMHDR, "Code")
    Switch $hWndFrom
        Case $g_hMonthCal
            Switch $iCode
                Case $MCN_GETDAYSTATE ; Отправляется элементом управления месячный календарь для запроса информации о том, как должны отображаться отдельные дни
                    _WM_NOTIFY_DebugEvent("$MCN_GETDAYSTATE", $tagNMDAYSTATE, $lParam, "IDFrom,,Year,Month,DOW,Day,,Hour,Minute,Second,MSecond,DayState,pDayState")
                    ; Адрес массива MONTHDAYSTATE (поле битов DWORD, содержащее состояние каждого дня месяца)
                    ; Каждый бит (с 1 по 31) представляет состояние дня месяца. Если бит установлен, соответствующий день будет
                    ; отображаться жирным шрифтом; иначе он будет отображаться обычным шрифтом.
                    ; Нет возвращаемого значения
                Case $MCN_SELCHANGE ; Отправляется элементом управления месячный календарь при изменении текущей выбранной даты или диапазона дат
                    _WM_NOTIFY_DebugEvent("$MCN_SELCHANGE", $tagNMSELCHANGE, $lParam, "IDFrom,,BegYear,BegMonth,BegDOW,BegDay,,BegHour,BegMinute,BegSecond,BegMSecond,,EndYear,EndMonth,EndDOW,EndDay,,EndHour,EndMinute,EndSecond,EndMSeconds")
                    ; Нет возвращаемого значения
                Case $MCN_SELECT ; Отправляется элементом управления месячный календарь, когда пользователь делает явный выбор даты в элементе управления месячный календарь
                    _WM_NOTIFY_DebugEvent("$MCN_SELECT", $tagNMSELCHANGE, $lParam, "IDFrom,,BegYear,BegMonth,BegDOW,BegDay,,BegHour,BegMinute,BegSecond,BegMSecond,,EndYear,EndMonth,EndDOW,EndDay,,EndHour,EndMinute,EndSecond,EndMSeconds")
                    ; Нет возвращаемого значения
            EndSwitch
    EndSwitch
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_NOTIFY
