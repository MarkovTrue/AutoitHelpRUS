_GUIToolTip_ToolExists

Определяет, существует ли инструмент в элементе управления ToolTip.

#include <GuiToolTip.au3>
_GUIToolTip_ToolExists ( $hTool )

Параметры

$hTool Дескриптор элемента управления ToolTip (возвращается функцией _GUIToolTip_Create).

Возвращаемое значение

True: Возвращает инструмент существует.
False: инструмент не существует.

Примечания

Эта функция вернёт true только если элемент управления ToolTip в данный момент отображает один из добавленных инструментов.

Пример

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <GUIToolTip.au3>
#include <MsgBoxConstants.au3>

Global $g_hToolTip

; Нажмите клавишу "G" для отображения информации
HotKeySet("g", "Get_Tool")

Example()

Func Example()
    Local $hGUI = GUICreate("ToolTip ToolExists - v(" & @AutoItVersion & ")", 450, 300, 100, 100)

    Local $idButton = GUICtrlCreateButton("Button ToolTip", 30, 32, 130, 28)
    Local $hButton = GUICtrlGetHandle($idButton)
    ; создать элемент tooltip с настройками по умолчанию
    $g_hToolTip = _GUIToolTip_Create(0)

;~ $hGUI = 0 ; is OK
    ; добавить tool в элемент tooltip
    _GUIToolTip_AddTool($g_hToolTip, $hGUI, "This is a ToolTip", $hButton)

    GUISetState(@SW_SHOW)

    ; Показать tooltip, связанный с кнопкой
    Opt("MouseCoordMode", 2)
    MouseMove(50, 42, 0)
    Sleep(250)

    Get_Tool()

    While 1
        If GUIGetMsg() = $GUI_EVENT_CLOSE Then ExitLoop
    WEnd

    ; Уничтожить элемент tooltip
    _GUIToolTip_Destroy($g_hToolTip)
    GUIDelete($hGUI)
EndFunc   ;==>Example

Func Get_Tool()
    ; Это отобразит "True" если инструмент отображается
    _MemoMsgBox($MB_SYSTEMMODAL, "", "Tooltip Exists = " & _GUIToolTip_ToolExists($g_hToolTip))
EndFunc   ;==>Get_Tool
