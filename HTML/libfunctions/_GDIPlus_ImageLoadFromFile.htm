Справочник функций

_GDIPlus_ImageLoadFromFile

Создаёт объект Image на основе файла.

#include <GDIPlus.au3>
_GDIPlus_ImageLoadFromFile ( $sFileName )

Параметры

$sFileName Полное имя файла изображения.

Возвращаемое значение

Успех: Возвращает дескриптор нового объекта Image.
Ошибка: Возвращает 0 и устанавливает @error ≠ 0.
@extended: Может содержать код ошибки GPSTATUS ($GDIP_ERR* см. GPIPlusConstants.au3).

См. также

_GDIPlus_ImageSaveToFile, _GDIPlus_ImageSaveToFileEx

Дополнительно

Искать GdipLoadImageFromFile в библиотеке MSDN

Пример

Пример 1

#include <GDIPlus.au3>
#include <ScreenCapture.au3>

Example()

Func Example()
    Local $hImage, $sCLSID, $tData, $tParams

    ; Захват экрана
    _ScreenCapture_Capture(@MyDocumentsDir & "\GDIPlus_Image.jpg")

    ; Инициализировать библиотеку GDI+
    _GDIPlus_Startup()

    ; Загрузить изображение
    $hImage = _GDIPlus_ImageLoadFromFile(@MyDocumentsDir & "\GDIPlus_Image.jpg")

    ; Получить CLSID JPEG-кодера
    $sCLSID = _GDIPlus_EncodersGetCLSID("JPG")

    ; Установить параметры поворота на 90 градусов
    $tData = DllStructCreate("int Data")
    DllStructSetData($tData, "Data", $GDIP_EVTTRANSFORMROTATE90)
    $tParams = _GDIPlus_ParamInit(1)
    _GDIPlus_ParamAdd($tParams, $GDIP_EPGTRANSFORMATION, 1, $GDIP_EPTLONG, DllStructGetPtr($tData, "Data"))

    ; Сохранить изображение с поворотом
    _GDIPlus_ImageSaveToFileEx($hImage, @MyDocumentsDir & "\GDIPlus_Image2.jpg", $sCLSID, $tParams)

    ; Освободить ресурсы
    _GDIPlus_ImageDispose($hImage)

    ; Завершить работу с библиотекой GDI+
    _GDIPlus_Shutdown()

    ShellExecute(@MyDocumentsDir & "\GDIPlus_Image2.jpg")
EndFunc   ;==>Example

Пример 2

; Обходное решение для PNG от Zedna

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>
#include <WinAPIGdi.au3>
#include <WindowsNotifsConstants.au3>

; Создать GUI
Global $g_hGUI = GUICreate("Show PNG", 250, 250)

; Загрузить изображение PNG
_GDIPlus_Startup()
Global $g_hImage = _GDIPlus_ImageLoadFromFile("..\GUI\Torus.png")
Global $g_hGraphic = _GDIPlus_GraphicsCreateFromHWND($g_hGUI)

GUIRegisterMsg($WM_PAINT, "MY_WM_PAINT")
GUISetState(@SW_SHOW)

; Цикл до выхода пользователя.
While 1
    Switch GUIGetMsg()
        Case $GUI_EVENT_CLOSE
            ExitLoop

    EndSwitch
WEnd

; Освободить ресурсы
_GDIPlus_GraphicsDispose($g_hGraphic)
_GDIPlus_ImageDispose($g_hImage)
_GDIPlus_Shutdown()

; Нарисовать изображение PNG
Func MY_WM_PAINT($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $wParam, $lParam
    _WinAPI_RedrawWindow($g_hGUI, 0, 0, $RDW_UPDATENOW)
    _GDIPlus_GraphicsDrawImage($g_hGraphic, $g_hImage, 0, 0)
    _WinAPI_RedrawWindow($g_hGUI, 0, 0, $RDW_VALIDATE)
    Return $GUI_RUNDEFMSG
EndFunc   ;==>MY_WM_PAINT
