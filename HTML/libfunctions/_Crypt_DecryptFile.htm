Справочник функций

_Crypt_DecryptFile

Расшифровывает файл с помощью указанного ключа и алгоритма.

#include <Crypt.au3>
_Crypt_DecryptFile ( $sSourceFile, $sDestinationFile, $vCryptKey, $iAlgID )

Параметры

$sSourceFile Файл для обработки.
$sDestinationFile Файл для сохранения обработанного результата.
$vCryptKey Пароль или дескриптор ключа при использовании флага CALG_USERKEY.
$iAlgID Используемый алгоритм. См. _Crypt_DeriveKey().

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False и устанавливает @error ≠ 0.
@error: 2 – не удалось открыть исходный файл
3 – не удалось открыть файл назначения
10–30 – не удалось создать ключ
400+ – ошибка расшифровки последнего блока
500+ – ошибка расшифровки блока
1000+ – ошибка _Crypt_Startup()

См. также

_Crypt_DeriveKey, _Crypt_EncryptFile

Пример

#include <ComboConstants.au3>
#include <Crypt.au3>
#include <GUIConstantsEx.au3>
#include <MsgBoxConstants.au3>
#include <StringConstants.au3>

Example()

Func Example()
    Local $iAlgorithm = $CALG_RC4

    Local $hGUI = GUICreate("File Decrypter", 425, 100)
    Local $idInp_Source = GUICtrlCreateInput("", 5, 5, 200, 20)
    Local $idBtn_SourceBrowse = GUICtrlCreateButton("...", 210, 5, 35, 20)

    Local $idInp_Destination = GUICtrlCreateInput("", 5, 30, 200, 20)
    Local $idBtn_DestinationBrowse = GUICtrlCreateButton("...", 210, 30, 35, 20)

    GUICtrlCreateLabel("Password:", 5, 60, 200, 20)
    Local $idInp_Password = GUICtrlCreateInput("", 5, 75, 200, 20)

    Local $idCombo = GUICtrlCreateCombo("", 210, 75, 100, 20, $CBS_DROPDOWNLIST)
    GUICtrlSetData($idCombo, "3DES (168bit)|AES (128bit)|AES (192bit)|AES (256bit)|DES (56bit)|RC2 (128bit)|RC4 (128bit)", "RC4 (128bit)")
    Local $idBtn_Decrypt = GUICtrlCreateButton("Decrypt", 355, 70, 65, 25)
    GUISetState(@SW_SHOW, $hGUI)

    Local $sDestinationRead = "", $sFilePath = "", $sPasswordRead = "", $sSourceRead = ""
    While 1
        Switch GUIGetMsg()
            Case $GUI_EVENT_CLOSE
                ExitLoop

            Case $idBtn_SourceBrowse
                $sFilePath = FileOpenDialog("Select a file to decrypt.", "", "All files (*.*)") ; Выбираем файл для расшифровки.
                If @error Then
                    ContinueLoop
                EndIf
                GUICtrlSetData($idInp_Source, $sFilePath) ; Задать поле ввода путём к файлу.

            Case $idBtn_DestinationBrowse
                $sFilePath = FileSaveDialog("Save the file as ...", "", "All files (*.*)") ; Выбираем файл для сохранения расшифрованных данных.
                If @error Then
                    ContinueLoop
                EndIf
                GUICtrlSetData($idInp_Destination, $sFilePath) ; Задать поле ввода путём к файлу.

            Case $idCombo ; Проверить выбор в combobox и получить соответствующий алгоритм.
                Switch GUICtrlRead($idCombo) ; Прочитать выбор в combobox.
                    Case "3DES (168bit)"
                        $iAlgorithm = $CALG_3DES

                    Case "AES (128bit)"
                        $iAlgorithm = $CALG_AES_128

                    Case "AES (192bit)"
                        $iAlgorithm = $CALG_AES_192

                    Case "AES (256bit)"
                        $iAlgorithm = $CALG_AES_256

                    Case "DES (56bit)"
                        $iAlgorithm = $CALG_DES

                    Case "RC2 (128bit)"
                        $iAlgorithm = $CALG_RC2

                    Case "RC4 (128bit)"
                        $iAlgorithm = $CALG_RC4

                EndSwitch

            Case $idBtn_Decrypt
                $sSourceRead = GUICtrlRead($idInp_Source) ; Прочитать ввод исходного пути.
                $sDestinationRead = GUICtrlRead($idInp_Destination) ; Прочитать ввод пути назначения.
                $sPasswordRead = GUICtrlRead($idInp_Password) ; Прочитать ввод пароля.
                If StringStripWS($sSourceRead, $STR_STRIPALL) <> "" And StringStripWS($sDestinationRead, $STR_STRIPALL) <> "" And StringStripWS($sPasswordRead, $STR_STRIPALL) <> "" And FileExists($sSourceRead) Then ; Проверяем наличие файла для расшифровки и пароля.
                    If _Crypt_DecryptFile($sSourceRead, $sDestinationRead, $sPasswordRead, $iAlgorithm) Then ; Расшифровываем файл.
                        MsgBox($MB_SYSTEMMODAL, "Success", "Operation succeeded.")
                    Else
                        Switch @error
                            Case 2
                                MsgBox($MB_SYSTEMMODAL, "Error", "Couldn't open the source file.")
                            Case 3
                                MsgBox($MB_SYSTEMMODAL, "Error", "Couldn't open the destination file.")
                            Case 30
                                MsgBox($MB_SYSTEMMODAL, "Error", "Failed to create the key.")
                            Case 400 Or 500
                                MsgBox($MB_SYSTEMMODAL, "Error", "Decryption error.")
                            Case Else
                                MsgBox($MB_SYSTEMMODAL, "Error", "Unexpected @error = " & @error)
                        EndSwitch
                    EndIf
                Else
                    MsgBox($MB_SYSTEMMODAL, "Error", "Please ensure the relevant information has been entered correctly.")
                EndIf
        EndSwitch
    WEnd

    GUIDelete($hGUI) ; Удалить предыдущий GUI и все элементы управления.
EndFunc   ;==>Example
