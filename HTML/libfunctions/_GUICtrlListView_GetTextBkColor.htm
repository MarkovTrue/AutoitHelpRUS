Справочник функций

_GUICtrlListView_GetTextBkColor

Получает.

#include <GuiListView.au3>
_GUICtrlListView_GetTextBkColor ( $hWnd )

Параметры

$hWnd Идентификатор или дескриптор элемента управления.

Возвращаемое значение

Возвращает цвет фона текста.

См. также

_GUICtrlListView_SetTextBkColor

Пример

#include <ColorConstants.au3>
#include <GUIConstantsEx.au3>
#include <GuiListView.au3>
#include <MsgBoxConstants.au3>

Example()

Func Example()
    GUICreate("ListView Get/Set Text BkColor (v" & @AutoItVersion & ")", 400, 300)
    Local $idListview = GUICtrlCreateListView("", 2, 2, 394, 268)
    GUISetState(@SW_SHOW)

    ; Задать цвета
    _GUICtrlListView_SetBkColor($idListview, $CLR_DARKSEAGREEN)
    _GUICtrlListView_SetTextColor($idListview, $CLR_BLACK)
    _GUICtrlListView_SetTextBkColor($idListview, $CLR_DARKSEAGREEN)

    ; Добавить столбцы
    _GUICtrlListView_AddColumn($idListview, "Items", 100)

    ; Добавить элементы
    _GUICtrlListView_BeginUpdate($idListview)
    For $iI = 0 To 10
        _GUICtrlListView_AddItem($idListview, "Item " & $iI)
    Next
    _GUICtrlListView_EndUpdate($idListview)

    ; Показать цвета
    MsgBox($MB_SYSTEMMODAL, "Information", "Back Color ....: " & _GUICtrlListView_GetBkColor($idListview) & @CRLF & _
            "Text Color ....: " & _GUICtrlListView_GetTextColor($idListview) & @CRLF & _
            "Text Back Color: " & _GUICtrlListView_GetTextBkColor($idListview))

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
    GUIDelete()
EndFunc   ;==>Example
