Справочник функций

_GDIPlus_FontPrivateAddFont

Добавляет файл шрифта в закрытую коллекцию шрифтов.

#include <GDIPlus.au3>
_GDIPlus_FontPrivateAddFont ( $hFontCollection, $sFontFile )

Параметры

$hFontCollection Дескриптор объекта коллекции шрифтов.
$sFontFile Строка с именем файла шрифта.

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False и устанавливает @error ≠ 0.
@extended: Может содержать код ошибки GPSTATUS ($GDIP_ERR* см. GPIPlusConstants.au3).

См. также

_GDIPlus_FontPrivateCollectionDispose, _GDIPlus_FontFamilyCreateFromCollection, _GDIPlus_FontPrivateCreateCollection, _GDIPlus_FontCreate

Дополнительно

Искать GdipPrivateAddFontFile в библиотеке MSDN

Пример

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>
#include <WinAPIGdi.au3>

Example()

Func Example()
    Local $hGUI, $hGraphic, $hBrush, $hFormat, $hFamily, $hFont, $tLayout, $sFontInternalName, $hCollection, $sFontFile = @WindowsDir & "\Fonts\Times.ttf"

    ; Поддержка работы в X64
    Local $sWow64 = ""
    If @AutoItX64 Then $sWow64 = "\Wow6432Node"

    If _WinAPI_DwmIsCompositionEnabled() Then
        If Not @Compiled Then
            $sFontFile = StringRegExpReplace(@AutoItExe, "(.*)\\.*", "$1")
        Else
            $sFontFile = StringRegExpReplace(RegRead("HKLM\SOFTWARE" & $sWow64 & "\Microsoft\Windows\CurrentVersion\Uninstall\AutoItv3", "DisplayIcon"), "(.*)\\.*", "$1")
        EndIf
        If $sFontFile = "" Then Exit MsgBox(($MB_ICONERROR + $MB_TOPMOST), "Error", "Unable to locate SF Square Head Bold.ttf in AutoIt sub dir!")
        $sFontFile &= "\Examples\Helpfile\Extras\SF Square Head Bold.ttf"
    EndIf

    $sFontInternalName = _WinAPI_GetFontResourceInfo($sFontFile, Default, 1)

    $hGUI = GUICreate("GDI+", 720, 235)
    GUISetState()

    _GDIPlus_Startup()
    $hGraphic = _GDIPlus_GraphicsCreateFromHWND($hGUI)
    _GDIPlus_GraphicsSetSmoothingMode($hGraphic, $GDIP_SMOOTHINGMODE_HIGHQUALITY) ; Установить качество рендеринга graphics-объекта (сглаживание)
    _GDIPlus_GraphicsSetTextRenderingHint($hGraphic, $GDIP_TEXTRENDERINGHINTANTIALIASGRIDFIT) ; установить сглаживание для шрифтов

    $hCollection = _GDIPlus_FontPrivateCreateCollection() ; создать PrivateFontCollection для последующего добавления шрифтов
    _GDIPlus_FontPrivateAddFont($hCollection, $sFontFile) ; добавить шрифт в коллекцию шрифтов
    $hFormat = _GDIPlus_StringFormatCreate() ; создать объект string format
    $hFamily = _GDIPlus_FontFamilyCreateFromCollection($sFontInternalName, $hCollection) ; создать объект font family
    $hFont = _GDIPlus_FontCreate($hFamily, 72) ; создать объект font
    $tLayout = _GDIPlus_RectFCreate(0, 0, 720, 235) ; создать структуру $tagGDIPRECTF для хранения позиции текста (x, y)
    _GDIPlus_StringFormatSetAlign($hFormat, 1) ; центрировать текст
    _GDIPlus_StringFormatSetLineAlign($hFormat, 1) ; центрировать текст
    $hBrush = _GDIPlus_BrushCreateSolid(0xFF00007F) ; создать кисть заданного цвета для текста
    _GDIPlus_GraphicsDrawStringEx($hGraphic, "AutoIt" & @LF & "forever", $hFont, $tLayout, $hFormat, $hBrush) ; нарисовать строку

    ; Цикл до выхода пользователя
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE

    ; Освободить ресурсы
    _GDIPlus_FontDispose($hFont)
    _GDIPlus_FontFamilyDispose($hFamily)
    _GDIPlus_StringFormatDispose($hFormat)
    _GDIPlus_FontPrivateCollectionDispose($hCollection)
    _GDIPlus_BrushDispose($hBrush)
    _GDIPlus_GraphicsDispose($hGraphic)
    _GDIPlus_Shutdown()
EndFunc   ;==>Example
