Справочник функций

_IEDocInsertHTML

Вставляет HTML-текст в элемент или вокруг него.

#include <IE.au3>
_IEDocInsertHTML ( ByRef $oObject, $sString [, $sWhere = "beforeend"] )

Параметры

$oObject Объектная переменная, указывающая на элемент документа.
$sString Строка с HTML-текстом для вставки.
$sWhere [необязательный] задаёт точку вставки строки
"beforebegin" = Вставка строки непосредственно перед объектом.
"afterbegin" = Вставка строки после начала объекта, но перед всем остальным содержимым объекта.
"beforeend" = (по умолчанию) Вставка строки непосредственно перед концом объекта, но после всего остального содержимого объекта.
"afterend" = Вставка строки сразу после конца объекта.

Возвращаемое значение

Успех: Возвращает 1.
Ошибка: Возвращает 0 и устанавливает @error ≠ 0.
@error: 2 ($_IEStatus_COMError) – ошибка COM при обращении к объекту
3 ($_IEStatus_InvalidDataType) – неверный тип данных
4 ($_IEStatus_InvalidObjectType) – неверный тип объекта
5 ($_IEStatus_InvalidValue) – недопустимое значение
@extended: Содержит номер недопустимого параметра.

Примечания

Возможности innerHTML, outerHTML, innerText и outerText из _IEPropertySet() позволяют динамически менять вставленное содержимое.

См. также

_IEBodyReadHTML, _IEBodyWriteHTML, _IEDocInsertText, _IEDocReadHTML, _IEHeadInsertEventScript, _IEPropertyGet, _IEPropertySet

Пример

Пример 1

; Вставить HTML вверху и внизу документа

#include <IE.au3>

Local $oIE = _IECreate("http://www.autoitscript.com")
Local $oBody = _IETagNameGetCollection($oIE, "body", 0)
_IEDocInsertHTML($oBody, "<h2><font color=red>This HTML is inserted After Begin</font></h2>", "afterbegin")
_IEDocInsertHTML($oBody, "<h2><font color=red>This HTML is inserted Before End</font></h2>", "beforeend")

Пример 2

; Открытие браузера со страницей базового примера, вставка HTML
; внутри и вокруг тега DIV с именем "IEAu3Data" и показать HTML тела документа

#include <IE.au3>
#include <MsgBoxConstants.au3>

Local $oIE = _IE_Example("basic")
Local $oDiv = _IEGetObjById($oIE, "IEAu3Data")

_IEDocInsertHTML($oDiv, "<b>(HTML beforebegin)</b>", "beforebegin")
_IEDocInsertHTML($oDiv, "<i>(HTML afterbegin)</i>", "afterbegin")
_IEDocInsertHTML($oDiv, "<b>(HTML beforeend)</b>", "beforeend")
_IEDocInsertHTML($oDiv, "<i>(HTML afterend)</i>", "afterend")

MsgBox($MB_SYSTEMMODAL, "Body", _IEBodyReadHTML($oIE) & @CRLF)

Пример 3

; Продвинутый пример
; Вставить часы и referrer-строку наверху каждой страницы, даже когда вы
; перейти по новому адресу. Используются _IEDocInsertText, _IEDocInsertHTML и
; Возможности _IEPropertySet "innerhtml" и "referrer"

#include <IE.au3>

Global $g_oIE = _IECreate("http://www.autoitscript.com")

AdlibRegister("UpdateClock", 1000) ; Обновлять часы раз в секунду

While 1
    Sleep(10000)
WEnd

Exit

Func UpdateClock()
    ; обновление, пока окно браузера существует
    If Not WinExists(_IEPropertyGet($g_oIE, "hwnd")) Then Exit

    Local $sCurTime = "<font color=red><b>Current Time is: </b>" & @HOUR & ":" & @MIN & ":" & @SEC & "</font>"
    ; _IEGetObjById ожидаемо вернёт ошибку NoMatch после перехода
    ; (до вставки DIV), поэтому временно отключить уведомления
    _IEErrorNotify(False)
    Local $oAutoItClock = _IEGetObjById($g_oIE, "AutoItClock")
    If Not IsObj($oAutoItClock) Then ; Вставить элемент DIV, если он не найден
        ;
        ; Получить ссылку на BODY, вставить DIV, получить ссылку на DIV, обновить время
        Local $oBody = _IETagNameGetCollection($g_oIE, "body", 0)
        _IEDocInsertHTML($oBody, "<div id='AutoItClock'></div>", "afterbegin")
        $oAutoItClock = _IEGetObjById($g_oIE, "AutoItClock")
        _IEPropertySet($oAutoItClock, "innerhtml", $sCurTime)
        ;
        ; Проверить строку referrer – если не пустая, вставить после часов
        _IELoadWait($g_oIE)
        Local $sReferrer = _IEPropertyGet($g_oIE, "referrer")
        If $sReferrer Then _IEDocInsertText($oAutoItClock, _
                " Referred by: <font color=red>" & $sReferrer & "</font>", "afterend")
    Else
        _IEPropertySet($oAutoItClock, "innerhtml", $sCurTime) ; обновить время
    EndIf
    _IEErrorNotify(True)
EndFunc   ;==>UpdateClock
