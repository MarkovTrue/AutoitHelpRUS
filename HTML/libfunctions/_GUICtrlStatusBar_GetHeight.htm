Справочник функций

_GUICtrlStatusBar_GetHeight

Получает высоту.

#include <GuiStatusBar.au3>
_GUICtrlStatusBar_GetHeight ( $hWnd )

Параметры

$hWnd Дескриптор элемента управления.

Возвращаемое значение

Возвращает высоту.

См. также

_GUICtrlStatusBar_GetWidth, _GUICtrlStatusBar_SetMinHeight

Пример

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <GuiStatusBar.au3>

Example()

Func Example()
    ; Создать GUI
    Local $hGUI = GUICreate("StatusBar Get Height/Width (v" & @AutoItVersion & ")", 450, 320, 100, 100)

    Local $hStatus = _GUICtrlStatusBar_Create($hGUI)
    Local $aParts[3] = [75, 150, -1]
    _GUICtrlStatusBar_SetParts($hStatus, $aParts)

    ; Создать элемент memo
    _MemoCreate(2, 8, 444, 259)
    GUISetState(@SW_SHOW)

    ; Получить высоту/ширину частей
    _MemoWrite("Height of parts .: " & _GUICtrlStatusBar_GetHeight($hStatus))
    _MemoWrite("Width of part 0 .: " & _GUICtrlStatusBar_GetWidth($hStatus, 0))
    _MemoWrite("Width of part 1 .: " & _GUICtrlStatusBar_GetWidth($hStatus, 1))
    _MemoWrite("Width of part 2 .: " & _GUICtrlStatusBar_GetWidth($hStatus, 2))

    _MemoMsgBoxStatus("", -1, $hGUI) ; больше действий нет, ждём закрытия GUI

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
    GUIDelete()
EndFunc   ;==>Example
