Справочник функций

_GUICtrlListBox_Create

Создаёт элемент управления ListBox.

#include <GuiListBox.au3>
_GUICtrlListBox_Create ( $hWnd, $sText, $iX, $iY [, $iWidth = 100 [, $iHeight = 200 [, $iStyle = 0x00B00002 [, $iExStyle = 0x00000200]]]] )

Параметры

$hWnd Дескриптор родительского или владеющего окна.
$sText Строка для добавления в ListBox.
$iX Горизонтальная позиция элемента управления.
$iY Вертикальная позиция элемента управления.
$iWidth [необязательный] Ширина элемента управления
$iHeight [необязательный] Высота элемента управления
$iStyle [необязательный] Стиль элемента управления:
$LBS_COMBOBOX – уведомляет список, что он часть combo box
$LBS_DISABLENOSCROLL – показывает отключённую вертикальную полосу прокрутки
$LBS_EXTENDEDSEL – разрешает выделять несколько элементов
$LBS_HASSTRINGS – список содержит элементы-строки
$LBS_MULTICOLUMN – многоколоночный список с горизонтальной прокруткой
$LBS_MULTIPLESEL – включает или выключает выделение строки при каждом щелчке
$LBS_NODATA – список без данных
$LBS_NOINTEGRALHEIGHT – размер точно такой, какой задало приложение
$LBS_NOREDRAW – внешний вид списка не обновляется при изменениях
$LBS_NOSEL – элементы списка можно просматривать, но нельзя выделять
$LBS_NOTIFY – уведомляет при каждом щелчке или двойном щелчке по строке
$LBS_OWNERDRAWFIXED – список отрисовывается владельцем
$LBS_OWNERDRAWVARIABLE – список отрисовывается владельцем, с переменной высотой
$LBS_SORT – сортирует строки списка по алфавиту
$LBS_STANDARD – стандартный стиль списка
$LBS_USETABSTOPS – список распознаёт и раскрывает символы табуляции
$LBS_WANTKEYBOARDINPUT – владелец получает сообщения WM_VKEYTOITEM

По умолчанию: $LBS_SORT, $WS_HSCROLL, $WS_VSCROLL, $WS_BORDER
Принудительно: $WS_CHILD, $WS_TABSTOP, $WS_VISIBLE, $LBS_NOTIFY.
$iExStyle [необязательный] Расширенный стиль элемента. Соответствует стандартным константам $WS_EX_*. См. таблицу расширенных стилей.

Возвращаемое значение

Успех: Возвращает дескриптор элемента управления ListBox.
Ошибка: Возвращает 0.

Примечания

Данная функция предназначена для опытных пользователей и для изучения принципов работы элемента управления.

См. также

_GUICtrlListBox_Destroy

Пример

Example : Created with UDF

; == Пример: создано с помощью UDF

#include "Extras\WM_NOTIFY.au3"

#include <DirConstants.au3>
#include <GUIConstantsEx.au3>
#include <GuiListBox.au3>
#include <MsgBoxConstants.au3>
#include <WindowsNotifsConstants.au3>

Global $g_hListBox

Example()

Func Example()
    ; Создать GUI
    Local $hGUI = GUICreate("ListBox Create (v" & @AutoItVersion & ")", 400, 296)
    $g_hListBox = _GUICtrlListBox_Create($hGUI, "String upon creation", 2, 2, 396, 296)
    GUISetState(@SW_SHOW)

    MsgBox($MB_SYSTEMMODAL, "Information", "Adding Items")

    GUIRegisterMsg($WM_COMMAND, "WM_COMMAND")

    ; Добавить файлы
    _GUICtrlListBox_BeginUpdate($g_hListBox)
    _GUICtrlListBox_ResetContent($g_hListBox)
    _GUICtrlListBox_InitStorage($g_hListBox, 100, 4096)
    _GUICtrlListBox_Dir($g_hListBox, @WindowsDir & "\win*.exe")
    _GUICtrlListBox_AddFile($g_hListBox, @WindowsDir & "\notepad.exe")
    _GUICtrlListBox_Dir($g_hListBox, "", $DDL_DRIVES)
    _GUICtrlListBox_Dir($g_hListBox, "", $DDL_DRIVES, False)
    _GUICtrlListBox_EndUpdate($g_hListBox)

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
EndFunc   ;==>Example

Func WM_COMMAND($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg
    Local $hWndListBox = $g_hListBox
    If Not IsHWnd($g_hListBox) Then $hWndListBox = GUICtrlGetHandle($g_hListBox)
    Local $hWndFrom = $lParam
    Local $iIDFrom = BitAND($wParam, 0xFFFF) ; Младшее слово
    Local $iCode = BitShift($wParam, 16) ; Старшее слово

    Switch $hWndFrom
        Case $g_hListBox, $hWndListBox
            Switch $iCode
                Case $LBN_DBLCLK ; Отправляется, когда пользователь дважды щёлкает по строке в ListBox
                    _WM_NOTIFY_DebugInfo("$LBN_DBLCLK", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $LBN_ERRSPACE ; Отправляется, когда ListBox не может выделить достаточно памяти для выполнения конкретного запроса
                    _WM_NOTIFY_DebugInfo("$LBN_ERRSPACE", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $LBN_KILLFOCUS ; Отправляется, когда ListBox теряет фокус клавиатуры
                    _WM_NOTIFY_DebugInfo("$LBN_KILLFOCUS", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $LBN_SELCANCEL ; Отправляется, когда пользователь отменяет выделение в ListBox
                    _WM_NOTIFY_DebugInfo("$LBN_SELCANCEL", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $LBN_SELCHANGE ; Отправляется, когда выделение в ListBox изменилось
                    _WM_NOTIFY_DebugInfo("$LBN_SELCHANGE", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $LBN_SETFOCUS ; Отправляется, когда ListBox получает фокус клавиатуры
                    _WM_NOTIFY_DebugInfo("$LBN_SETFOCUS", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
            EndSwitch
    EndSwitch
    ; Продолжить стандартные внутренние команды сообщений AutoIt3.
    ; Эту строку также можно вообще пропустить.
    ; !!! Но только 'Return' (без значения) не продолжит
    ; стандартную обработку сообщений AutoIt3 в дальнейшем !!!
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_COMMAND
