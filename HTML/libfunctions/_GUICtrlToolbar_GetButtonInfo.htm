Справочник функций

_GUICtrlToolbar_GetButtonInfo

Возвращает информацию о кнопке.

#include <GuiToolbar.au3>
_GUICtrlToolbar_GetButtonInfo ( $hWnd, $iCommandID )

Параметры

$hWnd Дескриптор элемента управления.
$iCommandID Идентификатор команды кнопки.

Возвращаемое значение

Возвращает массив следующего формата:
    [0] – индекс изображения кнопки (нумерация с 0)
    [1] – состояние кнопки, сочетание следующих значений:
        $TBSTATE_CHECKED – кнопка отмечена
        $TBSTATE_PRESSED – кнопка нажата
        $TBSTATE_ENABLED – кнопка принимает ввод от пользователя
        $TBSTATE_HIDDEN – кнопка не отображается
        $TBSTATE_INDETERMINATE – кнопка в неопределённом (сером) состоянии
        $TBSTATE_WRAP – после кнопки идёт перенос строки
        $TBSTATE_ELLIPSES – текст кнопки обрезан
        $TBSTATE_MARKED – кнопка помечена
    [2] – стиль кнопки, сочетание следующих значений:
        $BTNS_AUTOSIZE – элементу не следует задавать стандартную ширину
        $BTNS_BUTTON – обычная кнопка
        $BTNS_CHECK – переключается между нажатым и отжатым состояниями
        $BTNS_CHECKGROUP – кнопка остаётся нажатой, пока не нажата другая
        $BTNS_DROPDOWN – кнопка-выпадающий список
        $BTNS_GROUP – кнопка остаётся нажатой, пока не нажата другая
        $BTNS_NOPREFIX – без префикса акселератора в тексте
        $BTNS_SEP – разделитель
        $BTNS_SHOWTEXT – отображать текст кнопки
        $BTNS_WHOLEDROPDOWN – кнопка с раскрывающейся стрелкой
    [3] – ширина кнопки
    [4] – параметр кнопки

Пример

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <GuiToolbar.au3>
#include <WinAPIConstants.au3>
#include <WindowsStylesConstants.au3>

Example()

Func Example()
    Local $hGUI, $hToolbar, $aButton
    Local Enum $e_idNew = 1000, $e_idOpen, $e_idSave, $e_idHelp

    ; Создать GUI
    $hGUI = GUICreate("Toolbar", 400, 300)
    $hToolbar = _GUICtrlToolbar_Create($hGUI)
    _MemoCreate(2, 36, 396, 262, $WS_VSCROLL)
    GUISetState(@SW_SHOW)

    ; Добавить стандартные системные изображения
    Switch _GUICtrlToolbar_GetBitmapFlags($hToolbar)
        Case 0
            _GUICtrlToolbar_AddBitmap($hToolbar, 1, -1, $IDB_STD_SMALL_COLOR)
        Case 2
            _GUICtrlToolbar_AddBitmap($hToolbar, 1, -1, $IDB_STD_LARGE_COLOR)
    EndSwitch

    ; Добавить кнопки
    _GUICtrlToolbar_AddButton($hToolbar, $e_idNew, $STD_FILENEW)
    _GUICtrlToolbar_AddButton($hToolbar, $e_idOpen, $STD_FILEOPEN)
    _GUICtrlToolbar_AddButton($hToolbar, $e_idSave, $STD_FILESAVE)
    _GUICtrlToolbar_AddButtonSep($hToolbar)
    _GUICtrlToolbar_AddButton($hToolbar, $e_idHelp, $STD_HELP)

    ; Установить сведения о кнопке Save
    _GUICtrlToolbar_SetButtonInfo($hToolbar, $e_idSave, $STD_PRINT, BitOR($TBSTATE_PRESSED, $TBSTATE_ENABLED), -1, 100, 1234)

    ; Показать сведения о кнопке Save
    $aButton = _GUICtrlToolbar_GetButtonInfo($hToolbar, $e_idSave)
    _MemoWrite("Image index ....: " & $aButton[0])
    _MemoWrite("State flags ....: " & $aButton[1])
    _MemoWrite("Style flags ....: " & $aButton[2])
    _MemoWrite("Button width ...: " & $aButton[3])
    _MemoWrite("Param ..........: " & $aButton[4])

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
EndFunc   ;==>Example
