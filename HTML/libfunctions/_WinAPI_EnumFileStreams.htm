_WinAPI_EnumFileStreams

Перечисляет все потоки типа ::$DATA в указанном файле или каталоге.

#include <WinAPIFiles.au3>
_WinAPI_EnumFileStreams ( $sFilePath )

Параметры

$sFilePath Путь к файлу или каталогу. Файл должен находиться на томе с файловой системой NTFS.

Возвращаемое значение

Успех: Возвращает двумерный массив имён потоков и их размеров.
$aArray[0][0] – количество строк массива (n)
$aArray[0][1] – не используется
[n][0] – имя потока.
[n][1] – размер потока в байтах.
Ошибка: Устанавливает @error ≠ 0.
@extended: Может содержать код ошибки NTSTATUS.

Дополнительно

Искать ZwQueryInformationFile в библиотеке MSDN

Пример

#include <Array.au3>
#include <MsgBoxConstants.au3>
#include <WinAPIFiles.au3>
#include <WinAPIHObj.au3>
#include <WinAPIMem.au3>
#include <WinAPIMisc.au3>
#include <WinAPIShPath.au3>

_Example()

Func _Example()
    Local Const $sFile = @TempDir & '\Test.txt'

    ; Проверить файловую систему NTFS
    If StringCompare(DriveGetFileSystem(_WinAPI_PathStripToRoot($sFile)), 'NTFS') Then
        MsgBox(($MB_ICONERROR + $MB_SYSTEMMODAL), 'Error', 'The file must be on an NTFS file system volume.')
        Exit
    EndIf

    Local $hFile, $pData, $sName, $iBytes
    ; Создать текстовый файл с тремя альтернативными потоками AS1, AS2 и AS3
    For $i = 0 To 3
        If $i Then
            $pData = _WinAPI_CreateString('Alternative stream ' & $i)
            $sName = ':AS' & $i
        Else
            $pData = _WinAPI_CreateString('Main stream')
            $sName = ''
        EndIf
        $hFile = _WinAPI_CreateFile($sFile & $sName, 1, 4)
        _WinAPI_WriteFile($hFile, $pData, _WinAPI_GetMemorySize($pData) - 2, $iBytes)
        _WinAPI_CloseHandle($hFile)
        _WinAPI_FreeMemory($pData)
    Next

    ; Перечислить все существующие потоки файла
    Local $aData = _WinAPI_EnumFileStreams($sFile)

    _ArrayDisplay($aData, '_WinAPI_EnumFileStreams')

    ; Прочитать текстовые данные из каждого потока
    $pData = _WinAPI_CreateBuffer(1024)
    For $i = 1 To $aData[0][0]
        $hFile = _WinAPI_CreateFile($sFile & $aData[$i][0], 2, 2, 6)
        _WinAPI_ReadFile($hFile, $pData, $aData[$i][1], $iBytes)
        _WinAPI_CloseHandle($hFile)
        $aData[$i][1] = _WinAPI_GetString($pData)
    Next
    _WinAPI_FreeMemory($pData)

    _ArrayDisplay($aData, '_WinAPI_EnumFileStreams')

    FileDelete($sFile)
EndFunc   ;==>_Example
