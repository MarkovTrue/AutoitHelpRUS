Справочник функций

_GUICtrlTab_Create

Создаёт элемент управления Tab.

#include <GuiTab.au3>
_GUICtrlTab_Create ( $hWnd, $iX, $iY [, $iWidth = 150 [, $iHeight = 150 [, $iStyle = 0x00000040 [, $iExStyle = 0x00000000]]]] )

Параметры

$hWnd Дескриптор родительского или владеющего окна.
$iX Горизонтальная позиция элемента управления.
$iY Вертикальная позиция элемента управления.
$iWidth [необязательный] Ширина элемента управления
$iHeight [необязательный] Высота элемента управления
$iStyle [необязательный] Ширина элемента управления.
$iExStyle [необязательный] Расширенный стиль элемента управления. Они соответствуют стандартным константам $WS_EX_*.

Возвращаемое значение

Успех: Дескриптор элемента управления.
Ошибка: Возвращает 0.
@error: Возвращает -1, если $hWnd некорректен.
2 – окно $hWnd не существует.
4 – окно, на которое ссылается $hWnd, принадлежит другому процессу.
8 – определено слишком много элементов управления.
16 – не удалось создать элемент управления.

Примечания

Данная функция предназначена для опытных пользователей и для изучения принципов работы элемента управления.

См. также

_GUICtrlTab_Destroy

Пример

Пример 1

; =; == Пример 1: создание через UDF

#include "Extras\HelpFileInternals.au3"
#include "Extras\WM_NOTIFY.au3"

#include <GUIConstantsEx.au3>
#include <GuiTab.au3>
#include <WindowsStylesConstants.au3>

Global $g_hTab

Example()

Func Example()
    ; Создать GUI
    Local $hGUI = GUICreate("Tab Create (v" & @AutoItVersion & ")", 450, 300, 100, 100)
    $g_hTab = _GUICtrlTab_Create($hGUI, 2, 2, 446, 266)
    _MemoMsgBoxStatus() ; Создание статус-бара

    GUISetState(@SW_SHOW)

    _WM_NOTIFY_Register()

    _MemoMsgBox($MB_SYSTEMMODAL, "Information", "Create 3 Tabs")
    ; Добавить вкладки
    _GUICtrlTab_InsertItem($g_hTab, 0, "Tab 0")
    _GUICtrlTab_InsertItem($g_hTab, 1, "Tab 1")
    _GUICtrlTab_InsertItem($g_hTab, 2, "Tab 2")

    _MemoMsgBoxStatus("", -1, $hGUI) ; больше действий нет, ждём закрытия GUI

EndFunc   ;==>Example

Func WM_NOTIFY($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $wParam
    Local $hWndTab = $g_hTab
    If Not IsHWnd($g_hTab) Then $hWndTab = GUICtrlGetHandle($g_hTab)

    Local $tNMHDR = DllStructCreate($tagNMHDR, $lParam)
    Local $hWndFrom = HWnd(DllStructGetData($tNMHDR, "hWndFrom"))
    Local $iCode = DllStructGetData($tNMHDR, "Code")
    Switch $hWndFrom
        Case $hWndTab
            Switch $iCode
                Case $NM_CLICK ; Пользователь кликнул левой кнопкой мыши в элементе управления
                    _WM_NOTIFY_DebugEvent("$NM_CLICK", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Возвращаемое значение элементом вкладок игнорируется
                Case $NM_DBLCLK ; Пользователь дважды кликнул левой кнопкой мыши в элементе управления
                    _WM_NOTIFY_DebugEvent("$NM_DBLCLK", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Return 1 ; ненулевое – запретить стандартную обработку
                    Return 0 ; ноль – разрешить стандартную обработку
                Case $NM_RCLICK ; Пользователь кликнул правой кнопкой мыши в элементе управления
                    _WM_NOTIFY_DebugEvent("$NM_RCLICK", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Return 1 ; ненулевое – запретить стандартную обработку
                    Return 0 ; ноль – разрешить стандартную обработку
                Case $NM_RDBLCLK ; Пользователь дважды кликнул правой кнопкой мыши в элементе управления
                    _WM_NOTIFY_DebugEvent("$NM_RDBLCLK", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Return 1 ; ненулевое – запретить стандартную обработку
                    Return 0 ; ноль – разрешить стандартную обработку
                Case $NM_RELEASEDCAPTURE ; элемент освобождает захват мыши
                    _WM_NOTIFY_DebugEvent("$NM_RELEASEDCAPTURE", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Нет возвращаемого значения
            EndSwitch
    EndSwitch
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_NOTIFY

Example (OutProcess) : Tab Create to an External process

; == Example (OutProcess) : Tab Create to an External process

#include "Extras\HelpFileInternals.au3"

#include <GuiTab.au3>
#include <MsgBoxConstants.au3>

Example()

Func Example()
    Local $sFromTo
    Local $hWin = _MemoRunAU3OutProcess($sFromTo, False)
    _MemoCreateOutProcess($hWin, "SysTabControl32", 2, $sFromTo)

    _GUICtrlTab_Create($hWin, 2, 2, 446, 266)

    If @error Then _MemoMsgBox($MB_ICONERROR, "Info" & $sFromTo, "@error=" & @error & " _GUICtrlTab_Create()" & @CRLF & @TAB & "cannot be used for an external process")

    _MemoMsgBoxStatus("", -1, $hWin) ; больше действий нет, ждём закрытия GUI

EndFunc   ;==>Example
