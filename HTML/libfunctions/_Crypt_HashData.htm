_Crypt_HashData

Вычисляет хеш данных с помощью указанного алгоритма.

#include <Crypt.au3>
_Crypt_HashData ( $vData, $iAlgID [, $bFinal = True [, $hCryptHash = 0]] )

Параметры

$vData Данные для хеширования.
$iAlgID Идентификатор алгоритма хеширования. См. _Crypt_DeriveKey().
$bFinal [необязательный] False, если передаётся только фрагмент данных; в этом случае функция возвращает объект хеша, а не итоговый хеш.
$hCryptHash [необязательный] Объект хеша, возвращённый предыдущим вызовом _Crypt_HashData().

Возвращаемое значение

Успех: Возвращает двоичный хеш или объект хеша, если $bFinal = False.
Ошибка: Устанавливает @error ≠ 0.
@error: 10 – не удалось создать объект хеша
20 – ошибка хеширования данных
30 – не удалось получить размер хеша
40 – не удалось получить хеш
1000+ – ошибка _Crypt_Startup()

Примечания

Хеш-дайджест возвращается в виде двоичной строки; размер определяется алгоритмом.
Для обработки данных по фрагментам устанавливайте $bFinal в False для всех промежуточных фрагментов и передавайте возвращённый объект хеша во все последующие вызовы.

Пример

#include <ComboConstants.au3>
#include <Crypt.au3>
#include <EditConstants.au3>
#include <GUIConstantsEx.au3>
#include <StringConstants.au3>
#include <WinAPIConv.au3>
#include <WindowsNotifsConstants.au3>

Global $g_iAlgorithm = $CALG_SHA1, $g_idEdt_Input = -1, $g_idEdt_Output = -1

Example()

Func Example()
    Local $hGUI = GUICreate("Realtime Hashing", 400, 320)
    $g_idEdt_Input = GUICtrlCreateEdit("", 0, 0, 400, 150, $ES_WANTRETURN)
    $g_idEdt_Output = GUICtrlCreateEdit("", 0, 150, 400, 150, $ES_READONLY)
    Local $idCombo = GUICtrlCreateCombo("", 0, 300, 100, 20, $CBS_DROPDOWNLIST)
    GUICtrlSetData($idCombo, "MD2 (128bit)|MD4 (128bit)|MD5 (128bit)|SHA1 (160bit)|SHA_256|SHA_384|SHA_512", "SHA1 (160bit)")
    GUIRegisterMsg($WM_COMMAND, "WM_COMMAND")
    GUISetState(@SW_SHOW, $hGUI)

    _Crypt_Startup() ; Для оптимизации производительности запустить библиотеку Crypt.

    Local $dHash = 0, _
            $sRead = ""
    While 1
        Switch GUIGetMsg()
            Case $GUI_EVENT_CLOSE
                ExitLoop

            Case $idCombo ; Проверить выбор в combobox и получить соответствующий алгоритм.
                Switch GUICtrlRead($idCombo) ; Прочитать выбор в combobox.
                    Case "MD2 (128bit)"
                        $g_iAlgorithm = $CALG_MD2

                    Case "MD4 (128bit)"
                        $g_iAlgorithm = $CALG_MD4

                    Case "MD5 (128bit)"
                        $g_iAlgorithm = $CALG_MD5

                    Case "SHA1 (160bit)"
                        $g_iAlgorithm = $CALG_SHA1

                    Case "SHA_256"
                        $g_iAlgorithm = $CALG_SHA_256

                    Case "SHA_384"
                        $g_iAlgorithm = $CALG_SHA_384

                    Case "SHA_512"
                        $g_iAlgorithm = $CALG_SHA_512

                EndSwitch

                $sRead = GUICtrlRead($g_idEdt_Input)
                If StringStripWS($sRead, $STR_STRIPALL) <> "" Then ; Проверяем наличие текста для хеширования.
                    $dHash = _Crypt_HashData($sRead, $g_iAlgorithm) ; Создать хэш введённого текста.
                    GUICtrlSetData($g_idEdt_Output, $dHash) ; Заполнить поле вывода данными хэша.
                EndIf
        EndSwitch
    WEnd

    GUIDelete($hGUI) ; Удалить предыдущий GUI и все элементы управления.
    _Crypt_Shutdown() ; Завершить работу библиотеки Crypt.
EndFunc   ;==>Example

Func WM_COMMAND($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $lParam

    Switch _WinAPI_LoWord($wParam)
        Case $g_idEdt_Input
            Switch _WinAPI_HiWord($wParam)
                Case $EN_CHANGE
                    Local $dHash = _Crypt_HashData(GUICtrlRead($g_idEdt_Input), $g_iAlgorithm) ; Создать хэш введённого текста.
                    GUICtrlSetData($g_idEdt_Output, $dHash) ; Заполнить поле вывода данными хэша.
            EndSwitch
    EndSwitch
EndFunc   ;==>WM_COMMAND
