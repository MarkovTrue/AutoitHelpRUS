Справочник функций

_GUIToolTip_Deactivate

Деактивирует элемент управления ToolTip.

#include <GuiToolTip.au3>
_GUIToolTip_Deactivate ( $hTool )

Параметры

$hTool Дескриптор элемента управления ToolTip (возвращается функцией _GUIToolTip_Create).

Возвращаемое значение

Отсутствует.

См. также

_GUIToolTip_Activate

Пример

#include <GUIConstantsEx.au3>
#include <GUIToolTip.au3>

Example()

Func Example()
    Local $bActivate = True

    Local $hGUI = GUICreate(StringTrimRight(@ScriptName, StringLen(".exe")), 350, 200)

    Local $idBtn_ToggleTips = GUICtrlCreateButton("Tips: On", 30, 32, 180, 28)
    Local $hToggleTips = GUICtrlGetHandle($idBtn_ToggleTips)
    ; создать элемент управления ToolTip в стиле пузыря
    Local $hToolTip = _GUIToolTip_Create(0, $TTS_BALLOON)

    ; добавить инструмент к элементу управления ToolTip с параметрами по умолчанию.
    _GUIToolTip_AddTool($hToolTip, 0, "Tooltip for the GUI", $hGUI)
    ; добавить инструмент к элементу управления ToolTip, центрируя подсказку под кнопкой вместо отображения над указателем мыши
    _GUIToolTip_AddTool($hToolTip, 0, "This button toggles the tooltips on and off", $hToggleTips, 0, 0, 0, 0, BitOR($TTF_CENTERTIP, $TTF_SUBCLASS, $TTF_IDISHWND))
    GUISetState(@SW_SHOW)

    While 1
        Switch GUIGetMsg()
            Case $idBtn_ToggleTips
                $bActivate = Not $bActivate
                If $bActivate Then
                    _GUIToolTip_Activate($hToolTip)
                    GUICtrlSetData($idBtn_ToggleTips, 'Tips: On')
                Else
                    _GUIToolTip_Deactivate($hToolTip)
                    GUICtrlSetData($idBtn_ToggleTips, 'Tips: Off')
                EndIf
            Case $GUI_EVENT_CLOSE
                ExitLoop
        EndSwitch
    WEnd
    ; Уничтожить элемент tooltip
    _GUIToolTip_Destroy($hToolTip)
    GUIDelete($hGUI)
EndFunc   ;==>Example
