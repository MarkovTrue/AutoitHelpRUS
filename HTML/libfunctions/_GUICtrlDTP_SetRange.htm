Справочник функций

_GUICtrlDTP_SetRange

Задаёт текущие минимально и максимально допустимое системное время.

#include <GuiDateTimePicker.au3>
_GUICtrlDTP_SetRange ( $hWnd, ByRef $aRange )

Параметры

$hWnd Дескриптор элемента управления.
$aRange Массив следующего формата:
[ 0] – True, если задаётся минимум, иначе False
[ 1] – минимальный год
[ 2] – минимальный месяц
[ 3] – минимальный день
[ 4] – минимальный час
[ 5] – минимальная минута
[ 6] – минимальная секунда
[ 7] – True, если задаётся максимум, иначе False
[ 8] – максимальный год
[ 9] – максимальный месяц
[10] – максимальный день
[11] – максимальный час
[12] – максимальная минута
[13] – максимальная секунда

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False.

См. также

_GUICtrlDTP_GetRange

Пример

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <GuiDateTimePicker.au3>

Global $g_aRange[14] = [True, @YEAR, 1, 1, 21, 45, 32, True, @YEAR, 12, 31, 23, 59, 59]

Example()

Func Example()
    ; Создать GUI
    GUICreate("DateTimePick Get/Set Range (v" & @AutoItVersion & ")", 400, 300)
    Local $hDTP = GUICtrlGetHandle(GUICtrlCreateDate("", 2, 6, 190))
    _MemoCreate(2, 32, 396, 266, 0)
    GUISetState(@SW_SHOW)

    ; Установить формат отображения
    _GUICtrlDTP_SetFormat($hDTP, "ddd MMM dd, yyyy hh:mm ttt")

    ; Установить диапазон дат
    _GUICtrlDTP_SetRange($hDTP, $g_aRange)

    ; Показать диапазон дат
    $g_aRange = _GUICtrlDTP_GetRange($hDTP)
    _MemoWrite("Minimum date: " & GetDateStr(0))
    _MemoWrite("Maximum date: " & GetDateStr(7))
    _MemoWrite("Minimum time: " & GetTimeStr(4))
    _MemoWrite("Maximum time: " & GetTimeStr(11))

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
    GUIDelete()
EndFunc   ;==>Example

; Возвращает часть даты
Func GetDateStr($iOff = 0)
    Return StringFormat("%02d/%02d/%04d", $g_aRange[$iOff + 2], $g_aRange[$iOff + 3], $g_aRange[$iOff + 1])
EndFunc   ;==>GetDateStr

; Возвращает часть времени
Func GetTimeStr($iOff = 0)
    Return StringFormat("%02d:%02d:%02d", $g_aRange[$iOff], $g_aRange[$iOff + 1], $g_aRange[$iOff + 2])
EndFunc   ;==>GetTimeStr
