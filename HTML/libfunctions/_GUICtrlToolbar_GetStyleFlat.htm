Справочник функций

_GUICtrlToolbar_GetStyleFlat

Указывает, что элемент управления плоский.

#include <GuiToolbar.au3>
_GUICtrlToolbar_GetStyleFlat ( $hWnd )

Параметры

$hWnd Дескриптор элемента управления.

Возвращаемое значение

True: Возвращает признак наличия стиля.
False: Возвращает признак отсутствия стиля.

См. также

_GUICtrlToolbar_SetStyleFlat

Пример

#include <GUIConstantsEx.au3>
#include <GuiToolbar.au3>
#include <MsgBoxConstants.au3>
#include <WinAPIConstants.au3>

Example()

Func Example()
    Local $hGUI, $hToolbar, $bStyle
    Local Enum $e_idNew = 1000, $e_idOpen, $e_idSave, $e_idHelp

    ; Создать GUI
    $hGUI = GUICreate("Toolbar", 400, 300)
    $hToolbar = _GUICtrlToolbar_Create($hGUI)
    GUISetState(@SW_SHOW)

    ; Добавить стандартные системные изображения
    Switch _GUICtrlToolbar_GetBitmapFlags($hToolbar)
        Case 0
            _GUICtrlToolbar_AddBitmap($hToolbar, 1, -1, $IDB_STD_SMALL_COLOR)
        Case 2
            _GUICtrlToolbar_AddBitmap($hToolbar, 1, -1, $IDB_STD_LARGE_COLOR)
    EndSwitch

    ; Добавить кнопки
    _GUICtrlToolbar_AddButton($hToolbar, $e_idNew, $STD_FILENEW)
    _GUICtrlToolbar_AddButton($hToolbar, $e_idOpen, $STD_FILEOPEN)
    _GUICtrlToolbar_AddButton($hToolbar, $e_idSave, $STD_FILESAVE)
    _GUICtrlToolbar_AddButtonSep($hToolbar)
    _GUICtrlToolbar_AddButton($hToolbar, $e_idHelp, $STD_HELP)

    ; Изменить стиль элемента управления
    $bStyle = _GUICtrlToolbar_GetStyleFlat($hToolbar)
    MsgBox($MB_SYSTEMMODAL, "Information", "Toolbar has flat style .: " & $bStyle)
    GUISetState(@SW_LOCK)
    _GUICtrlToolbar_SetStyleFlat($hToolbar, Not $bStyle)
    GUISetState(@SW_UNLOCK)
    MsgBox($MB_SYSTEMMODAL, "Information", "Toolbar has flat style .: " & _GUICtrlToolbar_GetStyleFlat($hToolbar))

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
EndFunc   ;==>Example
