Справочник функций

_GUICtrlListView_SetImageList

Назначает элементу управления список изображений.

#include <GuiListView.au3>
_GUICtrlListView_SetImageList ( $hWnd, $hHandle [, $iType = 0] )

Параметры

$hWnd Идентификатор или дескриптор элемента управления.
$hHandle Дескриптор назначаемого списка изображений.
$iType [необязательный] Тип списка изображений:
0 – список с большими значками
1 – список с маленькими значками
2 – список с изображениями состояний

Возвращаемое значение

Успех: Возвращает дескриптор предыдущего списка изображений.
Ошибка: Возвращает 0.

Примечания

Текущий список изображений будет уничтожен вместе с элементом управления, если не задан стиль $LVS_SHAREIMAGELISTS.
Если через это сообщение один список изображений заменяется на другой, ваше приложение должно явно уничтожить все списки, кроме текущего.

См. также

_GUICtrlListView_GetImageList

Пример

#include <GUIConstantsEx.au3>
#include <GuiImageList.au3>
#include <GuiListView.au3>
#include <MsgBoxConstants.au3>

Example()

Func Example()
    GUICreate("ListView Get/Set Image List (v" & @AutoItVersion & ")", 400, 300)
    Local $idListview = GUICtrlCreateListView("", 2, 2, 394, 268)
    GUISetState(@SW_SHOW)

    ; Загрузить изображения
    Local $hImage = _GUIImageList_Create()
    _GUIImageList_Add($hImage, _GUICtrlListView_CreateSolidBitMap($idListview, 0xFF0000, 16, 16))
    _GUIImageList_Add($hImage, _GUICtrlListView_CreateSolidBitMap($idListview, 0x00FF00, 16, 16))
    _GUIImageList_Add($hImage, _GUICtrlListView_CreateSolidBitMap($idListview, 0x0000FF, 16, 16))
    Local $hPrevImageList = _GUICtrlListView_SetImageList($idListview, $hImage, 1)

    MsgBox($MB_SYSTEMMODAL, "Information", "Previous Image List Handle: 0x" & Hex($hPrevImageList) & @CRLF & _
            "IsPtr = " & IsPtr($hPrevImageList) & " IsHWnd = " & IsHWnd($hPrevImageList))

    ; Добавить столбцы
    _GUICtrlListView_AddColumn($idListview, "Column 0", 100)
    _GUICtrlListView_AddColumn($idListview, "Column 1", 100)
    _GUICtrlListView_AddColumn($idListview, "Column 2", 100)

    ; Добавить элементы
    _GUICtrlListView_AddItem($idListview, "Item 0", 0)
    _GUICtrlListView_AddItem($idListview, "Item 1", 1)
    _GUICtrlListView_AddItem($idListview, "Item 2", 2)

    ; Получить дескриптор image list
    MsgBox($MB_SYSTEMMODAL, "Information", "Image List Handle: 0x" & Hex(_GUICtrlListView_GetImageList($idListview, 1)))

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
    GUIDelete()
EndFunc   ;==>Example
