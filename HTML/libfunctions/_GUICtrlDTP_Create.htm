Справочник функций

_GUICtrlDTP_Create

Создаёт выбор даты и времени (DTP).

#include <GuiDateTimePicker.au3>
_GUICtrlDTP_Create ( $hWnd, $iX, $iY [, $iWidth = 120 [, $iHeight = 21 [, $iStyle = 0x00000000 [, $iExStyle = 0x00000000]]]] )

Параметры

$hWnd Дескриптор родительского или владеющего окна.
$iX Горизонтальная позиция элемента управления.
$iY Вертикальная позиция элемента управления.
$iWidth [необязательный] Ширина элемента управления
$iHeight [необязательный] Высота элемента управления
$iStyle [необязательный] Стили элемента управления:
$DTS_APPCANPARSE – разрешает владельцу разбирать введённые пользователем данные и принимать меры
$DTS_LONGDATEFORMAT – отображает дату в длинном формате
$DTS_RIGHTALIGN – календарь будет выровнен по правому краю
$DTS_SHOWNONE – отображает флажок, который можно установить после ввода даты
$DTS_SHORTDATEFORMAT – отображает дату в коротком формате
$DTS_SHORTDATECENTURYFORMAT – год является полем из четырёх цифр
$DTS_TIMEFORMAT – отображает время
$DTS_UPDOWN – размещает элемент управления вверх-вниз справа от элемента управления

Обязательные: $WS_CHILD, $WS_VISIBLE.
$iExStyle [необязательный] Расширенный стиль элемента. Соответствует стандартным константам $WS_EX_*. См. таблицу расширенных стилей.

Возвращаемое значение

Успех: Возвращает дескриптор элемента управления DTP.
Ошибка: Возвращает 0.

Примечания

Данная функция предназначена для опытных пользователей и для изучения принципов работы элемента управления.

См. также

_GUICtrlDTP_Destroy

Пример

#include "Extras\WM_NOTIFY.au3"

#include <GUIConstantsEx.au3>
#include <GuiDateTimePicker.au3>
#include <StructureConstants.au3>

Global $g_hDTP

Example()

Func Example()
    ; Создать GUI
    Local $hGUI = GUICreate("DateTimePick Create (v" & @AutoItVersion & ")", 400, 300)
    $g_hDTP = _GUICtrlDTP_Create($hGUI, 2, 6, 190)
    GUISetState(@SW_SHOW)

    _WM_NOTIFY_Register()

    ; Установить формат отображения
    _GUICtrlDTP_SetFormat($g_hDTP, "ddd MMM dd, yyyy hh:mm ttt")

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
    GUIDelete()
EndFunc   ;==>Example

Func WM_NOTIFY($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $wParam

    Local $tNMHDR = DllStructCreate($tagNMHDR, $lParam)
    Local $hWndFrom = HWnd(DllStructGetData($tNMHDR, "hWndFrom"))
    Local $iCode = DllStructGetData($tNMHDR, "Code")
    Switch $hWndFrom
        Case $g_hDTP
            Switch $iCode
                Case $DTN_CLOSEUP ; Отправляется элементом управления выбор даты и времени (DTP) при закрытии пользователем раскрывающегося месячного календаря
                    _WM_NOTIFY_DebugEvent("$DTN_CLOSEUP", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Возвращаемое значение этого уведомления не используется
                Case $DTN_DATETIMECHANGE ; Отправляется элементом управления выбор даты и времени (DTP) при каждом изменении
                    _WM_NOTIFY_DebugEvent("$DTN_DATETIMECHANGE", $tagNMDATETIMECHANGE, $lParam, "IDFrom,,Flag,Year,Month,DOW,Day,,Hour,Minute,Second,MSecond")
                    Return 0
                Case $DTN_DROPDOWN ; Отправляется элементом управления выбор даты и времени (DTP) при активации пользователем раскрывающегося месячного календаря
                    _WM_NOTIFY_DebugEvent("$DTN_DROPDOWN", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Возвращаемое значение этого уведомления не используется
                Case $DTN_FORMAT, $DTN_FORMATW ; Отправляется элементом управления выбор даты и времени (DTP) для запроса текста, который будет отображаться в поле обратного вызова
                    _WM_NOTIFY_DebugEvent("$DTN_FORMAT", $tagNMDATETIMEFORMAT, $lParam, "IDFrom,,*Format,Year,Month,DOW,Day,,Hour,Minute,Second,MSecond,*pDisplay")
                    Return 0
                Case $DTN_FORMATQUERY, $DTN_FORMATQUERYW ; Отправляется элементом управления выбор даты и времени (DTP) для получения максимально допустимого размера строки, которая будет отображаться в поле обратного вызова
                    _WM_NOTIFY_DebugEvent("$DTN_FORMATQUERY", $tagNMDATETIMEFORMATQUERY, $lParam, "IDFrom,,*Format,SizeX,SizeY")
                    Local $tInfo = DllStructCreate($tagNMDATETIMEFORMATQUERY, $lParam)
                    DllStructSetData($tInfo, "SizeX", 64)
                    DllStructSetData($tInfo, "SizeY", 10)
                    Return 0
                Case $DTN_USERSTRING, $DTN_USERSTRINGW ; Отправляется элементом управления выбор даты и времени (DTP) при завершении пользователем редактирования строки в элементе управления
                    _WM_NOTIFY_DebugEvent("$DTN_USERSTRING", $tagNMDATETIMESTRING, $lParam, "IDFrom,,*UserString,Year,Month,DOW,Day,,Hour,Minute,Second,MSecond,Flags")
                    Return 0
                Case $DTN_WMKEYDOWN, $DTN_WMKEYDOWNW ; Отправляется элементом управления выбор даты и времени (DTP) при вводе пользователем текста в поле обратного вызова
                    _WM_NOTIFY_DebugEvent("$DTN_WMKEYDOWN", $tagNMDATETIMEFORMATQUERY, $lParam, "IDFrom,,VirtKey,*Format,Year,Month,DOW,Day,,Hour,Minute,Second,MSecond")
                    Return 0
            EndSwitch
    EndSwitch
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_NOTIFY
