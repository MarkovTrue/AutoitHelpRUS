_GDIPlus_MatrixDispose

Освобождает объект Matrix.

#include <GDIPlus.au3>
_GDIPlus_MatrixDispose ( $hMatrix )

Параметры

$hMatrix Дескриптор объекта Matrix.

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False и устанавливает @error ≠ 0.
@extended: Может содержать код ошибки GPSTATUS ($GDIP_ERR* см. GPIPlusConstants.au3).

См. также

_GDIPlus_MatrixCreate

Дополнительно

Искать GdipDeleteMatrix в библиотеке MSDN

Пример

#include <GDIPlus.au3>
#include <ScreenCapture.au3>
#include <WinAPIHObj.au3>

Example()

Func Example()
    Local $hBitmap1, $hBitmap2, $hImage1, $hImage2, $hGraphic, $iWidth, $iHeight

    ; Инициализировать библиотеку GDI+
    _GDIPlus_Startup()

    ; Захватить весь экран
    $hBitmap1 = _ScreenCapture_Capture("")
    $hImage1 = _GDIPlus_BitmapCreateFromHBITMAP($hBitmap1)

    ; Захватить область экрана
    $hBitmap2 = _ScreenCapture_Capture("", 0, 0, 400, 300)
    $hImage2 = _GDIPlus_BitmapCreateFromHBITMAP($hBitmap2)

    $iWidth = _GDIPlus_ImageGetWidth($hImage2)
    $iHeight = _GDIPlus_ImageGetHeight($hImage2)

    ; Нарисовать одно изображение в другом
    $hGraphic = _GDIPlus_ImageGetGraphicsContext($hImage1)

    ; DrawInsert($hGraphic, $hImage2, $iX, $iY, $fAngle, $iWidth, $iHeight, $iARGB = 0xFF000000, $iPenWidth = 1)
    DrawInsert($hGraphic, $hImage2, 350, 100, 0, $iWidth + 2, $iHeight + 2, 0xFFFF8000, 2)
    DrawInsert($hGraphic, $hImage2, 340, 50, 15, 200, 150, 0xFFFF8000, 4)
    DrawInsert($hGraphic, $hImage2, 310, 30, 35, $iWidth + 4, $iHeight + 4, 0xFFFF00FF, 4)
    DrawInsert($hGraphic, $hImage2, 320, 790, -35, $iWidth, $iHeight)

    ; Сохранить итоговое изображение
    _GDIPlus_ImageSaveToFile($hImage1, @MyDocumentsDir & "\GDIPlus_Image.jpg")

    ; Освободить ресурсы
    _GDIPlus_ImageDispose($hImage1)
    _GDIPlus_ImageDispose($hImage2)
    _WinAPI_DeleteObject($hBitmap1)
    _WinAPI_DeleteObject($hBitmap2)

    ; Завершить работу с библиотекой GDI+
    _GDIPlus_Shutdown()

    ShellExecute(@MyDocumentsDir & "\GDIPlus_Image.jpg")
EndFunc   ;==>Example

; #FUNCTION# ==================================================================================================
; Name...........: DrawInsert
; Description ...: Рисование одного изображения внутри другого
; Syntax.........: DrawInsert($hGraphic, $hImage2, $iX, $iY, $fAngle, $iWidth, $iHeight, $iARGB = 0xFF000000, $iPenWidth = 1)
; вставляет Graphics $hImage2 в $hGraphic
; Parameters ....: $hGraphics – дескриптор объекта Graphics
; $hImage – дескриптор вставляемого объекта Image
; $iX – координата X верхнего левого угла вставляемого изображения
; $iY – координата Y верхнего левого угла вставляемого изображения
; $iWidth – ширина рамки вокруг вставляемого изображения
; $iHeight – высота прямоугольной рамки вокруг вставки
; $iARGB – компоненты альфа, R, G, B цвета пера (цвет рамки)
; $iPenWidth – ширина пера в единицах параметра $iUnit (ширина рамки)

; Return values .: Успех – True
; Ошибка – False
; ==================================================================================================
Func DrawInsert($hGraphic, $hImage2, $iX, $iY, $fAngle, $iWidth, $iHeight, $iARGB = 0xFF000000, $iPenWidth = 1)
    Local $hMatrix, $hPen2

    ; Матрица вращения
    $hMatrix = _GDIPlus_MatrixCreate()
    _GDIPlus_MatrixRotate($hMatrix, $fAngle, "False")
    _GDIPlus_GraphicsSetTransform($hGraphic, $hMatrix)

    _GDIPlus_GraphicsDrawImage($hGraphic, $hImage2, $iX, $iY)

    ; получить pen + цвет
    $hPen2 = _GDIPlus_PenCreate($iARGB, $iPenWidth)

    ; Нарисовать рамку вокруг вставленного изображения
    _GDIPlus_GraphicsDrawRect($hGraphic, $iX, $iY, $iWidth, $iHeight, $hPen2)

    ; Освободить ресурсы
    _GDIPlus_MatrixDispose($hMatrix)
    _GDIPlus_PenDispose($hPen2)
    Return 1
EndFunc   ;==>DrawInsert
