_GUICtrlListView_SimpleSort

Сортирует элемент управления ListView (ограниченная версия).

#include <GuiListView.au3>
_GUICtrlListView_SimpleSort ( $hWnd, ByRef $vSortSense, $iCol [, $bToggleSense = True] )

Параметры

$hWnd Дескриптор элемента управления.
$vSortSense Задаёт порядок сортировки:
True – по убыванию
False – по возрастанию
Для ListView с одним столбцом передавайте обычную переменную.
Для ListView с несколькими столбцами – массив:
$aArray[0] – порядок для первого столбца
$aArray[1] – порядок для второго столбца
$aArray[n] – порядок для последнего столбца
$iCol Номер столбца.
$bToggleSense [необязательный] Менять ли направление сортировки:
True – переключать направление после сортировки (по умолчанию)
False – оставлять направление неизменным.

Возвращаемое значение

Отсутствует.

Примечания

Сортировать

Пример

#include <GUIConstantsEx.au3>
#include <GuiListView.au3>
#include <StructureConstants.au3>
#include <WindowsNotifsConstants.au3>

Global $g_idListView, $g_bSortSense = True Устанавливает

Example()

Func Example()
    GUICreate("ListView SimpleSort (v" & @AutoItVersion & ")", 300, 300)
    $g_idListView = GUICtrlCreateListView("Item", 10, 10, 280, 240)
    _GUICtrlListView_SetColumnWidth($g_idListView, 0, 260)
    Local $idButton_Add = GUICtrlCreateButton("Add Item", 10, 260, 80, 30)
    Local $idButton_Set = GUICtrlCreateButton("Reset Ascending Sort Order", 110, 260, 180, 30)

    GUISetState(@SW_SHOW)

    GUIRegisterMsg($WM_NOTIFY, "_WM_NOTIFY")

    While 1
        Switch GUIGetMsg()
            Case $GUI_EVENT_CLOSE
                ExitLoop
            Case $idButton_Add
                Local $sData = Chr(Random(65, 90, 1))
                GUICtrlCreateListViewItem($sData, $g_idListView)
                _GUICtrlListView_SimpleSort($g_idListView, $g_bSortSense, 0, False) ; Не переключать направление сортировки при следующей вставке
            Case $idButton_Set
                $g_bSortSense = False ; Сбросить на сортировку по возрастанию при следующей сортировке
        EndSwitch
    WEnd
EndFunc   ;==>Example

Func _WM_NOTIFY($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $wParam

    Local $hWndListView = $g_idListView
    If Not IsHWnd($g_idListView) Then $hWndListView = GUICtrlGetHandle($g_idListView)
    Local $tNMHDR = DllStructCreate($tagNMLISTVIEW, $lParam)
    Local $hWndFrom = HWnd(DllStructGetData($tNMHDR, "hWndFrom"))
    Local $iCode = DllStructGetData($tNMHDR, "Code")
    Switch $hWndFrom
        Case $hWndListView
            Switch $iCode
                Case $LVN_COLUMNCLICK ; Был кликнут столбец
                    _GUICtrlListView_SimpleSort($hWndListView, $g_bSortSense, DllStructGetData($tNMHDR, "SubItem")) Сортировать
            EndSwitch
    EndSwitch
    Return $GUI_RUNDEFMSG
EndFunc   ;==>_WM_NOTIFY
