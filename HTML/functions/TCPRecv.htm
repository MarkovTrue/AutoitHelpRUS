TCPRecv

Получает данные из подключённого сокета.

TCPRecv ( mainsocket, maxlen [, flag = 0] )

Параметры

mainsocket Идентификатор подключённого сокета (SocketID), возвращённый функцией TCPAccept() или TCPConnect().
maxlen Максимальное количество принимаемых символов.
flag [необязательный] При значении 1 функция всегда возвращает двоичные данные (по умолчанию 0 – автоопределение).
$TCP_DATA_DEFAULT (0) – (по умолчанию) автоопределение двоичных/строковых данных
$TCP_DATA_BINARY (1) – возвращать двоичные данные

Константы определены в "AutoItConstants.au3".

Возвращаемое значение

Успех: Возвращает двоичные данные или строку, отправленные подключённым сокетом. См. раздел Примечания.
Ошибка: Возвращает пустую строку и устанавливает @error ≠ 0.
@error: -1 – недействительный сокет.
-2 – нет соединения.
Код возврата Windows API WSAGetLastError (см. MSDN).
@extended: 1 – байтов не получено.

Примечания

Для обратной совместимости функция по умолчанию пытается вернуть строку. Если среди принятых данных есть нулевые байты (0x00), возвращается тип Binary.
Чтобы функция всегда возвращала двоичные данные (наиболее надёжный вариант), используйте параметр flag = 1.
Для передачи строк Unicode их необходимо кодировать/декодировать с помощью StringToBinary()/BinaryToString().

Когда данных для получения больше нет, возвращается "" и @extended = 1.

Пример

Пример 1

#include <GUIConstantsEx.au3>
#include <MsgBoxConstants.au3>
#include <WinAPIError.au3>

; Сначала нажмите "1. Server"
; Затем запустите второй экземпляр скрипта, выбрав "2. Клиент"

Example()

Func Example()
    TCPStartup() ; Запустить TCP-сервис.

    ; Зарегистрировать OnAutoItExit для вызова при завершении скрипта.
    OnAutoItExitRegister("OnAutoItExit")

    ; Присвоить локальным переменным loopback-адрес IP и порт.
    Local $sIPAddress = "127.0.0.1" ; Этот IP-адрес работает только для тестирования на локальном компьютере.
    Local $iPort = 65432 ; Порт, используемый для подключения.

    #Region GUI
    Local $sTitle = "TCP Start"
    Local $hGUI = GUICreate($sTitle, 250, 70)

    Local $idBtn_Server = GUICtrlCreateButton("1. Server", 65, 10, 130, 22)

    Local $idBtn_Client = GUICtrlCreateButton("2. Client", 65, 40, 130, 22)

    GUISetState(@SW_SHOW, $hGUI)

    While 1
        Switch GUIGetMsg()
            Case $GUI_EVENT_CLOSE
                ExitLoop
            Case $idBtn_Server
                WinSetTitle($sTitle, "", "TCP Server started")
                GUICtrlSetState($idBtn_Client, $GUI_HIDE)
                GUICtrlSetState($idBtn_Server, $GUI_DISABLE)
                If Not MyTCP_Server($sIPAddress, $iPort) Then ExitLoop
            Case $idBtn_Client
                WinSetTitle($sTitle, "", "TCP Client started")
                GUICtrlSetState($idBtn_Server, $GUI_HIDE)
                GUICtrlSetState($idBtn_Client, $GUI_DISABLE)
                If Not MyTCP_Client($sIPAddress, $iPort) Then ExitLoop
        EndSwitch

        Sleep(10)
    WEnd

    #EndRegion GUI
EndFunc   ;==>Example

Func MyTCP_Client($sIPAddress, $iPort)
    ; Присвоить локальной переменной сокет и подключиться к слушающему сокету по указанным IP и порту.
    Local $iSocket = TCPConnect($sIPAddress, $iPort)
    Local $iError = 0

    ; Если произошла ошибка – показать код ошибки и вернуть False.
    If @error Then
        ; Сервер, вероятно, офлайн, или порт на сервере не открыт.
        $iError = @error
        MsgBox(BitOR($MB_SYSTEMMODAL, $MB_ICONHAND), "", "Client:" & @CRLF & "Could not connect, Error code: " & $iError & @CRLF & _WinAPI_GetErrorMessage($iError))
        Return False
    EndIf

    ; Отправить строку "tata" серверу.
    TCPSend($iSocket, "tata")

    ; Если произошла ошибка – показать код ошибки и вернуть False.
    If @error Then
        $iError = @error
        MsgBox(BitOR($MB_SYSTEMMODAL, $MB_ICONHAND), "", "Client:" & @CRLF & "Could not send the data, Error code: " & $iError & @CRLF & _WinAPI_GetErrorMessage($iError))
        Return False
    EndIf

    ; Закрыть сокет.
    TCPCloseSocket($iSocket)
EndFunc   ;==>MyTCP_Client

Func MyTCP_Server($sIPAddress, $iPort)
    Local $aPos = WinGetPos("TCP Server started")
    WinMove("TCP Server started", "", $aPos[0], $aPos[1] - 100)
    ; Присвоить локальной переменной сокет, привязанный к указанному IP-адресу и порту с максимум 100 ожидающих соединений.
    Local $iListenSocket = TCPListen($sIPAddress, $iPort, 100)
    Local $iError = 0

    If @error Then
        ; Возможно, кто-то уже слушает на этом IP и порту (скрипт уже запущен?).
        $iError = @error
        MsgBox(BitOR($MB_SYSTEMMODAL, $MB_ICONHAND), "", "Server:" & @CRLF & "Could not listen, Error code: " & $iError & @CRLF & _WinAPI_GetErrorMessage($iError))
        Return False
    EndIf

    ; Объявить локальную переменную для клиентского сокета.
    Local $iSocket = 0

    Do ; Ждать подключения (без лимита).
        ; Принимать входящие подключения (сокет закрывается по завершении; один сокет на клиента).
        $iSocket = TCPAccept($iListenSocket)

        ; Если произошла ошибка – показать код ошибки и вернуть False.
        If @error Then
            $iError = @error
            MsgBox(BitOR($MB_SYSTEMMODAL, $MB_ICONHAND), "", "Server:" & @CRLF & "Could not accept the incoming connection, Error code: " & $iError & @CRLF & _WinAPI_GetErrorMessage($iError))
            Return False
        EndIf

        If GUIGetMsg() = $GUI_EVENT_CLOSE Then Return False
    Until $iSocket <> -1 ;если не равно -1 – клиент подключён.

    ; Закрыть слушающий сокет, чтобы разрешить дальнейшие привязки.
    TCPCloseSocket($iListenSocket)

    ; Присвоить локальной переменной полученные данные.
    Local $sReceived = TCPRecv($iSocket, 4) ; Ожидаем строку "tata" или "toto" (пример TCPRecv): длина 4 байта.

    ; Примечание: если не знаете заранее длину данных,
    ; используйте, например, 2048 для maxlen и вызывайте функцию, пока она не вернёт пустоту или ошибку.

    ; Показать полученную строку.
    MsgBox($MB_SYSTEMMODAL, "", "Server:" & @CRLF & "Received: " & $sReceived)

    ; Закрыть сокет.
    TCPCloseSocket($iSocket)
EndFunc   ;==>MyTCP_Server

Func OnAutoItExit()
    TCPShutdown() ; Закрыть TCP-сервис.
EndFunc   ;==>OnAutoItExit

Пример 2

; Я клиент, запусти меня после сервера! (Сначала запусти пример 2 функции TCPSend).

#include <AutoItConstants.au3>
#include <FileConstants.au3>
#include <MsgBoxConstants.au3>
#include <WinAPIError.au3>

Example()

Func Example()
    Local $sMsgBoxTitle = "AutoItVersion = " & @AutoItVersion

    TCPStartup() ; Запустить TCP-сервис.

    ; Зарегистрировать OnAutoItExit для вызова при завершении скрипта.
    OnAutoItExitRegister("OnAutoItExit")

    ; Присвоить локальным переменным loopback-адрес IP и порт.
    Local $sIPAddress = "127.0.0.1" ; Этот IP-адрес работает только для тестирования на локальном компьютере.
    Local $iPort = 65432 ; Порт, используемый для подключения.

    ; Присвоить локальной переменной сокет и подключиться к слушающему сокету по указанным IP и порту.
    Local $iSocket = TCPConnect($sIPAddress, $iPort)

    ; Если произошла ошибка – показать код ошибки и вернуть False.
    If @error Then
        ; Сервер, вероятно, офлайн, или порт на сервере не открыт.
        MsgBox(($MB_SYSTEMMODAL + $MB_ICONHAND), $sMsgBoxTitle, "Client: Could not connect" & @CRLF & " Error code: " & @error & @CRLF & @CRLF & _WinAPI_GetErrorMessage(@error))
        Return False
    EndIf

    ; Присвоить локальной переменной путь к файлу, который будет принят.
    Local $sFilePath = @TempDir & "\temp.dat"
    FileDelete($sFilePath)

    ; Если произошла ошибка – показать код ошибки и вернуть False.
    If @error Then
        MsgBox(($MB_SYSTEMMODAL + $MB_ICONEXCLAMATION), $sMsgBoxTitle, "Client: Invalid file chosen" & @CRLF & " Error code: " & @error & @CRLF & @CRLF & _WinAPI_GetErrorMessage(@error))
        Return False
    EndIf

    ; Присвоить локальной переменной дескриптор файла, открытого в режиме двоичной перезаписи.
    Local $hFile = FileOpen($sFilePath, BitOR($FO_BINARY, $FO_OVERWRITE))

    ; Задать локальные константы: число 4 КиБ; двоичный код конца файла и его длину.
    Local $i4KiB = 4096
    If $CmdLine[0] Then $i4KiB = $CmdLine[1]

    Local $dData, $nReceivedBytes = 0, $n = 0, $fDiffTime, $hTimer = TimerInit()
    #forceref $dData
    While 1
        $dData = TCPRecv($iSocket, $i4KiB, $TCP_DATA_BINARY)

        ; Если произошла ошибка – показать код ошибки и вернуть False.
        If @error Then
            MsgBox(($MB_SYSTEMMODAL + $MB_ICONHAND), $sMsgBoxTitle, "Client: Connection lost n=" & $n & @CRLF & " Error code: " & @error & @CRLF & @CRLF & _WinAPI_GetErrorMessage(@error))
            Return False
        ElseIf @extended = 1 Or BinaryLen($dData) = 0 Then
            ; Если ничего не получено.
            ExitLoop
        EndIf

        $nReceivedBytes += BinaryLen($dData)
        $n += 1
;~ FileWrite($hFile, $dData) ; to be uncommented if file content to be written
    WEnd
    $fDiffTime = TimerDiff($hTimer)
    ; Закрыть дескриптор файла.
    FileClose($hFile)

    ; Отобразить сообщение об успешном завершении.
;~ MsgBox($MB_SYSTEMMODAL, "", "Client:" & @CRLF & "File received nbBytes=" & & FileGetSize($sFilePath))
    MsgBox($MB_SYSTEMMODAL, $sMsgBoxTitle, "Client: receivedBytes = " & $nReceivedBytes & " packetSize = " & $i4KiB & @CRLF & "File received TCPRecv = " & Int($fDiffTime / $n * 1000) & "µs Total = " & Int($fDiffTime) & "ms nPacket = " & $n)

    ; Закрыть сокет.
    TCPCloseSocket($iSocket)
EndFunc   ;==>Example

Func OnAutoItExit()
    TCPShutdown() ; Закрыть TCP-сервис.
EndFunc   ;==>OnAutoItExit
