Справочник функций

DllStructGetPtr

Возвращает указатель на структуру или на элемент структуры.

DllStructGetPtr ( Struct [, Element] )

Параметры

Struct Структура, возвращаемая функцией DllStructCreate().
Element [необязательный] Элемент структуры, указатель которого нужен: номер, начиная с 1, или имя элемента, определённое в DllStructCreate().

Возвращаемое значение

Успех: Возвращает указатель на структуру.
Ошибка: Возвращает 0.
@error: 1 – структура не является допустимой структурой, возвращённой DllStructCreate().
2 – элемент выходит за пределы диапазона.

Примечания

Используется в DllCall().

См. также

DllCall, DllStructCreate

Пример

Пример 1

#include <MsgBoxConstants.au3>
#include <StructureConstants.au3>

Example()

Func Example()
    ; Присвоить локальной переменной дескриптор текущего активного окна.
    Local $hWnd = WinGetHandle("") ; То же самое, что: "[active]".

    ; Присвоить локальной переменной прямоугольник окна (массив).
    Local $aWndPos = WinGetPos($hWnd)

    ; Присвоить локальной переменной структуру, созданную по определению tagRECT.
    Local $tRECT = DllStructCreate($tagRECT)
    Local $iError = 0

    ; Если произошла ошибка – показать код ошибки и вернуть False.
    If @error Then
        $iError = @error
        MsgBox(($MB_ICONERROR + $MB_SYSTEMMODAL), Default, "Error in DllStructCreate, Code: " & $iError)
        Return False
    EndIf

    ; Выполнить DllCall функции GetWindowRect.
    DllCall("user32.dll", "int", "GetWindowRect", _
            "hwnd", $hWnd, _
            "ptr", DllStructGetPtr($tRECT))

    ; Если произошла ошибка – показать код ошибки и вернуть False.
    If @error Then
        $iError = @error
        MsgBox(($MB_ICONERROR + $MB_SYSTEMMODAL), Default, "Error in DllCall, Code: " & $iError)
        Return False
    EndIf

    ; Примечание: второй параметр функции GetWindowRect требует указатель,
    ; результат DllStructCreate НЕ является указателем, но тип 'struct*' позволяет передать структуру как указатель.

    ; Присвоить локальным переменным возвращённый прямоугольник.
    Local $iLeft = DllStructGetData($tRECT, "Left") ; Или 1 вместо "Left".
    Local $iTop = DllStructGetData($tRECT, 2) ; Или "Top" вместо 2.
    Local $iRight = DllStructGetData($tRECT, 3) ; Или "Right" вместо 3.
    Local $iBottom = DllStructGetData($tRECT, "Bottom") ; Или 4 вместо "Bottom".

    ; Освободить ресурсы, занятые структурой.
    $tRECT = 0

    ; Отобразить результаты WinGetPos и возвращённый прямоугольник.
    MsgBox($MB_SYSTEMMODAL, "", "WinGetPos(): (" & $aWndPos[0] & ", " & $aWndPos[1] & ") " & _
            "(" & $aWndPos[2] + $aWndPos[0] & ", " & $aWndPos[3] + $aWndPos[1] & ")" & @CRLF & _
            "GetWindowRect(): (" & $iLeft & ", " & $iTop & ") (" & $iRight & ", " & $iBottom & ")")
EndFunc   ;==>Example

Пример 2

#include <MsgBoxConstants.au3>

Example()

Func Example()
    ; Присвоить локальной переменной структуру с определением int.
    Local $tSTRUCT1 = DllStructCreate("int")
    Local $iError = 0

    ; Если произошла ошибка – показать код ошибки и вернуть False.
    If @error Then
        $iError = @error
        MsgBox(($MB_ICONERROR + $MB_SYSTEMMODAL), Default, "Error in DllCall, Code: " & $iError)
        Return False
    EndIf

    ; Присвоить локальной переменной структуру tSTRUCT1, но с элементами, полученными согласно новому определению.
    Local $tSTRUCT2 = DllStructCreate("uint", DllStructGetPtr($tSTRUCT1, 1))

    ; Если произошла ошибка – показать код ошибки и вернуть False.
    If @error Then
        $iError = @error
        MsgBox(($MB_ICONERROR + $MB_SYSTEMMODAL), Default, "Error in DllCall, Code: " & $iError)
        Return False
    EndIf

    Local $tSTRUCT3 = DllStructCreate("float", DllStructGetPtr($tSTRUCT1, 1))

    ; Если произошла ошибка – показать код ошибки и вернуть False.
    If @error Then
        $iError = @error
        MsgBox(($MB_ICONERROR + $MB_SYSTEMMODAL), Default, "Error in DllCall, Code: " & $iError)
        Return False
    EndIf

    ; Примечания: – структуры указывают на одну и ту же структуру (имеют одинаковый указатель);
    ; Первый элемент, полученный через:
    ; структуру *tSTRUCT1, будет int.
    ; структуру *tSTRUCT2, будет unsigned int (элемент приводится из int в uint).
    ; структуру *tSTRUCT3, будет float (элемент приводится из int в float).
    ;
    ; – используется оператор приведения "reinterpret_cast".

    ; Установить данные первого элемента (int) в $tSTRUCT1.
    DllStructSetData($tSTRUCT1, 1, -1)

    ; Отобразить различные типы данных для одних и тех же данных.
    MsgBox($MB_SYSTEMMODAL, "", _
            "int: " & DllStructGetData($tSTRUCT1, 1) & @CRLF & _
            "uint: " & DllStructGetData($tSTRUCT2, 1) & @CRLF & _
            "float: " & DllStructGetData($tSTRUCT3, 1))

    ; Освободить ресурсы, используемые структурами.
    $tSTRUCT1 = 0
    $tSTRUCT2 = 0
    $tSTRUCT3 = 0
EndFunc   ;==>Example
