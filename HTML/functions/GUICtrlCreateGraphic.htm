GUICtrlCreateGraphic

Создаёт графический элемент управления (Graphic) для GUI.

GUICtrlCreateGraphic ( left, top [, width [, height [, style]]] )

Параметры

left Левая граница элемента управления. Если указано -1, значение будет вычислено согласно GUICoordMode.
top Верхняя граница элемента управления. Если указано -1, значение будет вычислено согласно GUICoordMode.
width [необязательный] Ширина элемента управления (по умолчанию – ранее использованная ширина).
height [необязательный] Высота элемента управления (по умолчанию – ранее использованная высота).
style [необязательный] Задаёт стиль элемента управления. См. приложение "Стили элементов управления GUI".
по умолчанию (-1) : $SS_NOTIFY.

Возвращаемое значение

Успех: Возвращает идентификатор нового элемента управления (controlID).
Ошибка: Возвращает 0.

Примечания

Для рисования в элементе управления см. GUICtrlSetGraphic().

Цвет рамки и фона задаётся с помощью GUICtrlSetBkColor() и GUICtrlSetColor().

Графический элемент управления реагирует на нажатия, поэтому он не должен перекрываться с другими элементами. Если перекрытие необходимо, важно отключить графический элемент: GUICtrlSetState(-1, $GUI_DISABLE).

Этот элемент управления не поддерживает изменение размера из-за фиксированной относительной адресации графики, однако можно использовать привязку через GUICtrlSetResizing().

См. также

GUICoordMode (Option), GUICtrlDelete, GUICtrlSetBkColor, GUICtrlSetColor, GUICtrlSetGraphic, GUICtrlSetResizing, GUICtrlSetStyle, GUIGetMsg

Пример

#include <GUIConstantsEx.au3>
#include <MsgBoxConstants.au3>
#include <StaticConstants.au3>

Global Const $g_MAXGr = 6
Global $g_a_idGrc_Array[$g_MAXGr + 1] ; записи 0 и $g_MAXGr не используются для возможности GUICtrlDelete result
Global $g_idBtn_Del

Example()

Func Example()
    Local $idMsg, $iInc, $i

    CreateChild()

    $i = 1
    $iInc = 1
    ; Цикл до выхода пользователя.
    Do
        $idMsg = GUIGetMsg()

        If $idMsg = $g_idBtn_Del Then
            GUICtrlDelete($g_a_idGrc_Array[$i])
            $i = $i + $iInc
            If $i < 0 Or $i > $g_MAXGr Then Exit
        EndIf
        If $idMsg > 0 Then MsgBox($MB_SYSTEMMODAL, "clicked", $idMsg & @CRLF & $g_a_idGrc_Array[5], 2)
    Until $idMsg = $GUI_EVENT_CLOSE
EndFunc   ;==>Example

Func CreateChild()
    GUICreate("My Draw")
    $g_idBtn_Del = GUICtrlCreateButton("Delete", 50, 165, 50)

    $g_a_idGrc_Array[1] = GUICtrlCreateGraphic(20, 50, 100, 100)
    GUICtrlSetBkColor(-1, 0xffffff)
    GUICtrlSetColor(-1, 0)

    GUICtrlSetGraphic(-1, $GUI_GR_COLOR, 0xff0000, 0xff0000)
    GUICtrlSetGraphic(-1, $GUI_GR_PIE, 50, 50, 40, 30, 270)
    GUICtrlSetGraphic(-1, $GUI_GR_COLOR, 0x00ff00, 0xffffff)
    GUICtrlSetGraphic(-1, $GUI_GR_PIE, 58, 50, 40, -60, 90)

    GUICtrlSetGraphic(-1, $GUI_GR_ELLIPSE, 100, 100, 50, 80)
    GUICtrlSetGraphic(-1, $GUI_GR_COLOR, 0x00ff00, 0xc0c0ff)
    GUICtrlSetGraphic(-1, $GUI_GR_RECT, 350, 200, 50, 80)
    GUICtrlCreateLabel("label", 65, 100, 30)
    GUICtrlSetColor(-1, 0xff)

    $g_a_idGrc_Array[2] = GUICtrlCreateGraphic(220, 50, 100, 100)
    GUICtrlSetStyle(-1, $SS_NOTIFY)
    GUICtrlSetGraphic(-1, $GUI_GR_COLOR, 0, 0xff)
    GUICtrlSetGraphic(-1, $GUI_GR_PIE, 50, 50, 40, 30, 270)
    GUICtrlSetGraphic(-1, $GUI_GR_COLOR, 0x00ff00, 0xffffff)
    GUICtrlSetGraphic(-1, $GUI_GR_PIE, 58, 50, 40, -60, 90)

    $g_a_idGrc_Array[3] = GUICtrlCreateGraphic(220, 150, 100, 100, 0)
    GUICtrlSetBkColor(-1, 0xf08080)
    GUICtrlSetColor(-1, 0xff)
    GUICtrlSetGraphic(-1, $GUI_GR_COLOR, 0xff00)
    GUICtrlSetGraphic(-1, $GUI_GR_RECT, 50, 50, 80, 80)

    $g_a_idGrc_Array[4] = GUICtrlCreateGraphic(20, 200, 80, 80)
    GUICtrlSetState(-1, $GUI_DISABLE)
    GUICtrlSetBkColor(-1, 0xffffff)
    GUICtrlSetGraphic(-1, $GUI_GR_MOVE, 10, 10)
    GUICtrlSetGraphic(-1, $GUI_GR_COLOR, 0xff)
    GUICtrlSetGraphic(-1, $GUI_GR_LINE, 30, 40)
    GUICtrlSetGraphic(-1, $GUI_GR_COLOR, 0xff00)
    GUICtrlSetGraphic(-1, $GUI_GR_LINE, 70, 70)
    GUICtrlSetGraphic(-1, $GUI_GR_COLOR, 0xff0000)
    GUICtrlSetGraphic(-1, $GUI_GR_LINE, 10, 50)
    GUICtrlSetGraphic(-1, $GUI_GR_COLOR, 0xffff00)
    GUICtrlSetGraphic(-1, $GUI_GR_LINE, 10, 10)

    $g_a_idGrc_Array[5] = GUICtrlCreateGraphic(150, 10, 50, 50, 0)
    GUICtrlSetBkColor(-1, 0xa0ffa0)
    GUICtrlSetGraphic(-1, $GUI_GR_MOVE, 20, 20) ; начальная точка
    ; лучше сначала рисовать линии, потом точки
    ; чтобы не менять цвет при каждой операции рисования
    GUICtrlSetGraphic(-1, $GUI_GR_COLOR, 0x0000ff)
    GUICtrlSetGraphic(-1, $GUI_GR_DOT, 30, 30)
    GUICtrlSetGraphic(-1, $GUI_GR_COLOR, 0)
    GUICtrlSetGraphic(-1, $GUI_GR_LINE, 20, 40)
    GUICtrlSetGraphic(-1, $GUI_GR_COLOR, 0xff0000)
    GUICtrlSetGraphic(-1, $GUI_GR_DOT, 25, 25)
    GUICtrlSetGraphic(-1, $GUI_GR_COLOR, 0)
    GUICtrlSetGraphic(-1, $GUI_GR_LINE, 40, 40)
    GUICtrlSetGraphic(-1, $GUI_GR_DOT, 40, 40)

    GUISetState(@SW_SHOW)
EndFunc   ;==>CreateChild
