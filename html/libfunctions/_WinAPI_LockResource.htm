Справочник функций

_WinAPI_LockResource

Блокирует указанный ресурс в памяти.

#include <WinAPIRes.au3>
_WinAPI_LockResource ( $hData )

Параметры

$hData Дескриптор блокируемого ресурса. Функция _WinAPI_LoadResource() возвращает этот дескриптор. Не
передавайте в качестве параметра никакое другое значение, кроме успешного возвращаемого значения функции LoadResource.

Возвращаемое значение

Успех: Возвращает указатель на первый байт ресурса.
Ошибка: Возвращает 0.

Примечания

Указатель, возвращаемый этой функцией, действителен до выгрузки модуля, содержащего ресурс. Не требуется
разблокировать ресурсы, так как система автоматически удаляет их при завершении процесса, который их создал.

Не пытайтесь блокировать ресурс, используя дескриптор, возвращаемый функцией _WinAPI_FindResource() или _WinAPI_FindResourceEx().
Такой дескриптор указывает на случайные данные.

Обратите внимание, что _WinAPI_LockResource() не блокирует память; она просто используется для получения указателя на память,
содержащую данные ресурса. Имя функции происходит из версий до Windows XP, когда она была
использована для блокировки глобального блока памяти, выделенного функцией _WinAPI_LoadResource().

См. также

_WinAPI_FindResource, _WinAPI_FindResourceEx, _WinAPI_LoadResource, _WinAPI_UnlockFile

Дополнительно

Искать LockResource в библиотеке MSDN

Пример

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <Memory.au3>
#include <StaticConstants.au3>
#include <WinAPIGdi.au3>
#include <WinAPIMisc.au3>
#include <WinAPIRes.au3>

Local Const $sJpg = @TempDir & '\~Tech.jpg'

; Загрузить Resources.dll в память
Local $sDll = _Extras_PathFull('Resources.dll')
Local $hInstance = _WinAPI_LoadLibraryEx($sDll, $LOAD_LIBRARY_AS_DATAFILE)

; Загрузить ресурс JPEG из библиотеки Resources.dll
Local $hResource = _WinAPI_FindResource($hInstance, 'JPEG', 1)
Local $iSize = _WinAPI_SizeOfResource($hInstance, $hResource)
Local $hData = _WinAPI_LoadResource($hInstance, $hResource)
Local $pData = _WinAPI_LockResource($hData)

; Сохранить ресурс в .jpg файл
Local $hFile = FileOpen($sJpg, 2 + 16)
Local $tData = DllStructCreate('byte[' & $iSize & ']', $pData)
FileWrite($hFile, DllStructGetData($tData, 1))
FileClose($hFile)

; Загрузить ресурс FONT из библиотеки Resources.dll
$hResource = _WinAPI_FindResource($hInstance, $RT_FONT, 'TECHNOVIA_CAPS')
$iSize = _WinAPI_SizeOfResource($hInstance, $hResource)
$hData = _WinAPI_LoadResource($hInstance, $hResource)
$pData = _WinAPI_LockResource($hData)

; Добавить шрифт из памяти в систему
Local $hFont = _WinAPI_AddFontMemResourceEx($pData, $iSize)

; Загрузить ресурс SOUND из библиотеки Resources.dll
$hResource = _WinAPI_FindResource($hInstance, 'SOUND', 1)
$iSize = _WinAPI_SizeOfResource($hInstance, $hResource)
$hData = _WinAPI_LoadResource($hInstance, $hResource)
$pData = _WinAPI_LockResource($hData)

; Скопировать WAV в память для использования позже
Local $hWave = _MemGlobalAlloc($iSize, $GMEM_MOVEABLE)
Local $pWave = _MemGlobalLock($hWave)
_MemMoveMemory($pData, $pWave, $iSize)
; _MemGlobalUnlock($hWave)

; Загрузить ресурс STRING из библиотеки Resources.dll
Local $sText = _WinAPI_LoadString($hInstance, 1)

; Выгрузить Resources.dll из памяти
_WinAPI_FreeLibrary($hInstance)

; Создать GUI
GUICreate('Test ' & StringReplace(@ScriptName, '.au3', '()'), 350, 350)
GUICtrlCreatePic($sJpg, 0, 0, 350, 350)
GUICtrlSetState(-1, $GUI_DISABLE)
GUICtrlCreateLabel($sText, 10, 18, 330, 36, $SS_CENTER)
GUICtrlSetFont(-1, 30, -1, -1, 'Technovia Caps')
GUICtrlSetBkColor(-1, $GUI_BKCOLOR_TRANSPARENT)
GUICtrlSetColor(-1, 0xF06000)
Local $idButton = GUICtrlCreateButton('Play Sound', 125, 316, 100, 23)
GUISetState(@SW_SHOW)

While 1
    Switch GUIGetMsg()
        Case $GUI_EVENT_CLOSE
            ExitLoop
        Case $idButton
            _WinAPI_PlaySound($pWave, BitOR($SND_ASYNC, $SND_MEMORY, $SND_NOWAIT))
    EndSwitch
WEnd

; Освободить ресурсы
_WinAPI_RemoveFontMemResourceEx($hFont)
FileDelete($sJpg)
