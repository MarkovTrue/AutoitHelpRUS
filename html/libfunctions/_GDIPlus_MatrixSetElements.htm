_GDIPlus_MatrixSetElements

Задаёт элементы матрицы.

#include <GDIPlus.au3>
_GDIPlus_MatrixSetElements ( $hMatrix [, $nM11 = 1 [, $nM12 = 0 [, $nM21 = 0 [, $nM22 = 1 [, $nDX = 0 [, $nDY = 0]]]]]] )

Параметры

$hMatrix Значение дескриптора.
$nM11 [необязательный] Строка 1, столбец 1 (горизонтальное масштабирование). По умолчанию 1.
$nM12 [необязательный] Значение с плавающей точкой. По умолчанию: 0
$nM21 [необязательный] Значение с плавающей точкой. По умолчанию: 0
$nM22 [необязательный] Строка 2, столбец 2 (вертикальное масштабирование). По умолчанию 1.
$nDX [необязательный] Значение с плавающей точкой. По умолчанию: 0
$nDY [необязательный] Значение с плавающей точкой. По умолчанию: 0

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False и устанавливает @error ≠ 0.
@extended: Может содержать код ошибки GPSTATUS ($GDIP_ERR* см. GPIPlusConstants.au3).

Пример

#include <GDIPlus.au3>
#include <GUIConstantsEx.au3>

Example()

Func Example()
    Local $iW, $iH, $hGUI, $hGraphic, $hBrush, $hPen, $hPath, $hFamily, $tLayout, $hMatrix, $aBounds

    ; Создать GUI
    $iW = 640
    $iH = 220
    $hGUI = GUICreate("GDI+", $iW, $iH)
    GUISetState(@SW_SHOW)

    ; Нарисовать строку через path
    _GDIPlus_Startup()
    $hGraphic = _GDIPlus_GraphicsCreateFromHWND($hGUI) ;Создать графический объект из дескриптора окна
    _GDIPlus_GraphicsSetSmoothingMode($hGraphic, $GDIP_SMOOTHINGMODE_HIGHQUALITY) ;Задаёт качество отрисовки графики (сглаживание)
    _GDIPlus_GraphicsClear($hGraphic, 0xFF000000)

    $hBrush = _GDIPlus_BrushCreateSolid(0xFFDD2200)
    $hPen = _GDIPlus_PenCreate(0xFFFFBB00, 2)

    $hPath = _GDIPlus_PathCreate() ;Создать новый объект path

    $hFamily = _GDIPlus_FontFamilyCreate("Arial") ;Создать объект font family
    $tLayout = _GDIPlus_RectFCreate() ;Создать ограничивающий прямоугольник строки X=0, Y=0
    _GDIPlus_PathAddString($hPath, "Fit to Window", $tLayout, $hFamily) ;Добавить контур строки в path

    ; Трансформировать путь для подгонки к окну
    $aBounds = _GDIPlus_PathGetWorldBounds($hPath) ;Получить ограничивающий прямоугольник пути
    $hMatrix = _GDIPlus_MatrixCreate()
    _GDIPlus_MatrixTranslate($hMatrix, -$aBounds[0], -$aBounds[1]) ;Сдвинуть матрицу на смещение ограничивающего прямоугольника
    _GDIPlus_MatrixScale($hMatrix, $iW / $aBounds[2], $iH / $aBounds[3], True) ;Матрица масштабирования
    _GDIPlus_PathTransform($hPath, $hMatrix) ;Сдвинуть и масштабировать путь

    _GDIPlus_GraphicsFillPath($hGraphic, $hPath, $hBrush) ;Залить path в дескрипторе графики (GUI)
    _GDIPlus_GraphicsDrawPath($hGraphic, $hPath, $hPen) ;Нарисовать path в дескрипторе графики (GUI)

    ; Трансформировать путь для центрирования
    _GDIPlus_PathReset($hPath) ;Сбросить путь (удалить предыдущую строку)
    _GDIPlus_PathAddString($hPath, "Center", $tLayout, $hFamily, 0, 100, 0) ;Добавить контур строки в path

    $aBounds = _GDIPlus_PathGetWorldBounds($hPath)
    _GDIPlus_MatrixSetElements($hMatrix, 1, 0, 0, 1, 0, 0) ;Сбросить матрицу
    _GDIPlus_MatrixTranslate($hMatrix, ($iW / 2) - $aBounds[0] - ($aBounds[2] / 2), ($iH / 2) - $aBounds[1] - ($aBounds[3] / 2))
    _GDIPlus_PathTransform($hPath, $hMatrix) ;Сдвиг (offset) пути

    _GDIPlus_BrushSetSolidColor($hBrush, 0x7F004488)
    _GDIPlus_PenSetColor($hPen, 0xFF00AAFF)
    _GDIPlus_GraphicsFillPath($hGraphic, $hPath, $hBrush) ;Залить path в дескрипторе графики (GUI)
    _GDIPlus_GraphicsDrawPath($hGraphic, $hPath, $hPen) ;Нарисовать path в дескрипторе графики (GUI)

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE

    ; Освободить ресурсы
    _GDIPlus_MatrixDispose($hMatrix)
    _GDIPlus_FontFamilyDispose($hFamily)
    _GDIPlus_PathDispose($hPath)
    _GDIPlus_PenDispose($hPen)
    _GDIPlus_BrushDispose($hBrush)
    _GDIPlus_GraphicsDispose($hGraphic)
    _GDIPlus_Shutdown()
EndFunc   ;==>Example
