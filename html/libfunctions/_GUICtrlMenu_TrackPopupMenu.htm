_GUICtrlMenu_TrackPopupMenu

Отображает контекстное меню в указанном месте.

#include <GuiMenu.au3>
_GUICtrlMenu_TrackPopupMenu ( $hMenu, $hWnd [, $iX = -1 [, $iY = -1 [, $iAlignX = 1 [, $iAlignY = 1 [, $iNotify = 0 [, $iButtons = 0]]]]]] )

Параметры

$hMenu Дескриптор отображаемого контекстного меню.
$hWnd Дескриптор окна.
$iX [необязательный] Горизонтальная позиция меню в экранных координатах.
Если -1 – используется текущая позиция мыши.
$iY [необязательный] Вертикальная позиция меню в экранных координатах.
Если -1 – используется текущая позиция мыши.
$iAlignX [необязательный] Горизонтальное позиционирование меню:
0 – центрировать меню по горизонтали относительно $iX
1 – левая сторона меню выравнивается по $iX
2 – правая сторона меню выравнивается по $iX
$iAlignY [необязательный] Вертикальное позиционирование меню:
0 – позиционировать меню над $iY
4 – центрировать по вертикали относительно $iY
8 – позиционировать меню ниже $iY
$iNotify [необязательный] Определяет выбор без родительского окна:
1 – не посылать уведомление о выборе
2 – вернуть идентификатор выбранного пункта
$iButtons [необязательный] Кнопка мыши для отслеживания:
0 – левая кнопка
1 – правая кнопка

Возвращаемое значение

Успех: Если $iNotify равен 2, возвращается идентификатор выбранного пункта. Если пользователь закрыл меню без выбора – возвращается 0.
Ошибка: Возвращает 0.

Искать TrackPopupMenu в библиотеке MSDN

Пример

#include <GUIConstantsEx.au3>
#include <GuiMenu.au3>
#include <WinAPIError.au3>
#include <WindowsNotifsConstants.au3>

Global Enum $e_idOpen = 1000, $e_idSave, $e_idInfo

Example()

Func Example()
    ; Создать GUI
    GUICreate("Menu", 400, 300)
    GUISetState(@SW_SHOW)

    ; Зарегистрировать обработчики сообщений
    GUIRegisterMsg($WM_COMMAND, "WM_COMMAND")
    GUIRegisterMsg($WM_CONTEXTMENU, "WM_CONTEXTMENU")

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
EndFunc   ;==>Example

; Обработать сообщения WM_COMMAND
Func WM_COMMAND($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $lParam
    Switch $wParam
        Case $e_idOpen
            _WinAPI_ShowMsg("Open")
        Case $e_idSave
            _WinAPI_ShowMsg("Save")
        Case $e_idInfo
            _WinAPI_ShowMsg("Info")
    EndSwitch
EndFunc   ;==>WM_COMMAND

; Обработать сообщения WM_CONTEXTMENU
Func WM_CONTEXTMENU($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $lParam
    Local $hMenu

    $hMenu = _GUICtrlMenu_CreatePopup()
    _GUICtrlMenu_InsertMenuItem($hMenu, 0, "Open", $e_idOpen)
    _GUICtrlMenu_InsertMenuItem($hMenu, 1, "Save", $e_idSave)
    _GUICtrlMenu_InsertMenuItem($hMenu, 3, "", 0)
    _GUICtrlMenu_InsertMenuItem($hMenu, 3, "Info", $e_idInfo)
    _GUICtrlMenu_TrackPopupMenu($hMenu, $wParam)
    _GUICtrlMenu_DestroyMenu($hMenu)
    Return True
EndFunc   ;==>WM_CONTEXTMENU
