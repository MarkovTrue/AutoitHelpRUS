Справочник функций

_GUICtrlEdit_Create

Создаёт элемент управления Edit.

#include <GuiEdit.au3>
_GUICtrlEdit_Create ( $hWnd, $sText, $iX, $iY [, $iWidth = 150 [, $iHeight = 150 [, $iStyle = 0x003010C4 [, $iExStyle = 0x00000200]]]] )

Параметры

$hWnd Дескриптор родительского или владеющего окна.
$sText Текст для отображения в элементе управления.
$iX Горизонтальная позиция элемента управления.
$iY Вертикальная позиция элемента управления.
$iWidth [необязательный] Ширина элемента управления
$iHeight [необязательный] Высота элемента управления
$iStyle [необязательный] Стили элемента управления:
$ES_AUTOHSCROLL – автоматически прокручивает текст вправо на 10 символов при вводе пользователем символа в конце строки.
$ES_AUTOVSCROLL – автоматически прокручивает текст вверх на одну страницу при нажатии пользователем клавиши ENTER на последней строке.
$ES_CENTER – центрирует текст в элементе управления Edit.
$ES_LEFT – выравнивает текст по левому краю.
$ES_LOWERCASE – преобразует все символы в нижний регистр при вводе в элемент управления Edit.
$ES_MULTILINE – обозначает многострочный элемент управления Edit.
$ES_NOHIDESEL – выбранный текст инвертирует, даже если элемент управления не имеет фокуса.
$ES_NUMBER – разрешает вводить только цифры в элемент управления Edit.
$ES_OEMCONVERT – преобразует текст, введённый в элемент управления Edit.
$ES_READONLY – предотвращает ввод и редактирование текста в элементе управления Edit.
$ES_RIGHT – выравнивает текст по правому краю в элементе управления Edit.
$ES_UPPERCASE – преобразует все символы в верхний регистр при вводе в элемент управления Edit.
$ES_WANTRETURN – указывает, что возврат каретки вставляется при нажатии пользователем клавиши ENTER.
$ES_PASSWORD – отображает звёздочку (*) для каждого введённого символа в элемент управления Edit

По умолчанию: $ES_MULTILINE, $ES_WANTRETURN, $WS_VSCROLL, $WS_HSCROLL, $ES_AUTOVSCROLL, $ES_AUTOHSCROLL
Обязательно: WS_CHILD, $WS_VISIBLE, $WS_TABSTOP только если не используется $ES_READONLY.
$iExStyle [необязательный] Расширенный стиль элемента. Соответствует стандартным константам $WS_EX_*. См. таблицу расширенных стилей.

Возвращаемое значение

Успех: Возвращает дескриптор элемента управления Edit.
Ошибка: Возвращает 0.

Примечания

Данная функция предназначена для опытных пользователей и для изучения принципов работы элемента управления.

См. также

_GUICtrlEdit_Destroy

Пример

Пример 1

; == Example 1 : _GUICtrlEdit_AppendText()

#include "Extras\WM_NOTIFY.au3"

#include <GUIConstantsEx.au3>
#include <GuiEdit.au3>
#include <WinAPIConv.au3>
#include <WindowsNotifsConstants.au3>

Global $g_hEdit

Example()

Func Example()
    ; Создать GUI
    Local $hGUI = GUICreate("Edit Create (v" & @AutoItVersion & ")", 400, 300)
    $g_hEdit = _GUICtrlEdit_Create($hGUI, "This is a test" & @CRLF & "Another Line", 2, 2, 394, 268)
    GUISetState(@SW_SHOW)

    GUIRegisterMsg($WM_COMMAND, "WM_COMMAND")

    _GUICtrlEdit_AppendText($g_hEdit, @CRLF & "Append to the end?")

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
    GUIDelete()
EndFunc   ;==>Example

Func WM_COMMAND($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg
    Local $hWndEdit = $g_hEdit
    If Not IsHWnd($g_hEdit) Then $hWndEdit = GUICtrlGetHandle($g_hEdit)
    Local $hWndFrom = $lParam
    Local $iIDFrom = _WinAPI_LoWord($wParam)
    Local $iCode = _WinAPI_HiWord($wParam)
    Switch $hWndFrom
        Case $g_hEdit, $hWndEdit
            Switch $iCode
                Case $EN_ALIGN_LTR_EC ; Отправляется при смене пользователем направления edit-элемента на слева-направо
                    _WM_NOTIFY_DebugInfo("$EN_ALIGN_LTR_EC", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $EN_ALIGN_RTL_EC ; Отправляется при смене пользователем направления edit-элемента на справа-налево
                    _WM_NOTIFY_DebugInfo("$EN_ALIGN_RTL_EC", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $EN_CHANGE ; Отправляется при действии пользователя, потенциально изменившем текст edit-элемента
                    _WM_NOTIFY_DebugInfo("$EN_CHANGE", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $EN_ERRSPACE ; Отправляется, когда edit-элемент не может выделить достаточно памяти для запроса
                    _WM_NOTIFY_DebugInfo("$EN_ERRSPACE", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $EN_HSCROLL ; Отправляется при клике пользователя по горизонтальной прокрутке edit-элемента
                    _WM_NOTIFY_DebugInfo("$EN_HSCROLL", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $EN_KILLFOCUS ; Отправляется при потере клавиатурного фокуса edit-элементом
                    _WM_NOTIFY_DebugInfo("$EN_KILLFOCUS", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $EN_MAXTEXT ; Отправляется, когда вставляемый текст превысил заданное число символов в edit-элементе
                    _WM_NOTIFY_DebugInfo("$EN_MAXTEXT", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; Это сообщение также отправляется, когда у edit-элемента нет стиля $ES_AUTOHSCROLL и число символов превышает доступную ширину
                    ; при вставке превысило бы ширину edit-элемента.
                    ; Это сообщение также отправляется, когда у edit-элемента нет стиля $ES_AUTOVSCROLL и общее число строк превышает видимую область
                    ; при вставке текста превысило бы высоту edit-элемента

                    ; нет возвращаемого значения
                Case $EN_SETFOCUS ; Отправляется при получении клавиатурного фокуса edit-элементом
                    _WM_NOTIFY_DebugInfo("$EN_SETFOCUS", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $EN_UPDATE ; Отправляется перед перерисовкой edit-элемента
                    _WM_NOTIFY_DebugInfo("$EN_UPDATE", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
                Case $EN_VSCROLL ; Отправляется при клике пользователя по вертикальной прокрутке edit-элемента или при прокрутке колеса мыши над ним
                    _WM_NOTIFY_DebugInfo("$EN_VSCROLL", "hWndFrom,IDFrom", $hWndFrom, $iIDFrom)
                    ; нет возвращаемого значения
            EndSwitch
    EndSwitch
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_COMMAND
