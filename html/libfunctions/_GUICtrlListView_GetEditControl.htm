Справочник функций

_GUICtrlListView_GetEditControl

Получает элемент.

#include <GuiListView.au3>
_GUICtrlListView_GetEditControl ( $hWnd )

Параметры

$hWnd Идентификатор или дескриптор элемента управления.

Возвращаемое значение

Успех: Возвращает дескриптор поля ввода.
Ошибка: Возвращает 0.

Примечания

Когда начинается редактирование надписи, создаётся, размещается и инициализируется поле ввода.
До того как оно станет видимым, элемент управления отправляет родительскому окну уведомление $LVN_BEGINLABELEDIT.

Чтобы настроить редактирование надписи, реализуйте обработчик $LVN_BEGINLABELEDITA или $LVN_BEGINLABELEDITAW и пусть он отправит элементу сообщение $LVM_GETEDITCONTROL.
Пока надпись редактируется, возвращаемым значением будет дескриптор поля ввода.
Используйте этот дескриптор, чтобы настроить поле ввода обычными сообщениями EM_XXX.

Когда пользователь завершает или отменяет редактирование, поле ввода уничтожается, и его дескриптор перестаёт быть действительным.
Поле ввода можно подклассировать, но уничтожать его не следует.
Чтобы отменить редактирование, отправьте элементу $WM_CANCELMODE.

Редактируется текущий сфокусированный элемент.
Чтобы найти элемент по его состоянию, используйте сообщение $LVM_GETNEXTITEM.

Пример

#include "Extras\HelpFileInternals.au3"
#include "Extras\WM_NOTIFY.au3"

#include <GUIConstantsEx.au3>
#include <GuiImageList.au3>
#include <GuiListView.au3>
#include <StructureConstants.au3>
#include <WindowsNotifsConstants.au3>
#include <WindowsStylesConstants.au3>

Global $g_hListView

Example()

Func Example()
    Local $hGui = GUICreate("ListView Get Edit Control (v" & @AutoItVersion & ")", 400, 300, Default, Default, Default, $WS_EX_CLIENTEDGE)
    $g_hListView = _GUICtrlListView_Create($hGui, "", 2, 2, 394, 118, BitOR($LVS_EDITLABELS, $LVS_REPORT))
    _MemoCreate(2, 124, 396, 174, 0)
    _GUICtrlListView_SetExtendedListViewStyle($g_hListView, BitOR($LVS_EX_GRIDLINES, $LVS_EX_FULLROWSELECT, $LVS_EX_DOUBLEBUFFER))
    GUISetState(@SW_SHOW)

    ; Загрузить изображения
    Local $hImage = _GUIImageList_Create()
    _GUIImageList_Add($hImage, _GUICtrlListView_CreateSolidBitMap(GUICtrlGetHandle($g_hListView), 0xFF0000, 16, 16))
    _GUIImageList_Add($hImage, _GUICtrlListView_CreateSolidBitMap(GUICtrlGetHandle($g_hListView), 0x00FF00, 16, 16))
    _GUIImageList_Add($hImage, _GUICtrlListView_CreateSolidBitMap(GUICtrlGetHandle($g_hListView), 0x0000FF, 16, 16))
    _GUICtrlListView_SetImageList($g_hListView, $hImage, 1)

    ; Добавить столбцы
    _GUICtrlListView_AddColumn($g_hListView, "Column 1", 100)
    _GUICtrlListView_AddColumn($g_hListView, "Column 2", 100)
    _GUICtrlListView_AddColumn($g_hListView, "Column 3", 100)

    ; Добавить элементы
    _GUICtrlListView_AddItem($g_hListView, "Row 1: Col 1", 0)
    _GUICtrlListView_AddSubItem($g_hListView, 0, "Row 1: Col 2", 1)
    _GUICtrlListView_AddSubItem($g_hListView, 0, "Row 1: Col 3", 2)
    _GUICtrlListView_AddItem($g_hListView, "Row 2: Col 1", 1)
    _GUICtrlListView_AddSubItem($g_hListView, 1, "Row 2: Col 2", 1)
    _GUICtrlListView_AddItem($g_hListView, "Row 3: Col 1", 2)

    ; Отслеживать сообщение WM_NOTIFY.
    _WM_NOTIFY_Register($_g_idLst_Memo)

    _GUICtrlListView_EditLabel($g_hListView, 0)
    _MemoWrite("Edit Handle: 0x" & Hex(_GUICtrlListView_GetEditControl($g_hListView)) & _
            " IsPtr = " & IsPtr(_GUICtrlListView_GetEditControl($g_hListView)) & " IsHWnd = " & IsHWnd(_GUICtrlListView_GetEditControl($g_hListView)))

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
    GUIDelete()
EndFunc   ;==>Example

Func WM_NOTIFY($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $wParam
    Local $hWndListView = $g_hListView
    If Not IsHWnd($g_hListView) Then $hWndListView = GUICtrlGetHandle($g_hListView)

    Local $tNMHDR = DllStructCreate($tagNMHDR, $lParam)
    Local $hWndFrom = HWnd(DllStructGetData($tNMHDR, "hWndFrom"))
    Local $iCode = DllStructGetData($tNMHDR, "Code")
    Switch $hWndFrom
        Case $hWndListView
            Switch $iCode
                Case $LVN_BEGINLABELEDITA, $LVN_BEGINLABELEDITW ; Начало редактирования подписи элемента
                    _WM_NOTIFY_DebugEvent("$LVN_BEGINLABELEDIT", $tagNMLVDISPINFO, $lParam, "IDFrom,,Mask,Item,SubItem,State,StateMask,Image,Param,Indent,,GroupID,Columns,pColumns")
                    Return False ; Позволить пользователю редактировать метку
                    ; Return True ; Запретить пользователю редактировать подпись
                Case $LVN_COLUMNCLICK ; Был кликнут столбец
                    _WM_NOTIFY_DebugEvent("$LVN_COLUMNCLICK", $tagNMLISTVIEW, $lParam, "IDFrom,,Item,SubItem,NewState,OldState,Changed,ActionX,ActionY,Param")
                    ; Нет возвращаемого значения
                Case $LVN_ENDLABELEDITA, $LVN_ENDLABELEDITW ; Завершение редактирования подписи элемента
                    Local $tInfo = DllStructCreate($tagNMLVDISPINFO, $lParam)
                    Local $pText = DllStructGetData($tInfo, "Text")
                    If $pText <> 0 Then
                        _WM_NOTIFY_DebugEvent("$LVN_ENDLABELEDIT", $tagNMLVDISPINFO, $lParam, "IDFrom,,Mask,Item,SubItem,State,StateMask,Image,Param,Indent,,GroupID,Columns,pColumns,,TextMax,*Text")
                        Return True ; Если Text пустой, возвращаемое значение игнорируется
                    EndIf
                Case $NM_CLICK ; Отправляется элементом list-view, когда пользователь кликает по элементу левой кнопкой мыши
                    _WM_NOTIFY_DebugEvent("$NM_CLICK", $tagNMITEMACTIVATE, $lParam, "IDFrom,,Index,SubItem,NewState,OldState,Changed,X,Y,,lParam,KeyFlags")
                    ; Нет возвращаемого значения
                Case $NM_DBLCLK ; Отправляется элементом list-view при двойном клике левой кнопкой мыши
                    _WM_NOTIFY_DebugEvent("$NM_DBLCLK", $tagNMITEMACTIVATE, $lParam, "IDFrom,,Index,SubItem,NewState,OldState,Changed,X,Y,,lParam,KeyFlags")
                    ; Нет возвращаемого значения
                Case $NM_KILLFOCUS ; Элемент управления потерял фокус ввода
                    _WM_NOTIFY_DebugEvent("$NM_KILLFOCUS", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Нет возвращаемого значения
                Case $NM_RCLICK ; Отправляется элементом list-view при клике правой кнопкой мыши
                    _WM_NOTIFY_DebugEvent("$NM_RCLICK", $tagNMITEMACTIVATE, $lParam, "IDFrom,,Index,SubItem,NewState,OldState,Changed,X,Y,,lParam,KeyFlags")
                    ;Return 1 ; чтобы запретить стандартную обработку
                    Return 0 ; разрешить стандартную обработку
                Case $NM_RDBLCLK ; Отправляется элементом list-view при двойном клике правой кнопкой мыши
                    _WM_NOTIFY_DebugEvent("$NM_RDBLCLK", $tagNMITEMACTIVATE, $lParam, "IDFrom,,Index,SubItem,NewState,OldState,Changed,X,Y,,lParam,KeyFlags")
                    ; Нет возвращаемого значения
                Case $NM_RETURN ; Элемент управления в фокусе и пользователь нажал ENTER
                    _WM_NOTIFY_DebugEvent("$NM_RETURN", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Нет возвращаемого значения
                Case $NM_SETFOCUS ; Элемент управления получил фокус ввода
                    _WM_NOTIFY_DebugEvent("$NM_SETFOCUS", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    ; Нет возвращаемого значения
            EndSwitch
    EndSwitch
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_NOTIFY
