_GUICtrlRichEdit_SetParaShading

Устанавливает затенение абзаца(ев) в текущем выделении или, если выделения нет, абзацев, вставленных в точку вставки.

#include <GuiRichEdit.au3>
_GUICtrlRichEdit_SetParaShading ( $hWnd [, $iWeight = Default [, $sStyle = Default [, $sForeColor = Default [, $sBackColor = Default]]]] )

Параметры

$hWnd Дескриптор элемента управления.
$iWeight [необязательный] Процент цвета переднего плана, остальное – цвет фона
$sStyle [необязательный] Стиль штриховки – одно из значений:
non – нет
dhz – тёмные горизонтальные линии
dvt – тёмные вертикальные линии
ddd – тёмная диагональ вниз
dud – тёмная диагональ вверх
dgr – тёмная сетка
dtr – тёмная решётка
lhz – светлые горизонтальные линии
lvt – светлые вертикальные линии
ldd – светлая диагональ вниз
lud – светлая диагональ вверх
lgr – светлая сетка
ltr – светлая решётка
$sForeColor [необязательный] Одно из:
"blk" – чёрный (начальное значение)
"blu" – синий
"cyn" – голубой
"grn" – зелёный
"mag" – пурпурный
"red" – красный
"yel" – жёлтый
"whi" – белый
"dbl" – тёмно-синий
"dgn" – тёмно-зелёный
"dmg" – тёмно-пурпурный
"drd" – тёмно-красный
"dyl" – тёмно-жёлтый
"dgy" – тёмно-серый
"lgy" – светло-серый
$sBackColor [необязательный] Те же значения, что у $sForeColor

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False и устанавливает @error ≠ 0.
@error: 101 – $hWnd не является дескриптором
103 – недопустимое значение $sStyle
104 – недопустимое значение $sForeColor
105 – недопустимое значение $sBackColor
1021 – $iWeight не является положительным числом
1022 – недопустимое значение $iWeight
700 – операция завершилась неудачно

Примечания

Если текст выделен, значения по умолчанию берутся из первого абзаца с выделенным текстом.

Пример

#include <GUIConstantsEx.au3>
#include <GuiRichEdit.au3>
#include <WindowsStylesConstants.au3>

Global $g_idLbl_Msg, $g_hRichEdit

Example()

Func Example()
    Local $hGui, $iMsg, $idBtn_Next, $iStep = 0
    $hGui = GUICreate("Example (" & StringTrimRight(@ScriptName, StringLen(".exe")) & ")", 320, 350, -1, -1)
    $g_hRichEdit = _GUICtrlRichEdit_Create($hGui, "This is a test.", 10, 10, 300, 220, _
            BitOR($ES_MULTILINE, $WS_VSCROLL, $ES_AUTOVSCROLL))
    $g_idLbl_Msg = GUICtrlCreateLabel("", 10, 235, 300, 60)
    $idBtn_Next = GUICtrlCreateButton("Next", 270, 310, 40, 30)
    GUISetState(@SW_SHOW)

    _GUICtrlRichEdit_SetText($g_hRichEdit, "First paragraph")
    Report("0. Para with default settings")

    While True
        $iMsg = GUIGetMsg()
        Select
            Case $iMsg = $GUI_EVENT_CLOSE
                _GUICtrlRichEdit_Destroy($g_hRichEdit) ; нужно, если только скрипт не упадёт
                ; GUIDelete() ; так тоже работает
                Exit
            Case $iMsg = $idBtn_Next
                $iStep += 1
                Switch $iStep
                    Case 1
                        _GUICtrlRichEdit_AppendText($g_hRichEdit, @CRLF & "Second paragraph")
                        _GUICtrlRichEdit_SetParaShading($g_hRichEdit, 60, "ddd", "blu", "dmg")
                        Report("1. Shading of second paragraph is ")
                    Case 2
                        _GUICtrlRichEdit_SetSel($g_hRichEdit, 0, 2)
                        Report("2. Style of first paragraph in the selection is ")
                    Case 3
                        _GUICtrlRichEdit_SetSel($g_hRichEdit, 10, 26)
                        _GUICtrlRichEdit_SetParaShading($g_hRichEdit, Default, "dgr")
                        Report("3. Change shading of both paragraphs")
                    Case 4
                        ; Передать весь текст на Рабочий стол – можно посмотреть настройки в Word
                        _GUICtrlRichEdit_Deselect($g_hRichEdit)
                        _GUICtrlRichEdit_StreamToFile($g_hRichEdit, @DesktopDir & "\gcre.rtf")
                        GUICtrlSetState($idBtn_Next, $GUI_DISABLE)
                EndSwitch
        EndSelect
    WEnd
EndFunc   ;==>Example

Func Report($sMsg)
    $sMsg = $sMsg & @CRLF & @CRLF & "Get function returns " & @CRLF & _GUICtrlRichEdit_GetParaShading($g_hRichEdit)
    GUICtrlSetData($g_idLbl_Msg, $sMsg)
EndFunc   ;==>Report
