Справочник функций

_Word_DocFind

Запускает или повторяет указанную операцию поиска.

#include <Word.au3>
_Word_DocFind ( $oDoc [, $sFindText = "" [, $vSearchRange = 0 [, $oFindRange = Default [, $bForward = True [, $bMatchCase = False [, $bMatchWholeWord = False [, $bMatchWildcards = False [, $bMatchSoundsLike = False [, $bMatchAllWordForms = False [, $bFormat = False]]]]]]]]]] )

Параметры

$oDoc Объект документа Word.
$sFindText [необязательный] Текст для поиска. Используйте пустую строку ("") для поиска только форматирования. Вы можете искать специальные символы, указав соответствующие коды символов. Например, "^p" соответствует отметке абзаца и "^t" соответствует символу табуляции (по умолчанию = "")
$vSearchRange [необязательный] Указывает выделение или диапазон для поиска. Может быть:     -1 – Указывает текущее выделение   0 – Указывает весь документ (по умолчанию) Любой объект диапазона Microsoft Word
$oFindRange [необязательный] Указывает диапазон, возвращаемый последним вызовом _Word_DocFind(). Это требуется, если вы хотите найти следующее или предыдущее вхождение $sFindText. Если установлено значение Default, поиск начинается с начала $vSearchRange (по умолчанию)
$bForward [необязательный] True для поиска вперёд (к концу документа) (по умолчанию = True)
$bMatchCase [необязательный] Если True, поиск чувствителен к регистру (по умолчанию = False)
$bMatchWholeWord [необязательный] Если True, искать только целые слова (по умолчанию = False)
$bMatchWildcards [необязательный] Если True, текст поиска имеет специальные операторы поиска (по умолчанию = False)
$bMatchSoundsLike [необязательный] Если True, найти слова, которые звучат похоже на текст поиска (по умолчанию = False)
$bMatchAllWordForms [необязательный] Если True, найти все формы текста поиска (например, "sit" находит "sitting" и "sat") (по умолчанию = False)
$bFormat [необязательный] True для поиска форматирования в дополнение к или вместо текста поиска (по умолчанию = False)

Возвращаемое значение

Успех: Возвращает объект диапазона для найденного текста.
Ошибка: Возвращает 0 и устанавливает @error ≠ 0.
@error: 1 – $oDoc не является объектом 2 – $vSearchRange не является объектом 3 – $oFindRange не является объектом 4 – $sFindText не найден. @extended задан код ошибки COM

Пример

Пример 1

#include "Extras\HelpFileInternals.au3"

#include <MsgBoxConstants.au3>
#include <Word.au3>

; Создать объект приложения
Local $oWord = _Word_Create()
If @error Then Exit MsgBox($MB_SYSTEMMODAL, "Word UDF: _Word_DocFind Example", _
        "Error creating the Word application object." & @CRLF & "@error = " & @error & ", @extended = " & @extended)
; Открыть тестовый документ только для чтения
Local $oDoc = _Word_DocOpen($oWord, _Extras_PathFull("Test.doc"), Default, Default, True)
If @error Then Exit MsgBox($MB_SYSTEMMODAL, "Word UDF: _Word_DocFind Example", _
        "Error opening '.\Extras\Test.doc'." & @CRLF & "@error = " & @error & ", @extended = " & @extended)

; Найти последний текст "test document" в документе и выделить его полужирным.
Local $oRangeFound
#forceref $oRangeFound
$oRangeFound = _Word_DocFind($oDoc, "test document", 0, Default, False)
If @error Then Exit MsgBox($MB_SYSTEMMODAL, "Word UDF: _Word_DocFind Example", _
        "Error locating the specified text in the document." & @CRLF & "@error = " & @error & ", @extended = " & @extended)
$oRangeFound.Bold = True
MsgBox($MB_SYSTEMMODAL, "Word UDF: _Word_DocFind Example", _
        "Last occurrence of string 'test document' in the document marked as bold.")

Пример 2

#include "Extras\HelpFileInternals.au3"

#include <MsgBoxConstants.au3>
#include <Word.au3>

; Создать объект приложения
Local $oWord = _Word_Create()
If @error Then Exit MsgBox($MB_SYSTEMMODAL, "Word UDF: _Word_DocFind Example", _
        "Error creating the Word application object." & @CRLF & "@error = " & @error & ", @extended = " & @extended)
; Открыть тестовый документ только для чтения
Local $oDoc = _Word_DocOpen($oWord, _Extras_PathFull("Test.doc"), Default, Default, True)
If @error Then Exit MsgBox($MB_SYSTEMMODAL, "Word UDF: _Word_DocFind Example", _
        "Error opening '.\Extras\Test.doc'." & @CRLF & "@error = " & @error & ", @extended = " & @extended)

; Подчеркнуть текст "line" в строках 2-4.
Local $oRangeFound, $oSearchRange
$oSearchRange = _Word_DocRangeSet($oDoc, -1, $wdParagraph, 1, $wdParagraph, 3)
$oRangeFound = _Word_DocFind($oDoc, "line", $oSearchRange)
If @error Then Exit MsgBox($MB_SYSTEMMODAL, "Word UDF: _Word_DocFind Example", _
        "Error locating the specified text in the document." & @CRLF & "@error = " & @error & ", @extended = " & @extended)
$oRangeFound.Underline = True
While 1
    $oRangeFound = _Word_DocFind($oDoc, "line", $oSearchRange, $oRangeFound)
    If @error Then ExitLoop
    $oRangeFound.Underline = True
WEnd
MsgBox($MB_SYSTEMMODAL, "Word UDF: _Word_DocFind Example", _
        "All occurrences of string 'line' in paragraphs 2-4 marked as underlined.")

Пример 3

#include "Extras\HelpFileInternals.au3"

#include <MsgBoxConstants.au3>
#include <Word.au3>

; Создать объект приложения
Local $oWord = _Word_Create()
If @error Then Exit MsgBox($MB_SYSTEMMODAL, "Word UDF: _Word_DocFind Example", _
        "Error creating the Word application object." & @CRLF & "@error = " & @error & ", @extended = " & @extended)
; Открыть тестовый документ только для чтения
Local $oDoc = _Word_DocOpen($oWord, _Extras_PathFull("Test.doc"), Default, Default, True)
If @error Then Exit MsgBox($MB_SYSTEMMODAL, "Word UDF: _Word_DocFind Example", _
        "Error opening '.\Extras\Test.doc'." & @CRLF & "@error = " & @error & ", @extended = " & @extended)

; Поиск текста "line" в строках 2-4 и выделение следующего числа полужирным.
; Если число "3", вставить текст перед текстом "line".
Local $oRangeFound, $oSearchRange, $oRangeText
$oSearchRange = _Word_DocRangeSet($oDoc, -1, $wdParagraph, 1, $wdParagraph, 3)
$oRangeFound = _Word_DocFind($oDoc, "line", $oSearchRange)
If @error Then Exit MsgBox($MB_SYSTEMMODAL, "Word UDF: _Word_DocFind Example", _
        "Error locating the specified text in the document." & @CRLF & "@error = " & @error & ", @extended = " & @extended)
; Выделить номер строки полужирным. Создать новый диапазон, смещённый на одно слово вправо
$oRangeText = _Word_DocRangeSet($oDoc, $oRangeFound, Default, 1, Default, 1)
$oRangeText.Bold = True
While 1
    $oRangeFound = _Word_DocFind($oDoc, "line", $oSearchRange, $oRangeFound)
    If @error Then ExitLoop
    ; Выделить номер строки полужирным. Создать новый диапазон (дублировать, чтобы не изменить результат операции поиска) и переместить его на одно слово вправо
    $oRangeText = $oRangeFound.Duplicate
    $oRangeText = _Word_DocRangeSet($oDoc, $oRangeText, Default, 1, Default, 1)
    $oRangeText.Bold = True
    ; Вставить текст, если номер строки = 3
    If StringStripWS($oRangeText.Text, 8) = "3" Then $oRangeFound.InsertBefore("(maybe) ")
WEnd
MsgBox($MB_SYSTEMMODAL, "Word UDF: _Word_DocFind Example", _
        "All line numbers in paragraphs 2-4 marked as bold." & @CRLF & "Text inserted before 'line 3'.")

#include "Extras\HelpFileInternals.au3"

#include <MsgBoxConstants.au3>
#include <Word.au3>

; Создать объект приложения
Local $oWord = _Word_Create()
If @error Then Exit MsgBox($MB_SYSTEMMODAL, "Word UDF: _Word_DocFind Example", _
        "Error creating the Word application object." & @CRLF & "@error = " & @error & ", @extended = " & @extended)
; Открыть тестовый документ только для чтения
Local $oDoc = _Word_DocOpen($oWord, _Extras_PathFull("Test.doc"), Default, Default, True)
If @error Then Exit MsgBox($MB_SYSTEMMODAL, "Word UDF: _Word_DocFind Example", _
        "Error opening '.\Extras\Test.doc'." & @CRLF & "@error = " & @error & ", @extended = " & @extended)

; Заменить управляющий символ абзаца на абзац + текст + абзац.
; Только изменить первое вхождение.
Local $oRangeFound
$oRangeFound = _Word_DocFind($oDoc, "^p")
If @error Then Exit MsgBox($MB_SYSTEMMODAL, "Word UDF: _Word_DocFind Example", _
        "Error locating paragraph control character in the document." & @CRLF & "@error = " & @error & ", @extended = " & @extended)
$oRangeFound.InsertAfter("[New Line] ")
If @error Then Exit MsgBox($MB_SYSTEMMODAL, "Word UDF: _Word_DocFind Example", _
        "Error inserting text after the paragraph control character in the document." & @CRLF & "@error = " & @error & _
        ", @extended = " & @extended)
MsgBox($MB_SYSTEMMODAL, "Word UDF: _Word_DocFind Example", "Paragraph control character successfully replaced." & @CRLF & _
        "Text inserted in paragraph 2.")
