Справочник функций

_WinAPI_AnimateWindow

Позволяет вам создавать специальные эффекты при отображении или скрытии окон.

#include <WinAPISysWin.au3>
_WinAPI_AnimateWindow ( $hWnd, $iFlags [, $iDuration = 1000] )

Параметры

$hWnd Дескриптор окна.
$iFlags Флаги, которые определяют тип анимации.
Этот параметр может быть одним или несколькими из следующих значений. Учтите, что по умолчанию эти флаги вступают в силу при отображении окна.
Чтобы они вступили в силу при скрытии окна, используйте AW_HIDE и логический оператор OR с соответствующими флагами:
$AW_ACTIVATE
$AW_BLEND
$AW_CENTER
$AW_HIDE
$AW_HOR_NEGATIVE
$AW_HOR_POSITIVE
$AW_SLIDE
$AW_VER_NEGATIVE
$AW_VER_POSITIVE
$iDuration [необязательный] Определяет, сколько времени занимает воспроизведение анимации в миллисекундах. По умолчанию 1000.

Возвращаемое значение

Успех: Возвращает True.
Ошибка: Возвращает False; вызовите _WinAPI_GetLastError() для получения подробных сведений об ошибке.

Дополнительно

Искать AnimateWindow в библиотеке MSDN

Пример

#NoTrayIcon

#include <GUIConstantsEx.au3>
#include <MenuConstants.au3>
#include <TrayConstants.au3>
#include <WinAPIGdi.au3>
#include <WinAPISysWin.au3>
#include <WindowsNotifsConstants.au3>

Opt('TrayMenuMode', 1)

Local $hTray = ControlGetHandle('[CLASS:Shell_TrayWnd]', '', 'TrayNotifyWnd1')

Local $idTrayRestoreItem = TrayCreateItem('Restore')
TrayItemSetState(-1, $TRAY_DEFAULT)
TrayCreateItem('')
Local $idTrayExitItem = TrayCreateItem('Exit')
TraySetClick(8)

Global $g_hForm = GUICreate('Test ' & StringReplace(@ScriptName, '.au3', '()'))
Global $g_idDummy = GUICtrlCreateDummy()
GUIRegisterMsg($WM_SYSCOMMAND, 'WM_SYSCOMMAND')
GUISetState(@SW_SHOW)

While 1
    Switch TrayGetMsg()
        Case $idTrayRestoreItem
            _WinAPI_DrawAnimatedRects($g_hForm, _WinAPI_GetWindowRect($hTray), _WinAPI_GetWindowRect($g_hForm))
            GUISetState(@SW_SHOW, $g_hForm)
            TraySetState(2)
        Case $idTrayExitItem
            ExitLoop
    EndSwitch
    Switch GUIGetMsg()
        Case $GUI_EVENT_CLOSE
            _WinAPI_AnimateWindow($g_hForm, BitOR($AW_BLEND, $AW_HIDE))
            ExitLoop
        Case $g_idDummy ; Свернуть
            _WinAPI_DrawAnimatedRects($g_hForm, _WinAPI_GetWindowRect($g_hForm), _WinAPI_GetWindowRect($hTray))
            GUISetState(@SW_HIDE, $g_hForm)
            TraySetState(1)
    EndSwitch
WEnd

Func WM_SYSCOMMAND($hWnd, $iMsg, $wParam, $lParam)
    #forceref $iMsg, $lParam

    Switch $hWnd
        Case $g_hForm
            Switch $wParam
                Case $SC_MINIMIZE
                    GUICtrlSendToDummy($g_idDummy)
                    Return 0
            EndSwitch
    EndSwitch
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_SYSCOMMAND
