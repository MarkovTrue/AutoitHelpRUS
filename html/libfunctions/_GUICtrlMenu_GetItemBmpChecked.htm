_GUICtrlMenu_GetItemBmpChecked

Получает растровое изображение, отображаемое при выборе пункта.

#include <GuiMenu.au3>
_GUICtrlMenu_GetItemBmpChecked ( $hMenu, $iItem [, $bByPos = True] )

Параметры

$hMenu Дескриптор меню.
$iItem Идентификатор или позиция пункта меню.
$bByPos [необязательный] Флаг идентификатора меню:
True – $iItem задаёт позицию элемента (начиная с нуля)
False – $iItem задаёт идентификатор элемента меню.

Возвращаемое значение

Возвращает дескриптор растрового изображения, отображаемого рядом с пунктом при его выборе.

См. также

_GUICtrlMenu_SetItemBmpChecked

Пример

#include "Extras\HelpFileInternals.au3"

#include <GUIConstantsEx.au3>
#include <GuiMenu.au3>
#include <WinAPIGdi.au3>

Example()

Func Example()
    Local $hGUI, $hFile, $hEdit, $hHelp, $hMain, $hBmp1, $hBmp2
    Global Enum $e_idNew = 1000, $e_idOpen, $e_idSave, $e_idExit, $e_idCut, $e_idCopy, $e_idPaste, $e_idAbout

    ; Создать GUI
    $hGUI = GUICreate("Menu", 400, 300)

    ; Создать меню Файл
    $hFile = _GUICtrlMenu_CreateMenu()
    _GUICtrlMenu_InsertMenuItem($hFile, 0, "&New", $e_idNew)
    _GUICtrlMenu_InsertMenuItem($hFile, 1, "&Open", $e_idOpen)
    _GUICtrlMenu_InsertMenuItem($hFile, 2, "&Save", $e_idSave)
    _GUICtrlMenu_InsertMenuItem($hFile, 3, "", 0)
    _GUICtrlMenu_InsertMenuItem($hFile, 4, "E&xit", $e_idExit)

    ; Создать меню Правка
    $hEdit = _GUICtrlMenu_CreateMenu()
    _GUICtrlMenu_InsertMenuItem($hEdit, 0, "&Cut", $e_idCut)
    _GUICtrlMenu_InsertMenuItem($hEdit, 1, "C&opy", $e_idCopy)
    _GUICtrlMenu_InsertMenuItem($hEdit, 2, "&Paste", $e_idPaste)

    ; Создать меню Справка
    $hHelp = _GUICtrlMenu_CreateMenu()
    _GUICtrlMenu_InsertMenuItem($hHelp, 0, "&About", $e_idAbout)

    ; Создать главное меню
    $hMain = _GUICtrlMenu_CreateMenu()
    _GUICtrlMenu_InsertMenuItem($hMain, 0, "&File", 0, $hFile)
    _GUICtrlMenu_InsertMenuItem($hMain, 1, "&Edit", 0, $hEdit)
    _GUICtrlMenu_InsertMenuItem($hMain, 2, "&Help", 0, $hHelp)

    ; Установить меню окна
    _GUICtrlMenu_SetMenu($hGUI, $hMain)

    ; Создать элемент memo
    _MemoCreate(2, 2, 396, 276, 0)
    GUISetState(@SW_SHOW)

    ; Задать отмеченные/неотмеченные bitmap-ы в меню Файл
    $hBmp1 = _WinAPI_CreateSolidBitmap($hGUI, 0xFF0000, 11, 11)
    $hBmp2 = _WinAPI_CreateSolidBitmap($hGUI, 0x00FF00, 11, 11)
    _GUICtrlMenu_SetItemBmpChecked($hFile, 0, $hBmp1)
    _GUICtrlMenu_SetItemBmpChecked($hFile, 1, $hBmp1)
    _GUICtrlMenu_SetItemBmpChecked($hFile, 2, $hBmp1)
    _GUICtrlMenu_SetItemBmpUnchecked($hFile, 0, $hBmp2)
    _GUICtrlMenu_SetItemBmpUnchecked($hFile, 1, $hBmp2)
    _GUICtrlMenu_SetItemBmpUnchecked($hFile, 2, $hBmp2)

    ; Отметить пункт меню Открыть
    _GUICtrlMenu_CheckMenuItem($hFile, 1)

    ; Показать, что дескрипторы bitmap совпадают
    _MemoWrite("Checked handle ......: 0x" & Hex($hBmp1))
    _MemoWrite("Unchecked handle ....: 0x" & Hex($hBmp2))
    _MemoWrite("Open checked handle .: 0x" & Hex(_GUICtrlMenu_GetItemBmpChecked($hFile, 0)))
    _MemoWrite("Open unchecked handle: 0x" & Hex(_GUICtrlMenu_GetItemBmpUnchecked($hFile, 0)))

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
EndFunc   ;==>Example
