_GUICtrlComboBoxEx_Create

Создаёт элемент управления ComboBoxEx.

#include <GuiComboBoxEx.au3>
_GUICtrlComboBoxEx_Create ( $hWnd, $sText, $iX, $iY [, $iWidth = 100 [, $iHeight = 200 [, $iStyle = 0x00200002 [, $iExStyle = 0x00000000]]]] )

Параметры

$hWnd Дескриптор родительского или владеющего окна.
$sText Текст с разделителями для добавления в ComboBox.
$iX Горизонтальная позиция элемента управления.
$iY Вертикальная позиция элемента управления.
$iWidth [необязательный] Ширина элемента управления
$iHeight [необязательный] Высота элемента управления
$iStyle [необязательный] Стиль элемента управления:
$CBS_DROPDOWN – похоже на $CBS_SIMPLE, кроме того, что список не отображается, пока пользователь не выберет значок рядом с edit элементом управления
$CBS_DROPDOWNLIST – похоже на $CBS_DROPDOWN, кроме того, что edit элемент заменяется статическим текстовым элементом, отображающим текущее выделение из списка
$CBS_SIMPLE – отображает список всё время

По умолчанию: $CBS_DROPDOWN, $WS_VSCROLL
Обязательно: $WS_CHILD, $WS_TABSTOP, $WS_VISIBLE.
$iExStyle [необязательный] Расширенный стиль элемента управления:
$CBES_EX_CASESENSITIVE – поиск в списке будет чувствителен к регистру
$CBES_EX_NOEDITIMAGE – edit поле и выпадающий список не будут отображать значки элементов
$CBES_EX_NOEDITIMAGEINDENT – edit поле и выпадающий список не будут отображать значки элементов
$CBES_EX_NOSIZELIMIT – позволяет изменять размер элемента управления ComboBoxEx вертикально меньше, чем его содержащийся ComboBox.

Возвращаемое значение

Успех: Возвращает обработчик элемента управления ComboBoxEx.
Ошибка: Возвращает 0.

См. также

_GUICtrlComboBoxEx_Destroy

Пример

#include "Extras\WM_NOTIFY.au3"

#include <GuiComboBoxEx.au3>
#include <GUIConstantsEx.au3>
#include <StructureConstants.au3>

Global $g_hCombo

Example()

Func Example()
    ; Создать GUI
    Local $hGUI = GUICreate("ComboBoxEx Create (v" & @AutoItVersion & ")", 400, 300)
    $g_hCombo = _GUICtrlComboBoxEx_Create($hGUI, "This is a test|Line 2", 2, 2, 394, 268)
    GUISetState(@SW_SHOW)

    _WM_NOTIFY_Register()

    _GUICtrlComboBoxEx_AddString($g_hCombo, "Some More Text")
    _GUICtrlComboBoxEx_InsertString($g_hCombo, "Text to be deleted", 1)
    _GUICtrlComboBoxEx_DeleteString($g_hCombo, 1)
    _GUICtrlComboBoxEx_InsertString($g_hCombo, "Inserted Text", 1)

    ; Цикл до выхода пользователя.
    Do
    Until GUIGetMsg() = $GUI_EVENT_CLOSE
EndFunc   ;==>Example

Func WM_NOTIFY($hWnd, $iMsg, $wParam, $lParam)
    #forceref $hWnd, $iMsg, $wParam
    Local $tNMHDR = DllStructCreate($tagNMHDR, $lParam)
    Local $hWndFrom = HWnd(DllStructGetData($tNMHDR, "hWndFrom"))
    Local $iCode = DllStructGetData($tNMHDR, "Code")
    Local $tInfo
    Switch $hWndFrom
        Case $g_hCombo
            Switch $iCode
                Case $CBEN_BEGINEDIT ; Отправляется когда пользователь активирует выпадающий список или нажимает в edit элемента управления.
                    _WM_NOTIFY_DebugEvent("$CBEN_BEGINEDIT", $tagNMHDR, $lParam, "hWndFrom,IDFrom")
                    Return 0
                Case $CBEN_DELETEITEM
                    _WM_NOTIFY_DebugEvent("$CBEN_DELETEITEM", $tagNMCOMBOBOXEX, $lParam, "IDFrom,,Mask,Item,*Text,TextMax,Indent,Image,SelectedImage,OverlayImage,Param")
                    Return 0
                Case $CBEN_DRAGBEGINA, $CBEN_DRAGBEGINW
                    $tInfo = DllStructCreate($tagNMCBEDRAGBEGIN, $lParam)
                    If DllStructGetData($tInfo, "ItemID") Then
                        _WM_NOTIFY_DebugEvent("$CBEN_DRAGBEGIN", $tagNMCOMBOBOXEX, $lParam, "IDFrom,,Mask,Item,*Text,TextMax,Indent,Image,SelectedImage,OverlayImage,Param")
                    EndIf
                    _WM_NOTIFY_DebugEvent("$CBEN_DRAGBEGIN", $tagNMCBEDRAGBEGIN, $lParam, "IDFrom,,ItemID,Text")
                    ; возвращаемое значение игнорируется
                Case $CBEN_ENDEDITA, $CBEN_ENDEDITW ; Отправляется когда пользователь завершает операцию в edit или выбирает элемент из выпадающего списка элемента управления.
                    _WM_NOTIFY_DebugEvent("$CBEN_ENDEDIT", $tagNMCBEENDEDIT, $lParam, "IDFrom,,fChanged,NewSelection,Text,Why")
                    Return False ; принять уведомление и разрешить элементу управления отображать выбранный элемент
                    ; Возвращает True или иное
                Case $CBEN_GETDISPINFOA, $CBEN_GETDISPINFOW ; Отправляется для получения информации отображения элемента с обратным вызовом
                    _WM_NOTIFY_DebugEvent("$CBEN_GETDISPINFOW", $tagNMCOMBOBOXEX, $lParam, "IDFrom,,Mask,Item,*Text,TextMax,Indent,Image,SelectedImage,OverlayImage,Param")
                    Return 0
                Case $CBEN_INSERTITEM
                    _WM_NOTIFY_DebugEvent("$CBEN_INSERTITEM", $tagNMCOMBOBOXEX, $lParam, "IDFrom,,Mask,Item,*Text,TextMax,Indent,Image,SelectedImage,OverlayImage,Param")
                    Return 0
            EndSwitch
    EndSwitch
    Return $GUI_RUNDEFMSG
EndFunc   ;==>WM_NOTIFY
